44  void VisualizeForwardTrajectories(const ros::Time& stamp) {
45  auto forward_trajs = p_bp_manager_->planner().forward_trajs();
46  int processed_winner_id = p_bp_manager_->processed_winner_id();
47  int original_winner_id = p_bp_manager_->original_winner_id();
48  visualization_msgs::MarkerArray traj_list_marker;
49  common::ColorARGB traj_color(0.5, 0.5, 0.5, 0.5);
50  double traj_z = 0.3;
51  for (int i = 0; i < static_cast<int>(forward_trajs.size()); ++i) {
52  if (i == processed_winner_id) {
53  traj_color = common::cmap.at("gold");
54  traj_z = 0.4;
55  } else if (i == original_winner_id) {
56  traj_color = common::cmap.at("spring green");
57  traj_z = 0.4;
58  } else {
59  traj_color = common::ColorARGB(0.5, 0.5, 0.5, 0.5);
60  traj_z = 0.3;
61  }
62  std::vector<common::Point> points;
63  for (const auto& v : forward_trajs[i]) {
64  common::Point pt(v.state().vec_position(0), v.state().vec_position(1));
65  // pt.z = v.state().time_stamp -
66  // forward_trajs[i].front().state().time_stamp;
67  pt.z = traj_z;
68  points.push_back(pt);
69  visualization_msgs::Marker point_marker;
70  // point_marker.ns = "point";
71  common::VisualizationUtil::GetRosMarkerCylinderUsingPoint(
72  common::Point(pt), Vec3f(0.5, 0.5, 0.1), traj_color, 0,
73  &point_marker);
74  traj_list_marker.markers.push_back(point_marker);
75  }
76  visualization_msgs::Marker line_marker;
77  // line_marker.ns = "line";
78  common::VisualizationUtil::GetRosMarkerLineStripUsingPoints(
79  points, Vec3f(0.1, 0.1, 0.1), traj_color, 0, &line_marker);
80  traj_list_marker.markers.push_back(line_marker);
81  }
82  int num_markers = static_cast<int>(traj_list_marker.markers.size());
83  common::VisualizationUtil::FillHeaderIdInMarkerArray(
84  stamp, std::string("/map"), last_forward_trajs_marker_cnt_,
85  &traj_list_marker);
86  last_forward_trajs_marker_cnt_ = num_markers;
87  forward_traj_vis_pub_.publish(traj_list_marker);
88  }
