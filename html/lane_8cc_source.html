6  decimal_t* curvature,
7  decimal_t* curvature_derivative) const {
8  if (CheckInputArcLength(arc_length) != kSuccess || LaneDim != 2) {
9  return kIllegalInput;
10  }
11 
12  Vecf<LaneDim> vel, acc, jrk;
13  GetDerivativeByArcLength(arc_length, 1, &vel);
14  // printf("vel: (%lf, %lf) at arc length %lf.\n", vel[0], vel[1], arc_length);
15  GetDerivativeByArcLength(arc_length, 2, &acc);
16  // printf("acc: (%lf, %lf) at arc length %lf.\n", acc[0], acc[1], arc_length);

18  // printf("jrk: (%lf, %lf) at arc length %lf.\n", jrk[0], jrk[1], arc_length);
19 
20  decimal_t c0 = vel[0] * acc[1] - vel[1] * acc[0];
21  decimal_t c1 = vel.norm();
22  *curvature = c0 / (c1 * c1 * c1);
23  *curvature_derivative =
24  ((acc[0] * acc[1] + vel[0] * jrk[1] - acc[1] * acc[0] - vel[1] * jrk[0]) /
25  c1 * c1 * c1 -
26  3 * c0 * (vel[0] * acc[0] + vel[1] * acc[1]) / (c1 * c1 * c1 * c1 * c1));
27  return kSuccess;

110 ErrorType Lane::GetArcLengthByVecPosition(const Vecf<LaneDim>& vec_position,
111  decimal_t* arc_length) const {
112  if (!IsValid()) {
113  return kWrongStatus;
114  }
115 
116  static constexpr int kMaxCnt = 4;
117  static constexpr decimal_t kMaxDistSquare = 900.0;
118 
119  const decimal_t val_lb = position_spline_.begin();
120  const decimal_t val_ub = position_spline_.end();
121  decimal_t step = (val_ub - val_lb) * 0.5;
122 
123  decimal_t s1 = val_lb;
124  decimal_t s2 = val_lb + step;
125  decimal_t s3 = val_ub;
126  decimal_t initial_guess = s2;
127 
128  // printf("[XXX]s1 = %lf, s2 = %lf, s3 = %lf\n", s1, s2, s3);
129 
130  Vecf<LaneDim> start_pos, mid_pos, final_pos;
131  position_spline_.evaluate(s1, &start_pos);
132  position_spline_.evaluate(s2, &mid_pos);
133  position_spline_.evaluate(s3, &final_pos);
134 
135  // ~ Step I: use binary search to find a initial guess
136  decimal_t d1 = (start_pos - vec_position).squaredNorm();
137  decimal_t d2 = (mid_pos - vec_position).squaredNorm();
138  decimal_t d3 = (final_pos - vec_position).squaredNorm();
139 
140  for (int i = 0; i < kMaxCnt; ++i) {
141  // printf("[XXX] - it = %d - s1 = %lf, s2 = %lf, s3 = %lf\n", i, s1, s2,
142  // s3);
143  decimal_t min_dis = std::min(std::min(d1, d2), d3);
144  if (min_dis < kMaxDistSquare) {
145  if (min_dis == d1) {
146  initial_guess = s1;
147  } else if (min_dis == d2) {
148  initial_guess = s2;
149  } else if (min_dis == d3) {
150  initial_guess = s3;
151  } else {
152  assert(false);
153  }
154  break;
155  }
156  step *= 0.5;
157  if (min_dis == d1) {
158  initial_guess = s1;
159  s3 = s2;
160  s2 = s1 + step;
161  position_spline_.evaluate(s2, &mid_pos);
162  position_spline_.evaluate(s3, &final_pos);
163  d2 = (mid_pos - vec_position).squaredNorm();
164  d3 = (final_pos - vec_position).squaredNorm();
165  } else if (min_dis == d2) {
166  initial_guess = s2;
167  s1 = s2 - step;
168  s3 = s2 + step;
169  position_spline_.evaluate(s1, &start_pos);
170  position_spline_.evaluate(s3, &final_pos);
171  d1 = (start_pos - vec_position).squaredNorm();
172  d3 = (final_pos - vec_position).squaredNorm();
173  } else if (min_dis == d3) {
174  initial_guess = s3;
175  s1 = s2;
176  s2 = s3 - step;
177  position_spline_.evaluate(s1, &start_pos);
178  position_spline_.evaluate(s2, &mid_pos);
179  d1 = (start_pos - vec_position).squaredNorm();
180  d2 = (mid_pos - vec_position).squaredNorm();
181  } else {
182  printf(
183  "[Lane]GetArcLengthByVecPosition - d1: %lf, d2: %lf, d3: %lf, "
184  "min_dis: %lf\n",
185  d1, d2, d3, min_dis);
186  assert(false);
187  }
188  }
189 
190  // printf("[XXX]initial_guess = %lf\n", initial_guess);
191 
192  // ~ Step II: use Newton's method to find the local minimum
193  GetArcLengthByVecPositionWithInitialGuess(vec_position, initial_guess,
194  arc_length);
195 
196  return kSuccess;
197 }

199 ErrorType Lane::GetArcLengthByVecPositionWithInitialGuess(
200  const Vecf<LaneDim>& vec_position, const decimal_t& initial_guess,
201  decimal_t* arc_length) const {
202  if (!IsValid()) {
203  return kWrongStatus;
204  }
205 
206  const decimal_t val_lb = position_spline_.begin();
207  const decimal_t val_ub = position_spline_.end();
208 
209  // ~ use Newton's method to find the local minimum
210  static constexpr decimal_t epsilon = 1e-3;
211  static constexpr int kMaxIter = 8;
212  decimal_t x = std::min(std::max(initial_guess, val_lb), val_ub);
213  Vecf<LaneDim> p, dp, ddp, tmp_vec;
214 
215  for (int i = 0; i < kMaxIter; ++i) {
216  position_spline_.evaluate(x, 0, &p);
217  position_spline_.evaluate(x, 1, &dp);
218  position_spline_.evaluate(x, 2, &ddp);
219 
220  tmp_vec = p - vec_position;
221  double f_1 = tmp_vec.dot(dp);
222  double f_2 = dp.dot(dp) + tmp_vec.dot(ddp);
223  double dx = -f_1 / f_2;
224 
225  if (std::fabs(dx) < epsilon) {
226  break;
227  }
228 
229  if (x + dx > val_ub) {
230  x = val_ub;
231  // printf(
232  // "[Lane]GetArcLengthByVecPosition - Out of range, use upper
233  // bound\n");
234  break;
235  } else if (x + dx < val_lb) {
236  x = val_lb;
237  // printf(
238  // "[Lane]GetArcLengthByVecPosition - Out of range, use lower
239  // bound\n");
240  break;
241  }
242 
243  x += dx;
244  }
245 
246  *arc_length = x;
247 
248  return kSuccess;
249 }
