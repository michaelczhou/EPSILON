87  int num_pts = vec_domain_.size();
88  if (num_pts < 1) return kIllegalInput;
89 
90  auto it = std::lower_bound(vec_domain_.begin(), vec_domain_.end(), s);
91  int idx =
92  std::min(std::max(static_cast<int>(it - vec_domain_.begin()) - 1, 0),
93  num_pts - 2);
94  decimal_t h = s - vec_domain_[idx];
95  if (s < vec_domain_[0]) {
96  return kIllegalInput;
97  } else {
98  decimal_t duration = vec_domain_[idx + 1] - vec_domain_[idx];
99  decimal_t normalized_s = h / duration;
100  // printf("derivative %d, normalized s: %lf.\n", d, normalized_s);
101  Vecf<N_DEG + 1> basis =
102  BezierUtils<N_DEG>::GetBezierBasis(d, normalized_s);
103  Vecf<N_DIM> result =
104  (pow(duration, 1 - d) * basis.transpose() * ctrl_pts_[idx])
105  .transpose();
106  *ret = result;
107  }
108  return kSuccess;

157  static Vecf<N_DEG + 1> GetBezierBasis(int derivative_degree, decimal_t t) {
158  t = std::max(std::min(1.0, t), 0.0);
159  Vecf<N_DEG + 1> basis;
160  switch (N_DEG) {
161  case 5:
162  if (derivative_degree == 0) {
163  basis << -pow(t - 1, 5), 5 * t * pow(t - 1, 4),
164  -10 * pow(t, 2) * pow(t - 1, 3), 10 * pow(t, 3) * pow(t - 1, 2),
165  -5 * pow(t, 4) * (t - 1), pow(t, 5);
166  } else if (derivative_degree == 1) {
167  basis << -5 * pow(t - 1, 4),
168  20 * t * pow(t - 1, 3) + 5 * pow(t - 1, 4),
169  -20 * t * pow(t - 1, 3) - 30 * pow(t, 2) * pow(t - 1, 2),
170  10 * pow(t, 3) * (2 * t - 2) + 30 * pow(t, 2) * pow(t - 1, 2),
171  -20 * pow(t, 3) * (t - 1) - 5 * pow(t, 4), 5 * pow(t, 4);
172  } else if (derivative_degree == 2) {
173  basis << -20 * pow(t - 1, 3),
174  60 * t * pow(t - 1, 2) + 40 * pow(t - 1, 3),
175  -120 * t * pow(t - 1, 2) - 20 * pow(t - 1, 3) -
176  30 * t * t * (2 * t - 2),
177  60 * t * pow(t - 1, 2) + 60 * t * t * (2 * t - 2) +
178  20 * t * t * t,
179  -60 * t * t * (t - 1) - 40 * t * t * t, 20 * t * t * t;
180  } else if (derivative_degree == 3) {
