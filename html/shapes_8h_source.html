331  template <typename T, int N_DIM>
332  static bool CheckIfAxisAlignedCubeNdIntersect(
333  const AxisAlignedCubeNd<T, N_DIM>& aabb_a,
334  const AxisAlignedCubeNd<T, N_DIM>& aabb_b,
335  std::array<bool, N_DIM * 2>* inter_dim_a,
336  std::array<bool, N_DIM * 2>* inter_dim_b) {
337  inter_dim_a->fill(false);
338  inter_dim_b->fill(false);
339  // ~ A contains B or B contains A
340  if (CheckIfAxisAlignedCubeAContainsAxisAlignedCubeB(aabb_a, aabb_b) ||
341  CheckIfAxisAlignedCubeAContainsAxisAlignedCubeB(aabb_b, aabb_a)) {
342  return false;
343  }
344  // ~ Check if collide
345  if (!CheckIfAxisAlignedCubeCollide(aabb_a, aabb_b)) {
346  return false;
347  }
348  // ~ Calculate on which A's surface collision happened
349  for (int i = 0; i < N_DIM; ++i) {
350  if (CheckIfAxisAlignedCubeNdCollideOnOneDim(aabb_a, aabb_b, i)) {
351  if (aabb_a.upper_bound[i] < aabb_b.upper_bound[i] &&
352  aabb_a.upper_bound[i] > aabb_b.lower_bound[i]) {
353  (*inter_dim_a)[2 * i] = true;
354  }
355  if (aabb_a.lower_bound[i] < aabb_b.upper_bound[i] &&
356  aabb_a.lower_bound[i] > aabb_b.lower_bound[i]) {
357  (*inter_dim_a)[2 * i + 1] = true;
358  }
359  }
360  }
361  // ~ Calculate on which B's surface collision happened
362  for (int i = 0; i < N_DIM; ++i) {
363  if (CheckIfAxisAlignedCubeNdCollideOnOneDim(aabb_a, aabb_b, i)) {
364  if (aabb_b.upper_bound[i] > aabb_a.lower_bound[i] &&
365  aabb_b.upper_bound[i] < aabb_a.upper_bound[i]) {
366  (*inter_dim_b)[2 * i] = true;
367  }
368  if (aabb_b.lower_bound[i] > aabb_a.lower_bound[i] &&
369  aabb_b.lower_bound[i] < aabb_a.upper_bound[i]) {
370  (*inter_dim_b)[2 * i + 1] = true;
371  }
372  }
373  }
374  return true;
375  }
