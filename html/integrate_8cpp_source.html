60 void lorenz( const state_type &x , state_type &dxdt , const value_type t )
61 {
62  //const value_type sigma( 10.0 );
63  const value_type R( 28.0 );
64  const value_type b( value_type( 8.0 ) / value_type( 3.0 ) );
65 
66  // first component trivial
67  dxdt[0] = 1.0; //sigma * ( x[1] - x[0] );
68  dxdt[1] = R * x[0] - x[1] - x[0] * x[2];
69  dxdt[2] = -b * x[2] + x[0] * x[1];
70 }
