104 ErrorType TrafficSignalManager::GetSpeedLimit(const State& state,
105  const Lane& lane,
106  decimal_t* speed_limit) const {
107  common::StateTransformer stf(lane);
108  common::FrenetState ref_fs;
109  if (stf.GetFrenetStateFromState(state, &ref_fs) != kSuccess) {
110  // printf("[GetSpeedLimit]Cannot get ref state frenet state.\n");
111  return kWrongStatus;
112  }
113 
114  const decimal_t acc_esti = 1.0;
115  decimal_t limit = kInf;
116  for (auto& speed_limit : speed_limit_list_) {
117  IntersectionType intersection_type;
118  decimal_t dist_to_startpt;
119  decimal_t dist_to_endpt;
120  if (CheckIntersectionTypeWithSignal(ref_fs, lane, speed_limit,
121  &intersection_type, &dist_to_startpt,
122  &dist_to_endpt) != kSuccess) {
123  continue;
124  }
125  if (intersection_type != kNotIntersect) {
126  decimal_t effect_speed_limit_dist =
127  state.velocity > speed_limit.max_velocity()
128  ? fabs(speed_limit.max_velocity() * speed_limit.max_velocity() -
129  state.velocity * state.velocity) /
130  (2.0 * acc_esti)
131  : 0.0;
132  // printf("[denny]cur vel %lf limit max vel %lf, effect dist %lf.\n",
133  // state.velocity, speed_limit.max_velocity(),
134  // effect_speed_limit_dist);
135  // if (intersection_type == kSignalAhead)
136  // printf("[denny]Speed limit ahead.\n");
137  // if (intersection_type == kSignalControlled)
138  // printf("[denny]under speed limit control.\n");
139  if ((intersection_type == kSignalAhead &&
140  dist_to_startpt < effect_speed_limit_dist) ||
141  intersection_type == kSignalControlled) {
142  limit = limit > speed_limit.max_velocity() ? speed_limit.max_velocity()
143  : limit;
144  }
145  }
146  }
147  *speed_limit = limit;
148  return kSuccess;
149 }

151 ErrorType TrafficSignalManager::CheckIntersectionTypeWithSignal(
152  const common::FrenetState& fs, const Lane& lane,
153  const common::TrafficSignal& signal, IntersectionType* intersection_type,
154  decimal_t* dist_to_startpt, decimal_t* dist_to_endpt) const {
155  // ~ NOTICE: use some naive solution to determine whether
156  // ~ a speed limit is effective for current <state, lane>
157  common::StateTransformer stf(lane);
158  Vec2f start_pt_fs, end_pt_fs;
159  if (stf.GetFrenetPointFromPoint(signal.start_point(), &start_pt_fs) !=
160  kSuccess) {
161  return kWrongStatus;
162  }
163  if (stf.GetFrenetPointFromPoint(signal.end_point(), &end_pt_fs) != kSuccess) {
164  return kWrongStatus;
165  }
166 
167  Vec2f lateral_range = signal.lateral_range();
168  IntersectionType int_type = kNotIntersect;
169  decimal_t dist_to_start = 0.0;
170  decimal_t dist_to_end = 0.0;
171  // * regard (projection onlane is reversed, start pt and end pt is too
172  // * far away from lateral range) as not intersect
173  // std::cout << "[denny]start pt fs" << start_pt_fs.transpose() << std::endl;
174  // std::cout << "[denny]end pt fs" << end_pt_fs.transpose() << std::endl;
175  if (start_pt_fs[0] < end_pt_fs[0] + kEPS) {
176  if (start_pt_fs[1] + lateral_range(1) > 0.0 &&
177  start_pt_fs[1] + lateral_range(0) < 0.0 &&
178  end_pt_fs[1] + lateral_range(1) > 0.0 &&
179  end_pt_fs[1] + lateral_range(0) < 0.0) {
180  if (fs.vec_s[0] < start_pt_fs[0]) {
181  int_type = kSignalAhead;
182  dist_to_start = start_pt_fs[0] - fs.vec_s[0];
183  dist_to_end = end_pt_fs[0] - fs.vec_s[0];
184  } else if (fs.vec_s[0] >= start_pt_fs[0] && fs.vec_s[0] < end_pt_fs[0]) {
185  int_type = kSignalControlled;
186  dist_to_start = start_pt_fs[0] - fs.vec_s[0];
187  dist_to_end = end_pt_fs[0] - fs.vec_s[0];
188  } else {
189  int_type = kNotIntersect;
190  }
191  }
192  }
193  *intersection_type = int_type;
194  *dist_to_startpt = dist_to_start;
195  *dist_to_endpt = dist_to_end;
196  return kSuccess;
197 }
