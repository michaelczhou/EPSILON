57 ErrorType DataRenderer::InjectObservationNoise() {
58  // * update uncertain vehicle ids if necessary
59  if (ego_id_ != 0) return kSuccess;
60  cnt_random_++;
61 
62  if (cnt_random_ == 10) {
63  uncertain_vehicle_ids_.clear();
64  const decimal_t angle_noise_std = 0.22;
65  std::normal_distribution<double> lat_pos_dist(0.0, 0.2);
66  std::normal_distribution<double> long_pos_dist(0.0, 0.7);
67  std::normal_distribution<double> angle_dist(0.0, angle_noise_std);
68 
69  // * sample vehicle
70  std::vector<int> surrounding_ids;
71  for (const auto &v : surrounding_vehicles_.vehicles) {
72  surrounding_ids.push_back(v.first);
73  }
74  std::shuffle(surrounding_ids.begin(), surrounding_ids.end(),
75  random_engine_);
76 
77  std::vector<int> sampled_ids;
78  for (int i = 0; i < 3 && i < surrounding_ids.size(); i++) {
79  sampled_ids.push_back(surrounding_ids[i]);
80  }
81 
82  // * inject noise
83  for (auto &v : surrounding_vehicles_.vehicles) {
84  if (std::find(sampled_ids.begin(), sampled_ids.end(), v.first) ==
85  sampled_ids.end())
86  continue;
87 
88  decimal_t lateral_position_noise = lat_pos_dist(random_engine_);
89  decimal_t long_position_noise = long_pos_dist(random_engine_);
90  decimal_t angle_noise = angle_dist(random_engine_);
91 
92  common::State original_state = v.second.state();
93  Vec2f original_position = original_state.vec_position;
94  decimal_t angle = original_state.angle;
95  Vec2f augmented_position =
96  Vec2f(original_position.x() + lateral_position_noise * sin(angle) +
97  long_position_noise * cos(angle),
98  original_position.y() - lateral_position_noise * cos(angle) +
99  long_position_noise * sin(angle));
100  original_state.vec_position = augmented_position;
101  original_state.angle =
102  normalize_angle(original_state.angle + angle_noise);
103 
104  if (fabs(angle_noise) > 1.5 * angle_noise_std &&
105  v.second.type().compare("brokencar") != 0)
106  uncertain_vehicle_ids_.push_back(v.first);
107 
108  v.second.set_state(original_state);
109  }
110  cnt_random_ = 0;
111  }
112  return kSuccess;
113 }

122 ErrorType DataRenderer::GetObstacleMap(
123  const common::ObstacleSet &obstacle_set) {
124  // ~ NOTICE:
125  // ~ Origin of OccupancyGrid is at left-bottom corner,
126  // ~ when x -> right, y -> up
127 
128  decimal_t x = ego_state_.vec_position(0) - obstacle_map_info_.h_metric / 2.0;
129  decimal_t y = ego_state_.vec_position(1) - obstacle_map_info_.w_metric / 2.0;
130  // Aligning to global coordinate to improve consistency
131  decimal_t x_r = std::round(x);
132  decimal_t y_r = std::round(y);
133 
134  // Use gridmap nd
135  p_obstacle_grid_->fill_data(GridMap2D::UNKNOWN);
136  std::array<decimal_t, 2> origin = {{x_r, y_r}};
137  p_obstacle_grid_->set_origin(origin);
138 
139  cv::Mat grid_mat =
140  cv::Mat(obstacle_map_info_.height, obstacle_map_info_.width,
141  CV_MAKETYPE(cv::DataType<ObstacleMapType>::type, 1),
142  p_obstacle_grid_->get_data_ptr());
143 
144  // Fill circle using opencv
145  for (const auto &obs : obstacle_set.obs_circle) {
146  std::array<decimal_t, 2> center_w = {
147  {obs.second.circle.center.x, obs.second.circle.center.y}};
148  auto center_coord = p_obstacle_grid_->GetCoordUsingGlobalPosition(center_w);
149  cv::Point2i center(center_coord[0], center_coord[1]);
150  int radius = obs.second.circle.radius / obstacle_map_info_.resolution;
151  cv::circle(grid_mat, center, radius, cv::Scalar(GridMap2D::OCCUPIED), -1);
152  }
153  // Fill polygon using opencv
154  std::vector<std::vector<cv::Point>> polys;
155  for (const auto &obs : obstacle_set.obs_polygon) {
156  std::vector<cv::Point> poly;
157  for (const auto &pt : obs.second.polygon.points) {
158  std::array<decimal_t, 2> pt_w = {{pt.x, pt.y}};
159  auto coord = p_obstacle_grid_->GetCoordUsingGlobalPosition(pt_w);
160  cv::Point2i coord_cv(coord[0], coord[1]);
161  poly.push_back(coord_cv);
162  }
163  polys.push_back(poly);
164  }
165  cv::fillPoly(grid_mat, polys, cv::Scalar(GridMap2D::OCCUPIED));
166 
167  return kSuccess;
168 }

170 ErrorType DataRenderer::FakeMapper() {
171  decimal_t dist_thres = obstacle_map_info_.h_metric / 2.0 * 0.8;
172  // obs_grids_.clear();
173  RayCastingOnObstacleMap();
174 
175  decimal_t ego_pos_x = ego_state_.vec_position(0);
176  decimal_t ego_pos_y = ego_state_.vec_position(1);
177  // Fill past obstacles and remove far ones
178  for (auto it = obs_grids_.begin(); it != obs_grids_.end();) {
179  decimal_t dx = abs((*it)[0] - ego_pos_x);
180  decimal_t dy = abs((*it)[1] - ego_pos_y);
181  if (dx >= dist_thres || dy >= dist_thres) {
182  it = obs_grids_.erase(it);
183  continue;
184  } else {
185  p_obstacle_grid_->SetValueUsingGlobalPosition(
186  *it, GridMap2D::SCANNED_OCCUPIED);
187  ++it;
188  }
189  }
190  return kSuccess;
191 }

273 ErrorType DataRenderer::GetSurroundingVehicles(

275  surrounding_vehicles_.vehicles.clear();
276 
277  for (const auto &v : vehicle_set.vehicles) {

282  v.second.state().vec_position(1) - ego_vehicle_.state().vec_position(1);
