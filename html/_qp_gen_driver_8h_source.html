41  if( argv[iargv][0] != '-' ) break; // done with the options
42  int iopt = 1; // Assume argv[iargv][1] is the start of the option name
43  if( argv[iargv][1] == '-' ) iopt = 2; // it is a --arg
44  if( 0 == strcmp( "print-level", &argv[iargv][iopt] ) ) {
45  if( ++iargv >= argc ) break;
46  char * endptr;
47  gOoqpPrintLevel = strtol( argv[iargv], &endptr, 10 );
48  if( '\0' != *endptr ) {
49  goodUsage = 0;
50  break;
51  }
52  } else if( 0 == strcmp( "quiet", &argv[iargv][iopt] ) ) {
53  gOoqpPrintLevel = 0;
54  } else if( 0 == strcmp( "verbose", &argv[iargv][iopt] ) ) {
55  gOoqpPrintLevel = 1000;
56  } else if( 0 == strcmp( "version", &argv[iargv][iopt] ) ) {
57  printOoqpVersionString();
58  exit(0);
59  } else if( 0 == strcmp( "scale", &argv[iargv][iopt] ) ) {
60  scale = 1;
61  } else {
62  // Unknown option, bug out of the option parsing loop
63  goodUsage = 0;
64  break;
65  }
66  iargv++;
67  }
68  if( iargv == argc - 1 && goodUsage ) {
69  strncpy( filename, argv[iargv], 255 );
70  filename[255] = '\0';
71  } else {
72  cerr << "Usage: " << argv[0] << " [ --version ] [ --print-level num ] "
73  "[ --quiet ] [ --verbose ] [--scale] problem.qps\n";
74  return 1;
75  }

79 #ifdef HAVE_GETRUSAGE
80  rusage before_read;

82 #endif

104  }

109 #endif
110

112  std::cerr << "Couldn't close file " << filename << std::endl

114  return 1;
115  }
