49  Acopy.makeCompressed();
50  Ccopy.makeCompressed();

52  assert(Ccopy.rows() == d.size());

55  // Refer to OOQP user guide section 2.2 (p. 10).
56  Matrix<char, Eigen::Dynamic, 1> useLowerLimitForX;

59  VectorXd upperLimitForX;

79  useUpperLimitForInequalityConstraints,

85  // Refer to OOQP user guide section 2.3 (p. 14).
86 
87  // Initialize new problem formulation.
88  int my = bcopy.size();
89  int mz = lowerLimitForInequalityConstraints.size();

92  int nnzC = Ccopy.nonZeros();
93 
94  QpGenSparseMa27* qp = new QpGenSparseMa27(nx, my, mz, nnzQ, nnzA, nnzC);
95  // Fill in problem data.
96  double* cp = &ccopy.coeffRef(0);
97  int* krowQ = Q_triangular.outerIndexPtr();
98  int* jcolQ = Q_triangular.innerIndexPtr();
99  double* dQ = Q_triangular.valuePtr();
100  double* xlow = &lowerLimitForX.coeffRef(0);
101  char* ixlow = &useLowerLimitForX.coeffRef(0);
102  double* xupp = &upperLimitForX.coeffRef(0);

105  int* jcolA = Acopy.innerIndexPtr();

107  double* bA = &bcopy.coeffRef(0);
108  int* krowC = Ccopy.outerIndexPtr();

111  double* clow = &lowerLimitForInequalityConstraints.coeffRef(0);
112  char* iclow = &useLowerLimitForInequalityConstraints.coeffRef(0);
113  double* cupp = &upperLimitForInequalityConstraints.coeffRef(0);
114  char* icupp = &useUpperLimitForInequalityConstraints.coeffRef(0);
115 
116  QpGenData* prob = (QpGenData*)qp->makeData(
117  cp, krowQ, jcolQ, dQ, xlow, ixlow, xupp, ixupp, krowA, jcolA, dA, bA,
118  krowC, jcolC, dC, clow, iclow, cupp, icupp);
119 
120  // Create object to store problem variables.
121  QpGenVars* vars = (QpGenVars*)qp->makeVariables(prob);
122  // if (isInDebugMode()) prob->print(); // Matrices are printed as [index_x,
123  // index_y, value]
124 
125  // Create object to store problem residual data.
126  QpGenResiduals* resid = (QpGenResiduals*)qp->makeResiduals(prob);
127 
128  // Create solver object.
129  GondzioSolver* s = new GondzioSolver(qp, prob);
130 
131  if (verbose) {
132  s->monitorSelf();
133  }
134 
135  // Solve.
136  int status = s->solve(prob, vars, resid);
137 
138  if ((status == SUCCESSFUL_TERMINATION) ||
139  (ignoreUnknownError && (status == UNKNOWN)))
140  vars->x->copyIntoArray(&x.coeffRef(0));
141 
142  if (verbose) {
143  printSolution(status, x);
144  }
145  // vars->x->writefToStream( cout, "x[%{index}] = %{value}" );
146 
147  delete s;
148  delete resid;
149  delete vars;
150  delete prob;
151  delete qp;
152 
153  return ((status == SUCCESSFUL_TERMINATION) ||
154  (ignoreUnknownError && (status == UNKNOWN)));

157 void OoQpItf::generateLimits(
158  const Eigen::VectorXd& l, const Eigen::VectorXd& u,
159  Eigen::Matrix<char, Eigen::Dynamic, 1>& useLowerLimit,
160  Eigen::Matrix<char, Eigen::Dynamic, 1>& useUpperLimit,
161  Eigen::VectorXd& lowerLimit, Eigen::VectorXd& upperLimit) {
162  int n = l.size();
163  useLowerLimit.setConstant(n, 1);
164  useUpperLimit.setConstant(n, 1);
165  lowerLimit = l;
166  upperLimit = u;
167 
168  for (int i = 0; i < n; i++) {
169  if (NumericalUtil::ApproximatelyEqual(
170  l(i), -std::numeric_limits<double>::max())) {
171  useLowerLimit(i) = 0;
172  lowerLimit(i) = 0.0;
173  }
174  if (NumericalUtil::ApproximatelyEqual(u(i),
175  std::numeric_limits<double>::max())) {
176  useUpperLimit(i) = 0;
177  upperLimit(i) = 0.0;
178  }
179  }
180 }
