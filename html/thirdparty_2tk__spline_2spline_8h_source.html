134 double& band_matrix::operator()(int i, int j) {
135  int k = j - i; // what band is the entry
136  assert((i >= 0) && (i < dim()) && (j >= 0) && (j < dim()));
137  assert((-num_lower() <= k) && (k <= num_upper()));
138  // k=0 -> diogonal, k<0 lower left part, k>0 upper right part
139  if (k >= 0)
140  return m_upper[k][i];
141  else
142  return m_lower[-k][i];
143 }

164 // LR-Decomposition of a band matrix
165 void band_matrix::lu_decompose() {
166  int i_max, j_max;
167  int j_min;
168  double x;
169 
170  // preconditioning
171  // normalize column i so that a_ii=1
172  for (int i = 0; i < this->dim(); i++) {
173  assert(this->operator()(i, i) != 0.0);
174  this->saved_diag(i) = 1.0 / this->operator()(i, i);
175  j_min = std::max(0, i - this->num_lower());
176  j_max = std::min(this->dim() - 1, i + this->num_upper());
177  for (int j = j_min; j <= j_max; j++) {
178  this->operator()(i, j) *= this->saved_diag(i);
179  }
180  this->operator()(i, i) = 1.0; // prevents rounding errors
181  }
182 
183  // Gauss LR-Decomposition
184  for (int k = 0; k < this->dim(); k++) {
185  i_max = std::min(this->dim() - 1,
186  k + this->num_lower()); // num_lower not a mistake!
187  for (int i = k + 1; i <= i_max; i++) {
188  assert(this->operator()(k, k) != 0.0);
189  x = -this->operator()(i, k) / this->operator()(k, k);
190  this->operator()(i, k) = -x; // assembly part of L
191  j_max = std::min(this->dim() - 1, k + this->num_upper());
192  for (int j = k + 1; j <= j_max; j++) {
193  // assembly part of R
194  this->operator()(i, j) =
195  this->operator()(i, j) + x * this->operator()(k, j);
196  }
197  }
198  }
199 }
200 // solves Ly=b
201 std::vector<double> band_matrix::l_solve(const std::vector<double>& b) const {
202  assert(this->dim() == (int)b.size());
203  std::vector<double> x(this->dim());
204  int j_start;
205  double sum;
206  for (int i = 0; i < this->dim(); i++) {
207  sum = 0;
208  j_start = std::max(0, i - this->num_lower());
209  for (int j = j_start; j < i; j++) sum += this->operator()(i, j) * x[j];
210  x[i] = (b[i] * this->saved_diag(i)) - sum;
211  }
212  return x;
213 }
214 // solves Rx=y
215 std::vector<double> band_matrix::r_solve(const std::vector<double>& b) const {
216  assert(this->dim() == (int)b.size());
217  std::vector<double> x(this->dim());
218  int j_stop;
219  double sum;
220  for (int i = this->dim() - 1; i >= 0; i--) {
221  sum = 0;
222  j_stop = std::min(this->dim() - 1, i + this->num_upper());
223  for (int j = i + 1; j <= j_stop; j++) sum += this->operator()(i, j) * x[j];
224  x[i] = (b[i] - sum) / this->operator()(i, i);
225  }
226  return x;
227 }

229 std::vector<double> band_matrix::lu_solve(const std::vector<double>& b,
230  bool is_lu_decomposed) {
231  assert(this->dim() == (int)b.size());
232  std::vector<double> x, y;
233  if (is_lu_decomposed == false) {
234  this->lu_decompose();
235  }
236  y = this->l_solve(b);
237  x = this->r_solve(y);
238  return x;
239 }

255 void spline::set_points(const std::vector<double>& x,
256  const std::vector<double>& y, bool cubic_spline) {
257  assert(x.size() == y.size());
258  assert(x.size() > 2);
259  m_x = x;
260  m_y = y;
261  int n = x.size();
262  // TODO: maybe sort x and y, rather than returning an error
263  for (int i = 0; i < n - 1; i++) {
264  assert(m_x[i] < m_x[i + 1]);
265  }
266 
267  if (cubic_spline == true) { // cubic spline interpolation
268  // setting up the matrix and right hand side of the equation system
269  // for the parameters b[]
270  band_matrix A(n, 1, 1);
271  std::vector<double> rhs(n);
272  for (int i = 1; i < n - 1; i++) {
273  A(i, i - 1) = 1.0 / 3.0 * (x[i] - x[i - 1]);
274  A(i, i) = 2.0 / 3.0 * (x[i + 1] - x[i - 1]);
275  A(i, i + 1) = 1.0 / 3.0 * (x[i + 1] - x[i]);
276  rhs[i] = (y[i + 1] - y[i]) / (x[i + 1] - x[i]) -
277  (y[i] - y[i - 1]) / (x[i] - x[i - 1]);
278  }
279  // boundary conditions
280  if (m_left == spline::second_deriv) {
281  // 2*b[0] = f''
282  A(0, 0) = 2.0;
283  A(0, 1) = 0.0;
284  rhs[0] = m_left_value;
285  } else if (m_left == spline::first_deriv) {
286  // c[0] = f', needs to be re-expressed in terms of b:
287  // (2b[0]+b[1])(x[1]-x[0]) = 3 ((y[1]-y[0])/(x[1]-x[0]) - f')
288  A(0, 0) = 2.0 * (x[1] - x[0]);
289  A(0, 1) = 1.0 * (x[1] - x[0]);
290  rhs[0] = 3.0 * ((y[1] - y[0]) / (x[1] - x[0]) - m_left_value);
291  } else {
292  assert(false);
293  }
294  if (m_right == spline::second_deriv) {
295  // 2*b[n-1] = f''
296  A(n - 1, n - 1) = 2.0;
297  A(n - 1, n - 2) = 0.0;
298  rhs[n - 1] = m_right_value;
299  } else if (m_right == spline::first_deriv) {
300  // c[n-1] = f', needs to be re-expressed in terms of b:
301  // (b[n-2]+2b[n-1])(x[n-1]-x[n-2])
302  // = 3 (f' - (y[n-1]-y[n-2])/(x[n-1]-x[n-2]))
303  A(n - 1, n - 1) = 2.0 * (x[n - 1] - x[n - 2]);
304  A(n - 1, n - 2) = 1.0 * (x[n - 1] - x[n - 2]);
305  rhs[n - 1] =
306  3.0 * (m_right_value - (y[n - 1] - y[n - 2]) / (x[n - 1] - x[n - 2]));
307  } else {
308  assert(false);
309  }
310 
311  // solve the equation system to obtain the parameters b[]
312  m_b = A.lu_solve(rhs);
313 
314  // calculate parameters a[] and c[] based on b[]
315  m_a.resize(n);
316  m_c.resize(n);
317  for (int i = 0; i < n - 1; i++) {
318  m_a[i] = 1.0 / 3.0 * (m_b[i + 1] - m_b[i]) / (x[i + 1] - x[i]);
319  m_c[i] = (y[i + 1] - y[i]) / (x[i + 1] - x[i]) -
320  1.0 / 3.0 * (2.0 * m_b[i] + m_b[i + 1]) * (x[i + 1] - x[i]);
321  }
322  } else { // linear interpolation
323  m_a.resize(n);
324  m_b.resize(n);
325  m_c.resize(n);
326  for (int i = 0; i < n - 1; i++) {
327  m_a[i] = 0.0;
328  m_b[i] = 0.0;
329  m_c[i] = (m_y[i + 1] - m_y[i]) / (m_x[i + 1] - m_x[i]);
330  }
331  }
332 
333  // for left extrapolation coefficients
334  m_b0 = (m_force_linear_extrapolation == false) ? m_b[0] : 0.0;
335  m_c0 = m_c[0];
336 
337  // for the right extrapolation coefficients
338  // f_{n-1}(x) = b*(x-x_{n-1})^2 + c*(x-x_{n-1}) + y_{n-1}
339  double h = x[n - 1] - x[n - 2];
340  // m_b[n-1] is determined by the boundary condition
341  m_a[n - 1] = 0.0;
342  m_c[n - 1] = 3.0 * m_a[n - 2] * h * h + 2.0 * m_b[n - 2] * h +
343  m_c[n - 2]; // = f'_{n-2}(x_{n-1})
344  if (m_force_linear_extrapolation == true) m_b[n - 1] = 0.0;
345 }

347 double spline::operator()(double x) const {
348  size_t n = m_x.size();
349  // find the closest point m_x[idx] < x, idx=0 even if x<m_x[0]
350  std::vector<double>::const_iterator it;
351  it = std::lower_bound(m_x.begin(), m_x.end(), x);
352  int idx = std::max(int(it - m_x.begin()) - 1, 0);
353 
354  double h = x - m_x[idx];
355  double interpol;
356  if (x < m_x[0]) {
357  // extrapolation to the left
358  interpol = (m_b0 * h + m_c0) * h + m_y[0];
359  } else if (x > m_x[n - 1]) {
360  // extrapolation to the right
361  interpol = (m_b[n - 1] * h + m_c[n - 1]) * h + m_y[n - 1];
362  } else {
363  // interpolation
364  interpol = ((m_a[idx] * h + m_b[idx]) * h + m_c[idx]) * h + m_y[idx];
365  }
366  return interpol;
367 }

371 double spline::get_coeff(int n, int j) const {
372  assert(n < static_cast<int>(m_x.size()));
373  assert(j < 4); // cubic spline
374  switch (j) {
375  case 0:
376  return m_y[n];
377  case 1:
378  return m_c[n];
379  case 2:
380  return m_b[n];
381  case 3:
382  return m_a[n];
383  default:
384  assert(false);
385  }
386  return 0.0;
387 }
