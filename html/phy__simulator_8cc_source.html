72 bool PhySimulation::UpdateVehicleStates(
73  const common::VehicleControlSignalSet &signal_set, const decimal_t &dt) {
74  if (signal_set.signal_set.size() != vehicle_set_.vehicles.size()) {

80  assert(false);

83  iter != vehicle_set_.vehicles.end(); ++iter) {
84  int id = iter->first;
85  common::VehicleControlSignal signal = signal_set.signal_set.at(id);
86  decimal_t steer_rate = signal.steer_rate;
87  decimal_t acc = signal.acc;
88

90 
91  if (!signal.is_openloop) {

93  simulator::VehicleModel::Control(steer_rate, acc));
94  model_iter->second.Step(dt);

97  }

99  }

103 bool PhySimulation::RemoveTemporaryObstacle(const common::Point &pt,
104  const double &s) {
105  for (auto it = temp_obstacle_set_.begin(); it != temp_obstacle_set_.end();) {
106  decimal_t sum_x = 0, sum_y = 0;
107  for (const auto &p : it->second.polygon.points) {
108  sum_x += p.x;
109  sum_y += p.y;
110  }
111  decimal_t center_x = sum_x / it->second.polygon.points.size();
112  decimal_t center_y = sum_y / it->second.polygon.points.size();
113  decimal_t dx = center_x - pt.x;
114  decimal_t dy = center_y - pt.y;
115  decimal_t d = std::hypot(dx, dy);
116 
117  if (d < s) {
118  auto it_obs_set = obstacle_set_.obs_polygon.find(it->first);
119  if (it_obs_set != obstacle_set_.obs_polygon.end()) {
120  obstacle_set_.obs_polygon.erase(it_obs_set);
121  }
122  it = temp_obstacle_set_.erase(it);
123  } else {
124  ++it;
125  }
126  }
127  return true;
128 }
