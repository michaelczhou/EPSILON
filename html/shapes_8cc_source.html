56 bool ShapeUtils::CheckIfOrientedBoundingBoxIntersect(
57  const OrientedBoundingBox2D& obb_a, const OrientedBoundingBox2D& obb_b) {
58  vec_E<Vecf<2>> vertices_a, vertices_b;
59  GetVerticesOfOrientedBoundingBox(obb_a, &vertices_a);
60  GetVerticesOfOrientedBoundingBox(obb_b, &vertices_b);
61  vec_E<Vecf<2>> axes;
62  GetPerpendicularAxesOfOrientedBoundingBox(vertices_a, &axes);
63  GetPerpendicularAxesOfOrientedBoundingBox(vertices_b, &axes);
64  Vecf<2> proj_a, proj_b;
65  decimal_t overlap_len;
66  // decimal_t minoverlap = std::numeric_limits<decimal_t>::infinity();
67  for (auto& axis : axes) {
68  GetProjectionOnAxis(vertices_a, axis, &proj_a);
69  GetProjectionOnAxis(vertices_b, axis, &proj_b);
70  GetOverlapLength(proj_a, proj_b, &overlap_len);
71  if (fabs(overlap_len) < kEPS) { // shapes are not overlapping
72  return false;
73  }
74  }
75  return true;
76 }

78 ErrorType ShapeUtils::GetVerticesOfOrientedBoundingBox(
79  const OrientedBoundingBox2D& obb, vec_E<Vecf<2>>* vertices) {
80  //
81  // corner2 corner1
82  // y<-- ________ x
83  // | | ^
84  // | | |
85  // | |
86  // |______|
87  // corner3 corner4
88  vertices->clear();
89  vertices->reserve(4);
90  decimal_t cos_theta = cos(obb.angle);
91  decimal_t sin_theta = sin(obb.angle);
92  Vecf<2> corner1(
93  obb.x + 0.5 * obb.length * cos_theta + 0.5 * obb.width * sin_theta,
94  obb.y + 0.5 * obb.length * sin_theta - 0.5 * obb.width * cos_theta);
95  Vecf<2> corner2(
96  obb.x + 0.5 * obb.length * cos_theta - 0.5 * obb.width * sin_theta,
97  obb.y + 0.5 * obb.length * sin_theta + 0.5 * obb.width * cos_theta);
98  Vecf<2> corner3(
99  obb.x - 0.5 * obb.length * cos_theta - 0.5 * obb.width * sin_theta,
100  obb.y - 0.5 * obb.length * sin_theta + 0.5 * obb.width * cos_theta);
101  Vecf<2> corner4(
102  obb.x - 0.5 * obb.length * cos_theta + 0.5 * obb.width * sin_theta,
103  obb.y - 0.5 * obb.length * sin_theta - 0.5 * obb.width * cos_theta);
104  vertices->push_back(corner1);
105  vertices->push_back(corner2);
106  vertices->push_back(corner3);
107  vertices->push_back(corner4);
108  return kSuccess;
109 }

122 ErrorType ShapeUtils::GetProjectionOnAxis(const vec_E<Vecf<2>>& vertices,
123  const Vecf<2>& axis, Vecf<2>* proj) {
124  decimal_t min = std::numeric_limits<decimal_t>::infinity();
125  decimal_t max = -std::numeric_limits<decimal_t>::infinity();
126  decimal_t projection;
127  for (auto& vertex : vertices) {
128  projection = vertex.dot(axis);
129  if (projection < min) {
130  min = projection;
131  }
132  if (projection > max) {
133  max = projection;
134  }
135  }
136  *proj = Vecf<2>(min, max);
137  return kSuccess;
138 }

140 ErrorType ShapeUtils::GetPerpendicularAxisOfOrientedBoundingBox(
141  const vec_E<Vecf<2>>& vertices, const int index, Vecf<2>* axis) {
142  assert(index >= 0 && index < 4);
143  Vecf<2> vec = vertices[index + 1] - vertices[index];
144  decimal_t length = vec.norm();
145  Vecf<2> normalized_vec = Vecf<2>::Zero();
146  if (length > kEPS) normalized_vec = vec / length;
147  // right hand normal vector
148  (*axis)[0] = -normalized_vec[1];
149  (*axis)[1] = normalized_vec[0];
150  return kSuccess;
151 }

153 ErrorType ShapeUtils::GetOverlapLength(const Vecf<2> a, const Vecf<2> b,
154  decimal_t* len) {
155  if (a.x() > b.y() || a.y() < b.x()) {
156  *len = 0.0;
157  return kSuccess;
158  }
159 
160  *len = std::min(a.y(), b.y()) - std::max(a.x(), b.x());
161  return kSuccess;
162 }
