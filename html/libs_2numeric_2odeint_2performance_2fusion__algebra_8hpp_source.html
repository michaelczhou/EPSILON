26  const boost::array< double , n > &a ,
27  const boost::array< T , dim > k_vector[n] , const double dt )
28  {
29  for( size_t i=0 ; i<dim ; ++i )
30  {
31  x_tmp[i] = x[i];// + a[0]*dt*k_vector[0][i];

33  x_tmp[i] += a[j]*dt*k_vector[j][i];
34  }
