40 void IdealSteerModel::TruncateControl(const decimal_t &dt) {
41  // truncate velocity
42  // decimal_t max_velocity_by_model =
43  // sqrt(max_lat_acc_ / std::min(fabs(state_.curvature), max_curvature_));
44  // control_.velocity =
45  // std::min(std::max(0.0, control_.velocity), max_velocity_by_model);
46  desired_lon_acc_ = (control_.velocity - state_.velocity) / dt;
47  decimal_t desired_lon_jerk = (desired_lon_acc_ - state_.acceleration) / dt;
48  desired_lon_jerk =
49  truncate(desired_lon_jerk, -max_lon_dec_jerk_, max_lon_acc_jerk_);
50  desired_lon_acc_ = desired_lon_jerk * dt + state_.acceleration;
51  desired_lon_acc_ = truncate(desired_lon_acc_, -max_lon_dec_, max_lon_acc_);
52  control_.velocity = std::max(state_.velocity + desired_lon_acc_ * dt, 0.0);
53 
54  desired_lat_acc_ =
55  pow(control_.velocity, 2) * (tan(control_.steer) / wheelbase_len_);
56  decimal_t lat_acc_ori = pow(state_.velocity, 2) * state_.curvature;
57  decimal_t lat_jerk_desired = (desired_lat_acc_ - lat_acc_ori) / dt;
58  lat_jerk_desired = truncate(lat_jerk_desired, -max_lat_jerk_, max_lat_jerk_);
59  desired_lat_acc_ = lat_jerk_desired * dt + lat_acc_ori;
60  desired_lat_acc_ = truncate(desired_lat_acc_, -max_lat_acc_, max_lat_acc_);
61  control_.steer = atan(desired_lat_acc_ * wheelbase_len_ /
62  std::max(pow(control_.velocity, 2), 0.1 * kBigEPS));
63  desired_steer_rate_ = normalize_angle(control_.steer - state_.steer) / dt;
64  desired_steer_rate_ =
65  truncate(desired_steer_rate_, -max_steer_rate_, max_steer_rate_);
66  control_.steer = normalize_angle(state_.steer + desired_steer_rate_ * dt);
67 }

69 void IdealSteerModel::Step(double dt) {
70  state_.steer = atan(state_.curvature * wheelbase_len_);
71  UpdateInternalState();
72  control_.velocity = std::max(0.0, control_.velocity);
73  control_.steer =
74  truncate(control_.steer, -max_steering_angle_, max_steering_angle_);
75  TruncateControl(dt);
76  desired_lon_acc_ = (control_.velocity - state_.velocity) / dt;
77  desired_steer_rate_ = normalize_angle(control_.steer - state_.steer) / dt;

79  odeint::integrate(boost::ref(*this), internal_state_, 0.0, dt, dt);
80  state_.vec_position(0) = internal_state_[0];

85  state_.curvature = tan(state_.steer) * 1.0 / wheelbase_len_;
