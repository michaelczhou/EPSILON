67 void oscillator( const state_type &x , deriv_type &dxdt , time_type t )
68 {
69  const units::quantity< si::frequency , value_type > omega = 1.0 * si::hertz;
70  fusion::at_c< 0 >( dxdt ) = fusion::at_c< 1 >( x );
71  fusion::at_c< 1 >( dxdt ) = - omega * omega * fusion::at_c< 0 >( x );
72 }
