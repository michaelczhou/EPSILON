5 ErrorType MobilBehaviorPrediction::RemapGainsToProb(
6  const bool is_lcl_safe, const decimal_t mobil_gain_left,
7  const bool is_lcr_safe, const decimal_t mobil_gain_right,
8  ProbDistOfLatBehaviors *res) {
9  decimal_t lower_bound = -1.0;
10  // decimal_t upper_bound = 1.0;
11  decimal_t upper_bound = 6.0;
12 
13  if (is_lcl_safe && is_lcr_safe) {
14  decimal_t gain_l = normalize_with_bound(mobil_gain_left, lower_bound,
15  upper_bound, 0.0, 1.0);
16  decimal_t gain_k =
17  normalize_with_bound(1.0, lower_bound, upper_bound, 0.0, 1.0);
18  decimal_t gain_r = normalize_with_bound(mobil_gain_right, lower_bound,
19  upper_bound, 0.0, 1.0);
20 
21  decimal_t p_l = gain_l / (gain_k + gain_l + gain_r);
22  decimal_t p_k = gain_k / (gain_k + gain_l + gain_r);
23  decimal_t p_r = gain_r / (gain_k + gain_l + gain_r);
24 
25  res->SetEntry(common::LateralBehavior::kLaneChangeLeft, p_l);
26  res->SetEntry(common::LateralBehavior::kLaneChangeRight, p_r);
27  res->SetEntry(common::LateralBehavior::kLaneKeeping, p_k);
28  res->is_valid = true;
29 
30  } else if (is_lcl_safe) {
31  decimal_t gain = normalize_with_bound(mobil_gain_left, lower_bound,
32  upper_bound, 0.0, 1.0);
33  res->SetEntry(common::LateralBehavior::kLaneChangeLeft, gain);
34  res->SetEntry(common::LateralBehavior::kLaneChangeRight, 0.0);
35  res->SetEntry(common::LateralBehavior::kLaneKeeping, 1.0 - gain);
36  res->is_valid = true;
37  } else if (is_lcr_safe) {
38  decimal_t gain = normalize_with_bound(mobil_gain_right, lower_bound,
39  upper_bound, 0.0, 1.0);
40  res->SetEntry(common::LateralBehavior::kLaneChangeLeft, 0.0);
41  res->SetEntry(common::LateralBehavior::kLaneChangeRight, gain);
42  res->SetEntry(common::LateralBehavior::kLaneKeeping, 1.0 - gain);
43  res->is_valid = true;
44  } else {
45  res->SetEntry(common::LateralBehavior::kLaneChangeLeft, 0.0);
46  res->SetEntry(common::LateralBehavior::kLaneChangeRight, 0.0);
47  res->SetEntry(common::LateralBehavior::kLaneKeeping, 1.0);
48  res->is_valid = true;
49  }
50  return kSuccess;
51 }

65  }

67  // * Notations here are following MOBIL paper

73  bool is_lcr_safe = false;

76  decimal_t mobil_gain_right = -kInf;
77 
78  // * When the vehicle is almost stop, we assume the behavior is lane keeping
79  decimal_t desired_vel = vehicle.state().velocity;
80  if (fabs(desired_vel) < kBigEPS) {
81  res->SetEntry(common::LateralBehavior::kLaneChangeLeft, 0.0);
82  res->SetEntry(common::LateralBehavior::kLaneChangeRight, 0.0);
83  res->SetEntry(common::LateralBehavior::kLaneKeeping, 1.0);
84  res->is_valid = true;
85  return kSuccess;
86  }
87 
88  // ~ For current lane
89  common::Lane lk_lane = lanes[0];
90  if (lk_lane.IsValid()) {

92  common::FrenetState leading_fs = leading_frenet_states[0];
93  common::Vehicle following_vehicle = following_vehicles[0];
94  common::FrenetState following_fs = follow_frenet_states[0];
95 
96  common::StateTransformer stf(lk_lane);

98  stf.GetFrenetStateFromState(vehicle.state(), &ego_frenet_state);
99 
100  if (MobilLaneChangingModel::GetMobilAccChangesOnCurrentLane(
101  ego_frenet_state, leading_vehicle, leading_fs, following_vehicle,
102  following_fs, &acc_o, &acc_o_tilda, &acc_c) != kSuccess) {
103  printf("[MobilBehaviorPrediction]lane-keep lane not valid.\n");
104  return kWrongStatus;
105  }
106  } else {
107  return kWrongStatus;
108  }
109 
110  // ~ For lane change left
111  common::Lane lcl_lane = lanes[1];
112  if (lcl_lane.IsValid()) {
113  common::Vehicle leading_vehicle = leading_vehicles[1];
114  common::FrenetState leading_fs = leading_frenet_states[1];
115  common::Vehicle following_vehicle = following_vehicles[1];
116  common::FrenetState following_fs = follow_frenet_states[1];

118

121  stf.GetFrenetStateFromState(vehicle.state(), &projected_frenet_state);
122 
123  MobilLaneChangingModel::GetMobilAccChangesOnTargetLane(
124  projected_frenet_state, leading_vehicle, leading_fs, following_vehicle,
125  following_fs, &is_lcl_safe, &acc_n, &acc_n_tilda, &acc_c_tilda);
126  if (is_lcl_safe) {
127  decimal_t d_a_c = acc_c_tilda - acc_c;
128  decimal_t d_a_n = acc_n_tilda - acc_n;
129  decimal_t d_a_o = acc_o_tilda - acc_o;
130  mobil_gain_left = d_a_c + politeness_coeff * (d_a_n + d_a_o);
131  }
132  }
133

135  common::Lane lcr_lane = lanes[2];
136  if (lcr_lane.IsValid()) {
137  common::Vehicle leading_vehicle = leading_vehicles[2];
138  common::FrenetState leading_fs = leading_frenet_states[2];
139  common::Vehicle following_vehicle = following_vehicles[2];
140  common::FrenetState following_fs = follow_frenet_states[2];
141  decimal_t acc_n = 0.0, acc_n_tilda = 0.0, acc_c_tilda = 0.0;
142 
143  common::StateTransformer stf(lcr_lane);
144  common::FrenetState projected_frenet_state;
145  stf.GetFrenetStateFromState(vehicle.state(), &projected_frenet_state);
146 
147  MobilLaneChangingModel::GetMobilAccChangesOnTargetLane(
148  projected_frenet_state, leading_vehicle, leading_fs, following_vehicle,
149  following_fs, &is_lcr_safe, &acc_n, &acc_n_tilda, &acc_c_tilda);
150  if (is_lcr_safe) {
151  decimal_t d_a_c = acc_c_tilda - acc_c;
152  decimal_t d_a_n = acc_n_tilda - acc_n;
153  decimal_t d_a_o = acc_o_tilda - acc_o;
154  mobil_gain_right = d_a_c + politeness_coeff * (d_a_n + d_a_o);
155  }
156  }

159  res);
