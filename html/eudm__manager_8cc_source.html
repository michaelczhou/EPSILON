297  }

311

313  bp_.cfg().function().active_lc().consistent_min_num_frame()) {

316  kEPS) {
317  last_lc_proposal_.valid = true;
318  last_lc_proposal_.trigger_time = stamp;

326  stamp, bp_.cfg()

328  .active_lc()

330  stamp;
331  last_lc_proposal_.ego_lane_id = ego_lane_id_;
332  last_lc_proposal_.lat = lat_behavior;
333  preliminary_active_requests_.clear();
334  LOG(WARNING) << std::fixed << std::setprecision(5)
335  << "[HMI]Gen proposal with trigger time "
336  << last_lc_proposal_.trigger_time << " lane id "
337  << last_lc_proposal_.ego_lane_id << " behavior "
338  << static_cast<int>(last_lc_proposal_.lat) << " operate at "
339  << last_lc_proposal_.operation_at_seconds;
340  } else {
341  preliminary_active_requests_.clear();
342  // LOG(WARNING) << "[HMI]Abandan Queue due to change time not legal.";
343  }
344  }
345 
346  return kSuccess;

706 ErrorType EudmManager::EvaluateReferenceVelocity(
707  const planning::eudm::Task& task, decimal_t* ref_vel) {
708  if (!last_snapshot_.ref_lane.IsValid()) {
709  *ref_vel = task.user_desired_vel;
710  return kSuccess;
711  }
712  common::StateTransformer stf(last_snapshot_.ref_lane);
713  common::FrenetState current_fs;
714  stf.GetFrenetStateFromState(last_snapshot_.plan_state, &current_fs);
715 
716  decimal_t c, cc;
717  decimal_t v_max_by_curvature;
718  decimal_t v_ref = kInf;
719 
720  decimal_t a_comfort = bp_.cfg().sim().ego().lon().limit().soft_brake();
721  decimal_t t_forward = last_snapshot_.plan_state.velocity / a_comfort;
722  decimal_t s_forward =
723  std::min(std::max(20.0, t_forward * last_snapshot_.plan_state.velocity),
724  last_snapshot_.ref_lane.end());
725  decimal_t resolution = 0.2;
726 
727  for (decimal_t s = current_fs.vec_s[0]; s < current_fs.vec_s[0] + s_forward;
728  s += resolution) {
729  if (last_snapshot_.ref_lane.GetCurvatureByArcLength(s, &c, &cc) ==
730  kSuccess) {
731  v_max_by_curvature =
732  sqrt(bp_.cfg().sim().ego().lat().limit().acc() / fabs(c));
733  v_ref = v_max_by_curvature < v_ref ? v_max_by_curvature : v_ref;
734  }
735  }
736 
737  *ref_vel = std::floor(std::min(std::max(v_ref, 0.0), task.user_desired_vel));
738 
739  LOG(WARNING) << "[Eudm][Desired]User ref vel: " << task.user_desired_vel
740  << ", final ref vel: " << *ref_vel;
741  return kSuccess;
742 }

744 ErrorType EudmManager::ReselectByContext(const decimal_t stamp,
745  const Snapshot& snapshot,
746  int* new_seq_id) {
747  // *new_seq_id = snapshot.original_winner_id;
748  int selected_seq_id;
749  int num_seqs = snapshot.action_script.size();
750  bool find_match = false;
751  decimal_t cost = kInf;
752 
753  for (int i = 0; i < num_seqs; i++) {
754  if (!snapshot.sim_res[i]) continue;
755  common::LateralBehavior lat_behavior;
756  decimal_t operation_at_seconds;
757  bool is_cancel_behavior;
758  bp_.ClassifyActionSeq(snapshot.action_script[i], &operation_at_seconds,
759  &lat_behavior, &is_cancel_behavior);
760 
761  if ((lc_context_.completed &&
762  lat_behavior == common::LateralBehavior::kLaneKeeping) ||
763  (!lc_context_.completed && stamp < lc_context_.desired_operation_time &&
764  lat_behavior == common::LateralBehavior::kLaneKeeping) ||
765  (!lc_context_.completed &&
766  stamp >= lc_context_.desired_operation_time &&
767  (lat_behavior == lc_context_.lat ||
768  lat_behavior == common::LateralBehavior::kLaneKeeping))) {
769  find_match = true;
770  if (snapshot.final_cost[i] < cost) {
771  cost = snapshot.final_cost[i];
772  selected_seq_id = i;
773  }
774  }
775  }
776 
777  if (!find_match) {
778  return kWrongStatus;
779  }
780  *new_seq_id = selected_seq_id;
781  return kSuccess;

784 ErrorType EudmManager::Run(
785  const decimal_t stamp,
786  const std::shared_ptr<semantic_map_manager::SemanticMapManager>& map_ptr,
787  const planning::eudm::Task& task) {
788  LOG(WARNING) << std::fixed << std::setprecision(4)
789  << "[Eudm]******************** RUN START: " << stamp
790  << "******************";
791  static TicToc eudm_timer;
792  eudm_timer.tic();
793 
794  // * I : Prepare
795  static TicToc prepare_timer;
796  prepare_timer.tic();
797  if (Prepare(stamp, map_ptr, task) != kSuccess) {
798  return kWrongStatus;
799  }
800  auto t_prepare = prepare_timer.toc();
801  LOG(WARNING) << std::fixed << std::setprecision(4)
802  << "[Eudm]Prepare time cost " << t_prepare << " ms";
803 
804  // * II : RunOnce
805  static TicToc runonce_timer;
806  runonce_timer.tic();
807  if (bp_.RunOnce() != kSuccess) {
808  LOG(WARNING) << "[Eudm][Fatal]BP runonce failed.";
809  return kWrongStatus;
810  }
811  auto t_runonce = runonce_timer.toc();
812  LOG(WARNING) << std::fixed << std::setprecision(4)
813  << "[Eudm]RunOnce time cost " << t_runonce << " ms";
814 
815  static TicToc sum_reselect_timer;
816  sum_reselect_timer.tic();
817  // * III: Summarize
818  Snapshot snapshot;
819  SaveSnapshot(&snapshot);
820  // * IV: Reselect
821  if (ReselectByContext(stamp, snapshot, &snapshot.processed_winner_id) !=
822  kSuccess) {
823  LOG(WARNING) << "[Eudm][Fatal]Reselect failed.";
824  return kWrongStatus;
825  }
826  LOG(WARNING) << "[Eudm]original id " << snapshot.original_winner_id
827  << " reselect : " << snapshot.processed_winner_id;
828  {
829  std::ostringstream line_info;
830  line_info << "[Eudm][Output]Reselected <if_risky:"
831  << snapshot.risky_res[snapshot.processed_winner_id] << ">[";
832  for (auto& a : snapshot.action_script[snapshot.processed_winner_id]) {
833  line_info << DcpTree::RetLonActionName(a.lon);
834  }
835  line_info << "|";
836  for (auto& a : snapshot.action_script[snapshot.processed_winner_id]) {
837  line_info << DcpTree::RetLatActionName(a.lat);
838  }
839  line_info << "]";
840  for (auto& v : snapshot.forward_trajs[snapshot.processed_winner_id]) {
841  line_info << std::fixed << std::setprecision(5) << "<"
842  << v.state().time_stamp - stamp << "," << v.state().velocity
843  << "," << v.state().acceleration << "," << v.state().curvature
844  << ">";
845  }
846  LOG(WARNING) << line_info.str();
847  }
848  auto t_sum_reselect = sum_reselect_timer.toc();
849  LOG(WARNING) << std::fixed << std::setprecision(4)
850  << "[Eudm]Sum & Reselect time cost " << t_sum_reselect << " ms";
851 
852  static TicToc lane_timer;
853  lane_timer.tic();
854  if (map_adapter_.map()->GetRefLaneForStateByBehavior(
855  snapshot.plan_state, std::vector<int>(),
856  snapshot.forward_lat_behaviors[snapshot.processed_winner_id].front(),
857  250.0, 20.0, true, &(snapshot.ref_lane)) != kSuccess) {
858  return kWrongStatus;
859  }
860 
861  last_snapshot_ = snapshot;
862  GenerateLaneChangeProposal(stamp, task);
863  // * V: Update
864  context_.is_valid = true;
865  context_.seq_start_time = stamp;
866  context_.action_seq = snapshot.action_script[snapshot.processed_winner_id];
867  auto t_lane = lane_timer.toc();
868  LOG(WARNING) << "[Eudm]Fit reflane & update cost: " << t_lane << " ms";
869 
870  auto t_sum = t_prepare + t_runonce + t_sum_reselect + t_lane;
871  auto t_eudmrun = eudm_timer.toc();
872  LOG(WARNING) << std::fixed << std::setprecision(4)
873  << "[Eudm]Sum of time: " << t_sum
874  << " ms, diff: " << t_eudmrun - t_sum << " ms";
875  LOG(WARNING) << std::fixed << std::setprecision(4)
876  << "[Eudm]******************** RUN FINISH: " << stamp << " +"
877  << t_eudmrun << " ms ******************";
878  return kSuccess;
879 }

881 bool EudmManager::GetReplanDesiredAction(const decimal_t current_time,
882  DcpAction* desired_action) {
883  if (!context_.is_valid) return false;
884  decimal_t time_since_last_plan = current_time - context_.seq_start_time;
885  if (time_since_last_plan < -kEPS) return false;
886  decimal_t t_aggre = 0.0;
887  bool find_match_action = false;
888  int action_seq_len = context_.action_seq.size();
889  for (int i = 0; i < action_seq_len; ++i) {
890  t_aggre += context_.action_seq[i].t;
891  if (time_since_last_plan + kEPS < t_aggre) {
892  *desired_action = context_.action_seq[i];
893  desired_action->t = t_aggre - time_since_last_plan;
894  find_match_action = true;
895  break;
896  }
897  }
898  if (!find_match_action) {
899  return false;
900  }
901  return true;
902 }
