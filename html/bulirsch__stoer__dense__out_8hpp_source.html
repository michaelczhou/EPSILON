146  template< class System , class StateIn , class DerivIn , class StateOut , class DerivOut >
147  controlled_step_result try_step( System system , const StateIn &in , const DerivIn &dxdt , time_type &t , StateOut &out , DerivOut &dxdt_new , time_type &dt )
148  {
149  BOOST_USING_STD_MIN();
150  BOOST_USING_STD_MAX();
151  using std::pow;
152 
153  static const value_type val1( 1.0 );
154 
155  typename odeint::unwrap_reference< System >::type &sys = system;
156 
157  bool reject( true );
158 
159  time_vector h_opt( m_k_max+1 );
160  inv_time_vector work( m_k_max+1 );
161 
162  m_k_final = 0;
163  time_type new_h = dt;
164 
165  //std::cout << "t=" << t <<", dt=" << dt << ", k_opt=" << m_current_k_opt << ", first: " << m_first << std::endl;
166 
167  for( size_t k = 0 ; k <= m_current_k_opt+1 ; k++ )
168  {
169  m_midpoint.set_steps( m_interval_sequence[k] );
170  if( k == 0 )
171  {
172  m_midpoint.do_step( sys , in , dxdt , t , out , dt , m_mp_states[k].m_v , m_derivs[k]);
173  }
174  else
175  {
176  m_midpoint.do_step( sys , in , dxdt , t , m_table[k-1].m_v , dt , m_mp_states[k].m_v , m_derivs[k] );
177  extrapolate( k , m_table , m_coeff , out );
178  // get error estimate
179  m_algebra.for_each3( m_err.m_v , out , m_table[0].m_v ,
180  typename operations_type::template scale_sum2< value_type , value_type >( val1 , -val1 ) );
181  const value_type error = m_error_checker.error( m_algebra , in , dxdt , m_err.m_v , dt );
182  h_opt[k] = calc_h_opt( dt , error , k );
183  work[k] = static_cast<value_type>( m_cost[k] ) / h_opt[k];
184 
185  m_k_final = k;
186 
187  if( (k == m_current_k_opt-1) || m_first )
188  { // convergence before k_opt ?
189  if( error < 1.0 )
190  {
191  //convergence
192  reject = false;
193  if( (work[k] < KFAC2*work[k-1]) || (m_current_k_opt <= 2) )
194  {
195  // leave order as is (except we were in first round)
196  m_current_k_opt = min BOOST_PREVENT_MACRO_SUBSTITUTION( static_cast<int>(m_k_max)-1 , max BOOST_PREVENT_MACRO_SUBSTITUTION( 2 , static_cast<int>(k)+1 ) );
197  new_h = h_opt[k] * static_cast<value_type>( m_cost[k+1] ) / static_cast<value_type>( m_cost[k] );
198  } else {
199  m_current_k_opt = min BOOST_PREVENT_MACRO_SUBSTITUTION( static_cast<int>(m_k_max)-1 , max BOOST_PREVENT_MACRO_SUBSTITUTION( 2 , static_cast<int>(k) ) );
200  new_h = h_opt[k];
201  }
202  break;
203  }
204  else if( should_reject( error , k ) && !m_first )
205  {
206  reject = true;
207  new_h = h_opt[k];
208  break;
209  }
210  }
211  if( k == m_current_k_opt )
212  { // convergence at k_opt ?
213  if( error < 1.0 )
214  {
215  //convergence
216  reject = false;
217  if( (work[k-1] < KFAC2*work[k]) )
218  {
219  m_current_k_opt = max BOOST_PREVENT_MACRO_SUBSTITUTION( 2 , static_cast<int>(m_current_k_opt)-1 );
220  new_h = h_opt[m_current_k_opt];
221  }
222  else if( (work[k] < KFAC2*work[k-1]) && !m_last_step_rejected )
223  {
224  m_current_k_opt = min BOOST_PREVENT_MACRO_SUBSTITUTION( static_cast<int>(m_k_max)-1 , static_cast<int>(m_current_k_opt)+1 );
225  new_h = h_opt[k]*static_cast<value_type>( m_cost[m_current_k_opt] ) / static_cast<value_type>( m_cost[k] );
226  } else
227  new_h = h_opt[m_current_k_opt];
228  break;
229  }
230  else if( should_reject( error , k ) )
231  {
232  reject = true;
233  new_h = h_opt[m_current_k_opt];
234  break;
235  }
236  }
237  if( k == m_current_k_opt+1 )
238  { // convergence at k_opt+1 ?
239  if( error < 1.0 )
240  { //convergence
241  reject = false;
242  if( work[k-2] < KFAC2*work[k-1] )
243  m_current_k_opt = max BOOST_PREVENT_MACRO_SUBSTITUTION( 2 , static_cast<int>(m_current_k_opt)-1 );
244  if( (work[k] < KFAC2*work[m_current_k_opt]) && !m_last_step_rejected )
245  m_current_k_opt = min BOOST_PREVENT_MACRO_SUBSTITUTION( static_cast<int>(m_k_max)-1 , static_cast<int>(k) );
246  new_h = h_opt[m_current_k_opt];
247  } else
248  {
249  reject = true;
250  new_h = h_opt[m_current_k_opt];
251  }
252  break;
253  }
254  }
255  }
256 
257  if( !reject )
258  {
259 
260  //calculate dxdt for next step and dense output
261  sys( out , dxdt_new , t+dt );
262 
263  //prepare dense output
264  value_type error = prepare_dense_output( m_k_final , in , dxdt , out , dxdt_new , dt );
265 
266  if( error > static_cast<value_type>(10) ) // we are not as accurate for interpolation as for the steps
267  {
268  reject = true;
269  new_h = dt * pow BOOST_PREVENT_MACRO_SUBSTITUTION( error , static_cast<value_type>(-1)/(2*m_k_final+2) );
270  } else {
271  t += dt;
272  }
273  }
274  //set next stepsize
275  if( !m_last_step_rejected || (new_h < dt) )
276  dt = new_h;
277 
278  m_last_step_rejected = reject;
279  if( reject )
280  return fail;
281  else
282  return success;
283  }

299  template< class System >
300  std::pair< time_type , time_type > do_step( System system )
301  {
302  const size_t max_count = 1000;
303 
304  if( m_first )
305  {
306  typename odeint::unwrap_reference< System >::type &sys = system;
307  sys( get_current_state() , get_current_deriv() , m_t );
308  }
309 
310  controlled_step_result res = fail;
311  m_t_last = m_t;
312  size_t count = 0;
313  while( res == fail )
314  {
315  res = try_step( system , get_current_state() , get_current_deriv() , m_t , get_old_state() , get_old_deriv() , m_dt );
316  m_first = false;
317  if( count++ == max_count )
318  throw std::overflow_error( "bulirsch_stoer : too much iterations!");
319  }
320  toggle_current_state();
321  return std::make_pair( m_t_last , m_t );
322  }

407  time_type calc_h_opt( time_type h , value_type error , size_t k ) const
408  {
409  BOOST_USING_STD_MIN();
410  BOOST_USING_STD_MAX();
411  using std::pow;
412 
413  value_type expo=1.0/(m_interval_sequence[k-1]);
414  value_type facmin = pow BOOST_PREVENT_MACRO_SUBSTITUTION( STEPFAC3 , expo );
415  value_type fac;
416  if (error == 0.0)
417  fac=1.0/facmin;
418  else
419  {
420  fac = STEPFAC2 / pow BOOST_PREVENT_MACRO_SUBSTITUTION( error / STEPFAC1 , expo );
421  fac = max BOOST_PREVENT_MACRO_SUBSTITUTION( facmin/STEPFAC4 , min BOOST_PREVENT_MACRO_SUBSTITUTION( 1.0/facmin , fac ) );
422  }
423  return h*fac;
424  }

433  bool should_reject( value_type error , size_t k ) const
434  {
435  if( k == m_current_k_opt-1 )
436  {
437  const value_type d = m_interval_sequence[m_current_k_opt] * m_interval_sequence[m_current_k_opt+1] /
438  (m_interval_sequence[0]*m_interval_sequence[0]);
439  //step will fail, criterion 17.3.17 in NR
440  return ( error > d*d );
441  }
442  else if( k == m_current_k_opt )
443  {
444  const value_type d = m_interval_sequence[m_current_k_opt+1] / m_interval_sequence[0];
445  return ( error > d*d );
446  } else
447  return error > 1.0;
448  }

450  template< class StateIn1 , class DerivIn1 , class StateIn2 , class DerivIn2 >
451  value_type prepare_dense_output( int k , const StateIn1 &x_start , const DerivIn1 &dxdt_start ,
452  const StateIn2 & /* x_end */ , const DerivIn2 & /*dxdt_end */ , time_type dt )
453  /* k is the order to which the result was approximated */
454  {
455 
456  /* compute the coefficients of the interpolation polynomial
457  * we parametrize the interval t .. t+dt by theta = -1 .. 1
458  * we use 2k+3 values at the interval center theta=0 to obtain the interpolation coefficients
459  * the values are x(t+dt/2) and the derivatives dx/dt , ... d^(2k+2) x / dt^(2k+2) at the midpoints
460  * the derivatives are approximated via finite differences
461  * all values are obtained from interpolation of the results from the increasing orders of the midpoint calls
462  */
463 
464  // calculate finite difference approximations to derivatives at the midpoint
465  for( int j = 0 ; j<=k ; j++ )
466  {
467  /* not working with boost units... */
468  const value_type d = m_interval_sequence[j] / ( static_cast<value_type>(2) * dt );
469  value_type f = 1.0; //factor 1/2 here because our interpolation interval has length 2 !!!
470  for( int kappa = 0 ; kappa <= 2*j+1 ; ++kappa )
471  {
472  calculate_finite_difference( j , kappa , f , dxdt_start );
473  f *= d;
474  }
475 
476  if( j > 0 )
477  extrapolate_dense_out( j , m_mp_states , m_coeff );
478  }
479 
480  time_type d = dt/2;
481 
482  // extrapolate finite differences
483  for( int kappa = 0 ; kappa<=2*k+1 ; kappa++ )
484  {
485  for( int j=1 ; j<=(k-kappa/2) ; ++j )
486  extrapolate_dense_out( j , m_diffs[kappa] , m_coeff , kappa/2 );
487 
488  // extrapolation results are now stored in m_diffs[kappa][0]
489 
490  // divide kappa-th derivative by kappa because we need these terms for dense output interpolation
491  m_algebra.for_each1( m_diffs[kappa][0].m_v , typename operations_type::template scale< time_type >( static_cast<time_type>(d) ) );
492 
493  d *= dt/(2*(kappa+2));
494  }
495 
496  // dense output coefficients a_0 is stored in m_mp_states[0], a_i for i = 1...2k are stored in m_diffs[i-1][0]
497 
498  // the error is just the highest order coefficient of the interpolation polynomial
499  // this is because we use only the midpoint theta=0 as support for the interpolation (remember that theta = -1 .. 1)
500 
501  value_type error = 0.0;
502  if( m_control_interpolation )
503  {
504  boost::numeric::odeint::copy( m_diffs[2*k+1][0].m_v , m_err.m_v );
505  error = m_error_checker.error( m_algebra , x_start , dxdt_start , m_err.m_v , dt );
506  }
507 
508  return error;
509  }

511  template< class DerivIn >
512  void calculate_finite_difference( size_t j , size_t kappa , value_type fac , const DerivIn &dxdt )
513  {
514  const int m = m_interval_sequence[j]/2-1;
515  if( kappa == 0) // no calculation required for 0th derivative of f
516  {
517  m_algebra.for_each2( m_diffs[0][j].m_v , m_derivs[j][m].m_v ,
518  typename operations_type::template scale_sum1< value_type >( fac ) );
519  }
520  else
521  {
522  // calculate the index of m_diffs for this kappa-j-combination
523  const int j_diffs = j - kappa/2;
524 
525  m_algebra.for_each2( m_diffs[kappa][j_diffs].m_v , m_derivs[j][m+kappa].m_v ,
526  typename operations_type::template scale_sum1< value_type >( fac ) );
527  value_type sign = -1.0;
528  int c = 1;
529  //computes the j-th order finite difference for the kappa-th derivative of f at t+dt/2 using function evaluations stored in m_derivs
530  for( int i = m+static_cast<int>(kappa)-2 ; i >= m-static_cast<int>(kappa) ; i -= 2 )
531  {
532  if( i >= 0 )
533  {
534  m_algebra.for_each3( m_diffs[kappa][j_diffs].m_v , m_diffs[kappa][j_diffs].m_v , m_derivs[j][i].m_v ,
535  typename operations_type::template scale_sum2< value_type , value_type >( 1.0 ,
536  sign * fac * boost::math::binomial_coefficient< value_type >( kappa , c ) ) );
537  }
538  else
539  {
540  m_algebra.for_each3( m_diffs[kappa][j_diffs].m_v , m_diffs[kappa][j_diffs].m_v , dxdt ,
541  typename operations_type::template scale_sum2< value_type , value_type >( 1.0 , sign * fac ) );
542  }
543  sign *= -1;
544  ++c;
545  }
546  }

550  void do_interpolation( time_type t , StateOut &out ) const
551  {
552  // interpolation polynomial is defined for theta = -1 ... 1
553  // m_k_final is the number of order-iterations done for the last step - it governs the order of the interpolation polynomial
554  const value_type theta = 2 * get_unit_value( (t - m_t_last) / (m_t - m_t_last) ) - 1;
555  // we use only values at interval center, that is theta=0, for interpolation
556  // our interpolation polynomial is thus of order 2k+2, hence we have 2k+3 terms
557 
558  boost::numeric::odeint::copy( m_mp_states[0].m_v , out );
559  // add remaining terms: x += a_1 theta + a2 theta^2 + ... + a_{2k} theta^{2k}
560  value_type theta_pow( theta );
561  for( size_t i=0 ; i<=2*m_k_final+1 ; ++i )
562  {
563  m_algebra.for_each3( out , out , m_diffs[i][0].m_v ,
564  typename operations_type::template scale_sum2< value_type >( static_cast<value_type>(1) , theta_pow ) );
565  theta_pow *= theta;
566  }
567  }
