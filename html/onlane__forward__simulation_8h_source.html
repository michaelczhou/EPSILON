42  static ErrorType GetTargetStateOnTargetLane(
43  const common::StateTransformer& stf_target,
44  const common::Vehicle& ego_vehicle,
45  const common::Vehicle& gap_front_vehicle,
46  const common::Vehicle& gap_rear_vehicle, const Param& param,
47  common::State* target_state) {
48  common::FrenetState ego_fs;
49  if (kSuccess !=
50  stf_target.GetFrenetStateFromState(ego_vehicle.state(), &ego_fs)) {
51  return kWrongStatus;
52  }
53 
54  decimal_t time_headaway = param.idm_param.kDesiredHeadwayTime;
55  decimal_t min_spacing = param.idm_param.kMinimumSpacing;
56 
57  bool has_front = false;
58  common::FrenetState front_fs;
59  decimal_t s_ref_front = -1; // tail of front vehicle
60  decimal_t s_thres_front = -1;
61  if (gap_front_vehicle.id() != -1 &&
62  kSuccess == stf_target.GetFrenetStateFromState(
63  gap_front_vehicle.state(), &front_fs)) {
64  has_front = true;
65  s_ref_front =
66  front_fs.vec_s[0] - (gap_front_vehicle.param().length() / 2.0 -
67  gap_front_vehicle.param().d_cr());
68  s_thres_front = s_ref_front - min_spacing -
69  time_headaway * ego_vehicle.state().velocity;
70  }
71 
72  bool has_rear = false;
73  common::FrenetState rear_fs;
74  decimal_t s_ref_rear = -1; // head of rear vehicle
75  decimal_t s_thres_rear = -1;
76  if (gap_rear_vehicle.id() != -1 &&
77  kSuccess == stf_target.GetFrenetStateFromState(gap_rear_vehicle.state(),
78  &rear_fs)) {
79  has_rear = true;
80  s_ref_rear = rear_fs.vec_s[0] + gap_rear_vehicle.param().length() / 2.0 +
81  gap_rear_vehicle.param().d_cr();
82  s_thres_rear = s_ref_rear + min_spacing +
83  time_headaway * gap_rear_vehicle.state().velocity;
84  }
85 
86  decimal_t desired_s = ego_fs.vec_s[0];
87  decimal_t desired_v = ego_vehicle.state().velocity;
88 
89  // ~ params
90  decimal_t k_v = 0.1; // coeff for dv. k_v * s_err
91  decimal_t p_v_ego = 0.1; // coeff for user preferred vel
92  decimal_t dv_lb = -3.0; // dv lower bound
93  decimal_t dv_ub = 5.0; // dv upper bound
94 
95  decimal_t ego_desired_vel =
96  ego_vehicle.state().velocity +
97  (param.idm_param.kDesiredVelocity - ego_vehicle.state().velocity) *
98  p_v_ego;
99  if (has_front && has_rear) {
100  // * has both front and rear vehicle
101  if (s_ref_front < s_ref_rear) {
102  return kWrongStatus;
103  }
104 
105  decimal_t ds = fabs(s_ref_front - s_ref_rear);
106  decimal_t s_star = s_ref_rear + ds / 2.0 - ego_vehicle.param().d_cr();
107 
108  s_thres_front = std::max(s_star, s_thres_front);
109  s_thres_rear = std::min(s_star, s_thres_rear);
110 
111  desired_s =
112  std::min(std::max(s_thres_rear, ego_fs.vec_s[0]), s_thres_front);
113 
114  decimal_t s_err_front = s_thres_front - ego_fs.vec_s[0];
115  decimal_t v_ref_front =
116  std::max(0.0, gap_front_vehicle.state().velocity +
117  truncate(s_err_front * k_v, dv_lb, dv_ub));
118 
119  decimal_t s_err_rear = s_thres_rear - ego_fs.vec_s[0];
120  decimal_t v_ref_rear =
121  std::max(0.0, gap_rear_vehicle.state().velocity +
122  truncate(s_err_rear * k_v, dv_lb, dv_ub));
123 
124  desired_v = std::min(std::max(v_ref_rear, ego_desired_vel), v_ref_front);
125 
126  } else if (has_front) {
127  // * only has front vehicle
128  desired_s = std::min(ego_fs.vec_s[0], s_thres_front);
129 
130  decimal_t s_err_front = s_thres_front - ego_fs.vec_s[0];
131  decimal_t v_ref_front =
132  std::max(0.0, gap_front_vehicle.state().velocity +
133  truncate(s_err_front * k_v, dv_lb, dv_ub));
134  desired_v = std::min(ego_desired_vel, v_ref_front);
135 
136  } else if (has_rear) {
137  // * only has rear vehicle
138  desired_s = std::max(ego_fs.vec_s[0], s_thres_rear);
139 
140  decimal_t s_err_rear = s_thres_rear - ego_fs.vec_s[0];
141  decimal_t v_ref_rear =
142  std::max(0.0, gap_rear_vehicle.state().velocity +
143  truncate(s_err_rear * k_v, dv_lb, dv_ub));
144  desired_v = std::max(v_ref_rear, ego_desired_vel);
145  }
146 
147  common::FrenetState target_fs;
148  target_fs.Load(Vecf<3>(desired_s, desired_v, 0.0), Vecf<3>(0.0, 0.0, 0.0),
149  common::FrenetState::kInitWithDs);
150 
151  if (kSuccess !=
152  stf_target.GetStateFromFrenetState(target_fs, target_state)) {
153  return kWrongStatus;
154  }
155 
156  return kSuccess;
157  }

159  static ErrorType PropagateOnceAdvancedLK(
160  const common::StateTransformer& stf, const common::Vehicle& ego_vehicle,
161  const Vehicle& leading_vehicle, const decimal_t& lat_track_offset,
162  const decimal_t& dt, const Param& param, State* desired_state) {
163  common::State current_state = ego_vehicle.state();
164  decimal_t wheelbase_len = ego_vehicle.param().wheel_base();
165  auto sim_param = param;
166 
167  // * Step I: Calculate steering
168  bool steer_calculation_failed = false;
169  common::FrenetState current_fs;
170  if (stf.GetFrenetStateFromState(current_state, &current_fs) != kSuccess ||
171  current_fs.vec_s[1] < -kEPS) {
172  // * ego Frenet state invalid or ego vehicle reverse gear
173  steer_calculation_failed = true;
174  }
175 
176  decimal_t steer, velocity;
177  if (!steer_calculation_failed) {
178  decimal_t approx_lookahead_dist =
179  std::min(std::max(param.steer_control_min_lookahead_dist,
180  current_state.velocity * param.steer_control_gain),
181  param.steer_control_max_lookahead_dist);
182  if (CalcualateSteer(stf, current_state, current_fs, wheelbase_len,
183  Vec2f(approx_lookahead_dist, lat_track_offset),
184  &steer) != kSuccess) {
185  steer_calculation_failed = true;
186  }
187  }
188 
189  steer = steer_calculation_failed ? current_state.steer : steer;
190  decimal_t sim_vel = param.idm_param.kDesiredVelocity;
191  if (param.auto_decelerate_if_lat_failed && steer_calculation_failed) {
192  sim_vel = 0.0;
193  }
194  sim_param.idm_param.kDesiredVelocity = std::max(0.0, sim_vel);
195 
196  // * Step II: Calculate velocity
197  common::FrenetState leading_fs;
198  if (leading_vehicle.id() == kInvalidAgentId ||
199  stf.GetFrenetStateFromState(leading_vehicle.state(), &leading_fs) !=
200  kSuccess) {
201  // ~ Without leading vehicle
202  CalcualateVelocityUsingIdm(current_state.velocity, dt, sim_param,
203  &velocity);
204  } else {
205  // ~ With leading vehicle
206  // * For IDM, vehicle length is subtracted to get the 'net' distance
207  // * between ego vehicle and the leading vehicle.
208  decimal_t eqv_vehicle_len;
209  GetIdmEquivalentVehicleLength(stf, ego_vehicle, leading_vehicle,
210  leading_fs, &eqv_vehicle_len);
211  sim_param.idm_param.kVehicleLength = eqv_vehicle_len;
212 
213  CalcualateVelocityUsingIdm(
214  current_fs.vec_s[0], current_state.velocity, leading_fs.vec_s[0],
215  leading_vehicle.state().velocity, dt, sim_param, &velocity);
216  }
217 
218  // * Step III: Get desired state using vehicle kinematic model
219  CalculateDesiredState(current_state, steer, velocity, wheelbase_len, dt,
220  sim_param, desired_state);
221  return kSuccess;
222  }

399  static ErrorType GetIdmEquivalentVehicleLength(
400  const common::StateTransformer& stf, const common::Vehicle& ego_vehicle,
401  const common::Vehicle& leading_vehicle,
402  const common::FrenetState& leading_fs, decimal_t* eqv_vehicle_len) {
403  // Different from original IDM, we still use the center of rear axle as the
404  // vehicle position, since we need to calculate the 'net' distance between
405  // ego vehicle and leading vehicle, here we can get a equivalent leading
406  // vehicle's length for IDM
407 
408  // In case the leading vehicle has an opposite angle due to MOT error
409  std::array<Vec2f, 2> leading_pts;
410  leading_vehicle.RetBumperVertices(&leading_pts);
411 
412  Vec2f fs_pt1, fs_pt2;
413  std::vector<decimal_t> s_vec;
414  if (kSuccess == stf.GetFrenetPointFromPoint(leading_pts[0], &fs_pt1)) {
415  s_vec.push_back(fs_pt1(0));
416  }
417  if (kSuccess == stf.GetFrenetPointFromPoint(leading_pts[1], &fs_pt2)) {
418  s_vec.push_back(fs_pt2(0));
419  }
420  s_vec.push_back(leading_fs.vec_s(0));
421  decimal_t s_nearest_vtx = *(std::min_element(s_vec.begin(), s_vec.end()));
422 
423  decimal_t len_rb2r = fabs(leading_fs.vec_s(0) - s_nearest_vtx);
424  // ego rear-axle to front bumper + leading rear bumper to rear-axle
425  *eqv_vehicle_len = ego_vehicle.param().length() / 2.0 +
426  ego_vehicle.param().d_cr() + len_rb2r;
427 
428  return kSuccess;
429  }

431  static ErrorType CalcualateSteer(const common::StateTransformer& stf,
432  const State& current_state,
433  const FrenetState& current_fs,
434  const decimal_t& wheelbase_len,
435  const Vec2f& lookahead_offset,
436  decimal_t* steer) {
437  common::FrenetState dest_fs;
438  dest_fs.Load(Vecf<3>(lookahead_offset(0) + current_fs.vec_s[0], 0.0, 0.0),
439  Vecf<3>(lookahead_offset(1), 0.0, 0.0),
440  common::FrenetState::kInitWithDs);
441 
442  State dest_state;
443  if (stf.GetStateFromFrenetState(dest_fs, &dest_state) != kSuccess) {
444  return kWrongStatus;
445  }
446 
447  decimal_t look_ahead_dist =
448  (dest_state.vec_position - current_state.vec_position).norm();
449  decimal_t cur_to_dest_angle =
450  vec2d_to_angle(dest_state.vec_position - current_state.vec_position);
451  decimal_t angle_diff =
452  normalize_angle(cur_to_dest_angle - current_state.angle);
453  control::PurePursuitControl::CalculateDesiredSteer(
454  wheelbase_len, angle_diff, look_ahead_dist, steer);
455  return kSuccess;
456  }

518  static ErrorType CalculateDesiredState(const State& current_state,
519  const decimal_t steer,
520  const decimal_t velocity,
521  const decimal_t wheelbase_len,
