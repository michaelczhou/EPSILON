46 /*
47  * The two systems are needed, since for steppers with more than
48  * one internal step it is difficult to calculate the exact result
49  *
50  * system1 is suited for euler
51  */
52 struct system1
53 {
54  template< class State , class Deriv >
55  void operator()( const State &x_ , Deriv &dxdt_ , double t )
56  {
57  typename boost::range_iterator< const State >::type x = boost::begin( x_ );
58  typename boost::range_iterator< Deriv >::type dxdt = boost::begin( dxdt_ );
59 
60  dxdt[0] = x[0];
61  dxdt[1] = 2.0;
62  dxdt[2] = 3.0;
63  }
64 
65  template< class State , class Deriv >
66  void operator()( const State &x_ , const Deriv &dxdt_ , double t )
67  {
68  typename boost::range_iterator< const State >::type x = boost::begin( x_ );
69  typename boost::range_iterator< Deriv >::type dxdt = boost::begin( dxdt_ );
70 
71  dxdt[0] = x[0];
72  dxdt[1] = 2.0;
73  dxdt[2] = 3.0;
74  }
75 };
