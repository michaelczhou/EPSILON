73  void do_step( System system , state_type &x , time_type t , time_type dt )
74  {
75  typedef typename odeint::unwrap_reference< System >::type system_type;
76  typedef typename odeint::unwrap_reference< typename system_type::first_type >::type deriv_func_type;
77  typedef typename odeint::unwrap_reference< typename system_type::second_type >::type jacobi_func_type;
78  system_type &sys = system;
79  deriv_func_type &deriv_func = sys.first;
80  jacobi_func_type &jacobi_func = sys.second;
81 
82  m_resizer.adjust_size( x , detail::bind( &stepper_type::template resize_impl<state_type> , detail::ref( *this ) , detail::_1 ) );
83 
84  for( size_t i=0 ; i<x.size() ; ++i )
85  m_pm.m_v[i] = i;
86 
87  t += dt;
88 
89  // apply first Newton step
90  deriv_func( x , m_dxdt.m_v , t );
91 
92  m_b.m_v = dt * m_dxdt.m_v;
93 
94  jacobi_func( x , m_jacobi.m_v , t );
95  m_jacobi.m_v *= dt;
96  m_jacobi.m_v -= boost::numeric::ublas::identity_matrix< value_type >( x.size() );
97 
98  solve( m_b.m_v , m_jacobi.m_v );
99 
100  m_x.m_v = x - m_b.m_v;
101 
102  // iterate Newton until some precision is reached
103  // ToDo: maybe we should apply only one Newton step -> linear implicit one-step scheme
104  while( boost::numeric::ublas::norm_2( m_b.m_v ) > m_epsilon )
105  {
106  deriv_func( m_x.m_v , m_dxdt.m_v , t );
107  m_b.m_v = x - m_x.m_v + dt*m_dxdt.m_v;
108 
109  // simplified version, only the first Jacobian is used
110  // jacobi( m_x , m_jacobi , t );
111  // m_jacobi *= dt;
112  // m_jacobi -= boost::numeric::ublas::identity_matrix< value_type >( x.size() );
113 
114  solve( m_b.m_v , m_jacobi.m_v );
115 
116  m_x.m_v -= m_b.m_v;
117  }
118  x = m_x.m_v;
119  }
