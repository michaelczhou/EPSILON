129 {
130  template< typename Value , typename T1 , typename T2 >
131  Value operator()( Value val , T1 t1 , T2 t2 ) const
132  {
133  return val + t1 + t2;
134  }
135 
136  template< typename Value , typename T1 , typename T2 , typename T3 >
137  Value operator()( Value val , T1 t1 , T2 t2 , T3 t3 ) const
138  {
139  return val + t1 + t2 + t3 ;
140  }
141

143  Value operator()( Value val , T1 t1 , T2 t2 , T3 t3 , T4 t4 ) const
