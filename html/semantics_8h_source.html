173 struct ProbDistOfLatBehaviors {
174  bool is_valid = false;
175  std::unordered_map<LateralBehavior, decimal_t, EnumClassHash> probs{
176  {common::LateralBehavior::kLaneChangeLeft, 0.0},
177  {common::LateralBehavior::kLaneChangeRight, 0.0},
178  {common::LateralBehavior::kLaneKeeping, 0.0}};
179 
180  void SetEntry(const LateralBehavior &beh, const decimal_t &val) {
181  probs[beh] = val;
182  }
183 
184  bool CheckIfNormalized() const {
185  decimal_t sum = 0.0;
186  for (const auto &entry : probs) {
187  sum += entry.second;
188  }
189  if (fabs(sum - 1.0) < kEPS) {
190  return true;
191  } else {
192  return false;
193  }
194  }
195 
196  bool GetMaxProbBehavior(LateralBehavior *beh) const {
197  if (!is_valid) return false;
198 
199  decimal_t max_prob = -1.0;
200  LateralBehavior max_beh;
201  for (const auto &entry : probs) {
202  if (entry.second > max_prob) {
203  max_prob = entry.second;
204  max_beh = entry.first;
205  }
206  }
207  *beh = max_beh;
208  return true;
209  }
