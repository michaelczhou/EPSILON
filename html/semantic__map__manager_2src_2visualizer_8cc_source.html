363 void Visualizer::VisualizeIntentionPrediction(
364  const ros::Time &stamp,
365  const common::SemanticVehicleSet &semantic_vehicles) {
366  visualization_msgs::MarkerArray mks;
367  int cnt = 0;
368  for (const auto &p_sv : semantic_vehicles.semantic_vehicles) {
369  auto semantic_vehicle = p_sv.second;
370  // * lateral behavior prediction
371  common::State state = semantic_vehicle.vehicle.state();
372  geometry_msgs::Point pt0, pt1;
373  decimal_t z = 2.0;
374  for (const auto &entry : semantic_vehicle.probs_lat_behaviors.probs) {
375  common::LateralBehavior beh = entry.first;
376  decimal_t prob = entry.second;
377 
378  if (prob < kEPS) continue;
379 
380  decimal_t angle_offset = 0.0;
381  decimal_t length = prob * 2.0;
382 
383  if (beh == common::LateralBehavior::kLaneChangeRight) {
384  angle_offset = -kPi / 2.0;
385  } else if (beh == common::LateralBehavior::kLaneChangeLeft) {
386  angle_offset = +kPi / 2.0;
387  } else if (beh == common::LateralBehavior::kLaneKeeping) {
388  angle_offset = 0.0;
389  }
390  pt0.x = state.vec_position(0);
391  pt0.y = state.vec_position(1);
392  pt0.z = z;
393 
394  pt1.x = state.vec_position(0) + cos(state.angle + angle_offset) * length;
395  pt1.y = state.vec_position(1) + sin(state.angle + angle_offset) * length;
396  pt1.z = z;
397 
398  visualization_msgs::Marker behavior_mk;
399  behavior_mk.id = cnt++;
400  behavior_mk.type = visualization_msgs::Marker::ARROW;
401  behavior_mk.action = visualization_msgs::Marker::MODIFY;
402  behavior_mk.points.push_back(pt0);
403  behavior_mk.points.push_back(pt1);
404  behavior_mk.scale.x = 0.2;
405  behavior_mk.scale.y = 0.5;
406  common::VisualizationUtil::FillColorInMarker(common::cmap.at("yellow"),
407  &behavior_mk);
408  mks.markers.push_back(behavior_mk);
409  }
410  }
411  int num_mks = static_cast<int>(mks.markers.size());
412  common::VisualizationUtil::FillHeaderIdInMarkerArray(
413  stamp, std::string("/map"), last_intention_marker_cnt_, &mks);
414  last_intention_marker_cnt_ = num_mks;
415  pred_intention_vis_pub_.publish(mks);
416 }
