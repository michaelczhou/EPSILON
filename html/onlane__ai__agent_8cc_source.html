75 void PublishControl() {
76  if (!has_init_state) return;
77  if (p_bp_server_ == nullptr) return;
78  if (p_ctrl_input_smm_buff_ == nullptr) return;
79 
80  bool is_map_updated = false;
81  decimal_t previous_stamp = last_smm.time_stamp();
82  while (p_ctrl_input_smm_buff_->try_dequeue(last_smm)) {
83  is_map_updated = true;
84  }
85  if (!is_map_updated) return;
86 
87  decimal_t delta_t = last_smm.time_stamp() - previous_stamp;
88  if (delta_t > 100.0 / fs_work_rate) delta_t = 1.0 / fs_work_rate;
89 
90  decimal_t command_vel = p_bp_server_->reference_desired_velocity();
91  decimal_t speed_limit;
92  if (last_smm.GetSpeedLimit(last_smm.ego_vehicle().state(),
93  last_smm.ego_behavior().ref_lane,
94  &speed_limit) == kSuccess) {
95  command_vel = std::min(speed_limit, command_vel);
96  }
97  common::Vehicle ego_vehicle = last_smm.ego_vehicle();
98  ego_vehicle.set_state(desired_state);
99  // ~ Add noise
100  // command_vel = std::max(command_vel + vel_noise, 0.0);
101  sim_param.idm_param.kDesiredVelocity = command_vel;
102 
103  common::Vehicle leading_vehicle;
104  common::State state;
105  decimal_t distance_residual_ratio = 0.0;
106  const decimal_t lat_range = 2.2;
107  last_smm.GetLeadingVehicleOnLane(last_smm.ego_behavior().ref_lane,
108  desired_state,
109  last_smm.surrounding_vehicles(), lat_range,
110  &leading_vehicle, &distance_residual_ratio);
111  if (planning::OnLaneForwardSimulation::PropagateOnce(
112  common::StateTransformer(last_smm.ego_behavior().ref_lane),
113  ego_vehicle, leading_vehicle, delta_t, sim_param,
114  &state) != kSuccess) {
115  printf("[AiAgent]Err-Simulation error (with leading vehicle).\n");
116  return;
117  }
118 
119  common::VehicleControlSignal ctrl(state);
120  {
121  vehicle_msgs::ControlSignal ctrl_msg;
122  vehicle_msgs::Encoder::GetRosControlSignalFromControlSignal(
123  ctrl, ros::Time::now(), std::string("map"), &ctrl_msg);
124  ctrl_signal_pub_.publish(ctrl_msg);
125  }
126  desired_state = ctrl.state;
127 
128  // visualization
129  {
130  ros::Time tnow = ros::Time::now();
131  if (tnow >= next_vis_pub_time) {
132  next_vis_pub_time += ros::Duration(1.0 / visualization_msg_rate);
133  p_smm_vis_->VisualizeDataWithStamp(tnow, last_smm);
134  p_smm_vis_->SendTfWithStamp(tnow, last_smm);
135  }
136  }
137 }
