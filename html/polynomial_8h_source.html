88  inline decimal_t J(decimal_t s, int d) const {
89  if (d == 3) {
90  // integration of squared jerk
91  return coeff_(0) * coeff_(0) / 20.0 * pow(s, 5) +
92  coeff_(0) * coeff_(1) / 4 * pow(s, 4) +
93  (coeff_(1) * coeff_(1) + coeff_(0) * coeff_(2)) / 3 * pow(s, 3) +
94  coeff_(1) * coeff_(2) * s * s + coeff_(2) * coeff_(2) * s;
95  } else if (d == 2) {
96  // integration of squared accleration
97  return coeff_(0) * coeff_(0) / 252 * pow(s, 7) +
98  coeff_(0) * coeff_(1) / 36 * pow(s, 6) +
99  (coeff_(1) * coeff_(1) / 20 + coeff_(0) * coeff_(2) / 15) *
100  pow(s, 5) +
101  (coeff_(0) * coeff_(3) / 12 + coeff_(1) * coeff_(2) / 4) *
102  pow(s, 4) +
103  (coeff_(2) * coeff_(2) / 3 + coeff_(1) * coeff_(3) / 3) *
104  pow(s, 3) +
105  coeff_(2) * coeff_(3) * s * s + coeff_(3) * coeff_(3) * s;
106  } else {
107  assert(false);
108  }
109  return 0.0;
110  }
