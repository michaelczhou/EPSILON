44 void lorenz( const state_type &x , state_type &dxdt , const value_type t )
45 {
46  const value_type sigma( 10.0 );
47  const value_type R( 28.0 );
48  const value_type b( value_type( 8.0 ) / value_type( 3.0 ) );
49 
50  dxdt[0] = sigma * ( x[1] - x[0] );
51  dxdt[1] = R * x[0] - x[1] - x[0] * x[2];
52  dxdt[2] = -b * x[2] + x[0] * x[1];
53 }
