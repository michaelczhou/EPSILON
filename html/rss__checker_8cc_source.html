5 ErrorType RssChecker::CalculateSafeLongitudinalDistance(
6  const decimal_t ego_vel, const decimal_t other_vel,
7  const LongitudinalDirection& direction, const RssConfig& config,
8  decimal_t* distance) {
9  decimal_t ret = 0.0;
10  decimal_t ego_vel_abs = fabs(ego_vel);
11  decimal_t other_vel_abs = fabs(other_vel);
12  decimal_t ego_vel_at_response_time =
13  ego_vel_abs + config.longitudinal_acc_max * config.response_time;
14  decimal_t other_vel_at_response_time =
15  other_vel_abs + config.longitudinal_acc_max * config.response_time;
16 
17  decimal_t ego_distance_driven, other_distance_driven;
18  if (direction == Front) {
19  ego_distance_driven =
20  (ego_vel_abs + ego_vel_at_response_time) / 2.0 * config.response_time +
21  ego_vel_at_response_time * ego_vel_at_response_time /
22  (2 * config.longitudinal_brake_min);
23  if (ego_vel >= 0.0 && other_vel >= 0.0) {
24  // ego vehicle ==> other vehicle ->
25  other_distance_driven =
26  (other_vel_abs * other_vel_abs) / (2 * config.longitudinal_brake_max);
27  ret = ego_distance_driven - other_distance_driven;
28  } else if (ego_vel >= 0.0 && other_vel <= 0.0) {
29  // ego vehicle ==> <-- other vehicle
30  other_distance_driven = (other_vel_abs + other_vel_at_response_time) /
31  2.0 * config.response_time +
32  other_vel_at_response_time *
33  other_vel_at_response_time /
34  (2 * config.longitudinal_brake_min);
35  ret = ego_distance_driven + other_distance_driven;
36  } else {
37  // printf("[RssChecker]Currently do not support rear gear %lf.\n",
38  // ego_vel);
39  ret = 0.0;
40  }
41  } else if (direction == Rear) {
42  ego_distance_driven =
43  ego_vel_abs * ego_vel_abs / (2 * config.longitudinal_brake_max);
44  if (ego_vel >= 0.0 && other_vel >= 0.0) {
45  // other car --> ego car ==>
46  other_distance_driven = (other_vel_abs + other_vel_at_response_time) /
47  2.0 * config.response_time +
48  other_vel_at_response_time *
49  other_vel_at_response_time /
50  (2 * config.longitudinal_brake_min);
51  ret = other_distance_driven - ego_distance_driven;
52  } else if (ego_vel >= 0.0 && other_vel <= 0.0) {
53  ret = 0.0;
54  } else {
55  // printf("[RssChecker]Currently do not support rear gear %lf.\n",
56  // ego_vel);
57  ret = 0.0;
58  }
59  }
60  *distance = ret > 0.0 ? ret : 0.0;
61  return kSuccess;
62 }

64 ErrorType RssChecker::CalculateSafeLongitudinalVelocity(
65  const decimal_t other_vel, const LongitudinalDirection& direction,
66  const decimal_t& lon_distance_abs, const RssConfig& config,
67  decimal_t* ego_vel_low, decimal_t* ego_vel_upp) {
68  decimal_t other_vel_abs = fabs(other_vel);
69  decimal_t other_vel_at_response_time =
70  other_vel_abs + config.longitudinal_acc_max * config.response_time;
71  decimal_t other_distance_driven;
72  if (direction == Front) {
73  if (other_vel >= 0.0) {
74  // ego ---->(lon_distance) other --->
75  // other hard brake
76  other_distance_driven =
77  (other_vel_abs * other_vel_abs) / (2 * config.longitudinal_brake_max);
78  // ego has vel upp
79  decimal_t a = 1.0 / (2.0 * config.longitudinal_brake_min);
80  decimal_t b = config.response_time +
81  (config.longitudinal_acc_max * config.response_time /
82  config.longitudinal_brake_min);
83  decimal_t c = 0.5 *
84  (config.longitudinal_acc_max +
85  pow(config.longitudinal_acc_max, 2) /
86  config.longitudinal_brake_min) *
87  pow(config.response_time, 2) -
88  other_distance_driven - lon_distance_abs;
89  *ego_vel_upp = (-b + sqrt(pow(b, 2) - 4 * a * c)) / (2 * a);
90  *ego_vel_low = 0.0;
91  } else {
92  // ego ----> <---- other
93  other_distance_driven = (other_vel_abs + other_vel_at_response_time) /
94  2.0 * config.response_time +
95  other_vel_at_response_time *
96  other_vel_at_response_time /
97  (2 * config.longitudinal_brake_min);
98  if (other_distance_driven > lon_distance_abs) {
99  *ego_vel_upp = 0.0;
100  *ego_vel_low = 0.0;
101  } else {
102  decimal_t a = 1.0 / (2.0 * config.longitudinal_brake_min);
103  decimal_t b = config.response_time +
104  (config.longitudinal_acc_max * config.response_time /
105  config.longitudinal_brake_min);
106  decimal_t c = 0.5 *
107  (config.longitudinal_acc_max +
108  pow(config.longitudinal_acc_max, 2) /
109  config.longitudinal_brake_min) *
110  pow(config.response_time, 2) -
111  (lon_distance_abs - other_distance_driven);
112  *ego_vel_upp = (-b + sqrt(pow(b, 2) - 4 * a * c)) / (2 * a);
113  *ego_vel_low = 0.0;
114  }
115  }
116  } else {
117  if (other_vel >= 0.0) {
118  // other ---> ego--->
119  other_distance_driven = (other_vel_abs + other_vel_at_response_time) /
120  2.0 * config.response_time +
121  other_vel_at_response_time *
122  other_vel_at_response_time /
123  (2 * config.longitudinal_brake_min);
124  if (other_distance_driven < lon_distance_abs) {
125  *ego_vel_upp = kInf;
126  *ego_vel_low = 0.0;
127  } else {
128  *ego_vel_upp = kInf;
129  *ego_vel_low = sqrt(2 * config.longitudinal_brake_max *
130  (other_distance_driven - lon_distance_abs));
131  }
132  } else {
133  // <----other ego-->
134  *ego_vel_upp = kInf;
135  *ego_vel_low = 0.0;
136  }
137  }
138  return kSuccess;
139 }

141 ErrorType RssChecker::CalculateSafeLateralDistance(
142  const decimal_t ego_vel, const decimal_t other_vel,
143  const LateralDirection& direction, const RssConfig& config,
144  decimal_t* distance) {
145  decimal_t ret = 0.0;
146  decimal_t ego_lat_vel_abs = fabs(ego_vel);
147  decimal_t other_lat_vel_abs = fabs(other_vel);
148  decimal_t distance_correction = config.lateral_miu;
149  decimal_t ego_lat_vel_at_response_time =
150  ego_lat_vel_abs + config.response_time * config.lateral_acc_max;
151  decimal_t other_lat_vel_at_response_time =
152  other_lat_vel_abs + config.response_time * config.lateral_acc_max;
153  decimal_t ego_active_brake_distance =
154  ego_lat_vel_abs * ego_lat_vel_abs / (2 * config.lateral_brake_max);
155  decimal_t ego_passive_brake_distance =
156  (ego_lat_vel_abs + ego_lat_vel_at_response_time) / 2.0 *
157  config.response_time +
158  ego_lat_vel_at_response_time * ego_lat_vel_at_response_time /
159  (2 * config.lateral_brake_min);
160  decimal_t other_active_brake_distance =
161  other_lat_vel_abs * other_lat_vel_abs / (2 * config.lateral_brake_max);
162  decimal_t other_passive_brake_distance =
163  (other_lat_vel_abs + other_lat_vel_at_response_time) / 2.0 *
164  config.response_time +
165  other_lat_vel_at_response_time * other_lat_vel_at_response_time /
166  (2 * config.lateral_brake_min);
167  if (direction == Right) {
168  if (ego_vel >= 0.0 && other_vel >= 0.0) {
169  // -------------------------------
170  // ego ^^^^^^^^
171  // -------------------------------
172  // other ^^^^^^^^
173  // -------------------------------
174  ret = other_passive_brake_distance - ego_active_brake_distance;
175  } else if (ego_vel >= 0.0 && other_vel < 0.0) {
176  // -------------------------------
177  // ego ^^^^^^^^
178  // -------------------------------
179  // other vvvvvvvv
180  // -------------------------------
181  ret = 0.0;
182  } else if (ego_vel < 0.0 && other_vel < 0.0) {
183  // -------------------------------
184  // ego vvvvvvvv
185  // -------------------------------
186  // other vvvvvvvv
187  // -------------------------------
188  ret = ego_passive_brake_distance - other_active_brake_distance;
189  } else if (ego_vel < 0.0 && other_vel >= 0.0) {
190  // -------------------------------
191  // ego vvvvvvvv
192  // -------------------------------
193  // other ^^^^^^^^
194  // -------------------------------
195  ret = ego_passive_brake_distance + other_passive_brake_distance;
196  } else {
197  // printf("[RssChecker]Lat Error configuration.\n");
198  // assert(false);
199  ret = 0.0;
200  }
201  } else if (direction == Left) {
202  if (ego_vel >= 0.0 && other_vel >= 0.0) {
203  // -------------------------------
204  // other ^^^^^^^^
205  // -------------------------------
206  // ego ^^^^^^^^
207  // -------------------------------
208  ret = ego_passive_brake_distance - other_active_brake_distance;
209  } else if (ego_vel >= 0.0 && other_vel < 0.0) {
210  // -------------------------------
211  // other vvvvvvvv
212  // -------------------------------
213  // ego ^^^^^^^^
214  // -------------------------------
215  ret = ego_passive_brake_distance + other_passive_brake_distance;
216  } else if (ego_vel < 0.0 && other_vel < 0.0) {
217  // -------------------------------
218  // other vvvvvvvv
219  // -------------------------------
220  // ego vvvvvvvv
221  // -------------------------------
222  ret = other_passive_brake_distance - ego_active_brake_distance;
223  } else if (ego_vel < 0.0 && other_vel >= 0.0) {
224  // -------------------------------
225  // other ^^^^^^^^
226  // -------------------------------
227  // ego vvvvvvvv
228  // -------------------------------
229  ret = 0.0;
230  } else {
231  // printf("[RssChecker]Lat Error configuration.\n");
232  // assert(false);
233  ret = 0.0;
234  }
235  }
236  ret = ret > 0.0 ? ret : 0.0;
237  ret += distance_correction;
238  *distance = ret;
239  return kSuccess;
240 }

259  const FrenetState& other_fs,
260  const RssConfig& config, bool* is_safe) {
261  LongitudinalDirection lon_direct;
262  LateralDirection lat_direct;
263  if (ego_fs.vec_s[0] >= other_fs.vec_s[0]) {
264  lon_direct = Rear;
265  } else {
266  lon_direct = Front;
267  }
268 
269  if (ego_fs.vec_dt[0] >= other_fs.vec_dt[0]) {
270  lat_direct = Right;
271  } else {
272  lat_direct = Left;
273  }
274 
275  std::vector<decimal_t> ego_vels{ego_fs.vec_s[1], ego_fs.vec_dt[1]};
276  std::vector<decimal_t> other_vels{other_fs.vec_s[1], other_fs.vec_dt[1]};
277  std::vector<decimal_t> safe_distances;
278  CalculateRssSafeDistances(ego_vels, other_vels, lon_direct, lat_direct,
279  config, &safe_distances);
280 
281  if (fabs(ego_fs.vec_s[0] - other_fs.vec_s[0]) < safe_distances[0] &&
282  fabs(ego_fs.vec_dt[0] - other_fs.vec_dt[0]) < safe_distances[1]) {
283  *is_safe = false;
284  } else {
285  *is_safe = true;
286  }
287  return kSuccess;

290 ErrorType RssChecker::RssCheck(const Vehicle& ego_vehicle,

293  bool* is_safe, LongitudinalViolateType* lon_type,
294  decimal_t* rss_vel_low, decimal_t* rss_vel_up) {

296  // TODO(lu.zhang): construct stf is a little bit heavy
297  // StateTransformer stf(ref_lane);

301  }
302  if (stf.GetFrenetStateFromState(other_vehicle.state(), &other_fs) !=
303  kSuccess) {
304  printf("[RssChecker]other %d not on ref lane.\n", other_vehicle.id());

306  }
307 
308  LongitudinalDirection lon_direct;
309  LateralDirection lat_direct;
310 
311  if (ego_fs.vec_s[0] >= other_fs.vec_s[0]) {
312  lon_direct = Rear;
313  } else {
314  lon_direct = Front;
315  }
316 
317  if (ego_fs.vec_dt[0] >= other_fs.vec_dt[0]) {
318  lat_direct = Right;
319  } else {
320  lat_direct = Left;
321  }
322 
323  if (ego_fs.vec_s[1] < 0.0) {
324  *is_safe = true;
325  *lon_type = LongitudinalViolateType::Legal;
326  *rss_vel_up = 0.0;
327  *rss_vel_low = 0.0;
328  return kSuccess;
329  }
330 
331  decimal_t safe_lat_distance;
332  CalculateSafeLateralDistance(ego_fs.vec_dt[1], other_fs.vec_dt[1], lat_direct,
333  config, &safe_lat_distance);
334  safe_lat_distance +=
335  0.5 * (ego_vehicle.param().width() + other_vehicle.param().width());
336 
337  if (fabs(ego_fs.vec_dt[0] - other_fs.vec_dt[0]) > safe_lat_distance) {
338  *is_safe = true;
339  *lon_type = LongitudinalViolateType::Legal;
340  *rss_vel_up = 0.0;
341  *rss_vel_low = 0.0;
342  return kSuccess;
343  }
344 
345  decimal_t lon_distance_abs, ego_vel_low, ego_vel_upp;
346  if (lon_direct == Rear) {
347  decimal_t other_rear_wheel_to_front_bump =
348  0.5 * other_vehicle.param().length() + other_vehicle.param().d_cr();
349  decimal_t ego_rear_wheel_to_back_bump =
350  fabs(0.5 * ego_vehicle.param().length() - ego_vehicle.param().d_cr());
351  lon_distance_abs = fabs(ego_fs.vec_s[0] - other_fs.vec_s[0]) -
352  other_rear_wheel_to_front_bump -
353  ego_rear_wheel_to_back_bump;
354  } else if (lon_direct == Front) {
355  decimal_t ego_rear_wheel_to_front_bump =
356  0.5 * ego_vehicle.param().length() + ego_vehicle.param().d_cr();
357  decimal_t other_rear_wheel_to_back_bump = fabs(
358  0.5 * other_vehicle.param().length() - other_vehicle.param().d_cr());
359  lon_distance_abs = fabs(ego_fs.vec_s[0] - other_fs.vec_s[0]) -
360  ego_rear_wheel_to_front_bump -
361  other_rear_wheel_to_back_bump;
362  }
363 
364  if (lon_distance_abs < 0.0 && lon_direct == Front) {
365  *is_safe = false;
366  *lon_type = LongitudinalViolateType::TooFast;
367  *rss_vel_up = 0.0;
368  *rss_vel_low = 0.0;
369  return kSuccess;
370  }
371 
372  CalculateSafeLongitudinalVelocity(other_fs.vec_s[1], lon_direct,
373  lon_distance_abs, config, &ego_vel_low,
374  &ego_vel_upp);
375  if (ego_fs.vec_s[1] > ego_vel_upp + kEPS) {
376  *is_safe = false;
377  *lon_type = LongitudinalViolateType::TooFast;
378  *rss_vel_up = ego_vel_upp;
379  *rss_vel_low = ego_vel_low;
380  } else if (ego_fs.vec_s[1] < ego_vel_low - kEPS) {
381  *is_safe = false;
382  *lon_type = LongitudinalViolateType::TooSlow;
383  *rss_vel_up = ego_vel_upp;
384  *rss_vel_low = ego_vel_low;
385  } else {
386  *is_safe = true;
387  *lon_type = LongitudinalViolateType::Legal;
388  *rss_vel_up = 0.0;
389  *rss_vel_low = 0.0;
390  }
391  return kSuccess;
392 }
