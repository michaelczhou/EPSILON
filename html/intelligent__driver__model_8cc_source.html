11  cur_state.v * param.kDesiredHeadwayTime +
12  cur_state.v * (cur_state.v - cur_state.v_front) /

23 ErrorType IntelligentDriverModel::GetIIdmDesiredAcceleration(
24  const IntelligentDriverModel::Param &param,
25  const IntelligentDriverModel::State &cur_state, decimal_t *acc) {
26  // The Improved IntelligentDriverModel (IIDM) tries to address two
27  // deficiencies of the original IDM model:
28  // 1) If the actual speed exceeds the desired speed (e.g., after entering a
29  // zone with a reduced speed limit), the deceleration is unrealistically
30  // large, particularly for large values of the acceleration exponent δ.
31  // 2) Near the desired speed v0, the steady-state gap becomes much
32  // greater than s∗(v, 0) = s0 + vT so that the model parameter T loses its
33  // meaning as the desired time gap. This means that a platoon of identical
34  // drivers and vehicles disperses much more than observed. Moreover, not all
35  // cars will reach the desired speed
36  decimal_t a_free =
37  cur_state.v <= param.kDesiredVelocity
38  ? param.kAcceleration *
39  (1 - pow(cur_state.v / param.kDesiredVelocity, param.kExponent))
40  : -param.kComfortableBrakingDeceleration *
41  (1 - pow(param.kDesiredVelocity / cur_state.v,
42  param.kAcceleration * param.kExponent /
43  param.kComfortableBrakingDeceleration));
44  decimal_t s_alpha =
45  std::max(0.0, cur_state.s_front - cur_state.s - param.kVehicleLength);
46  decimal_t z =
47  (param.kMinimumSpacing +
48  std::max(0.0,
49  cur_state.v * param.kDesiredHeadwayTime +
50  cur_state.v * (cur_state.v - cur_state.v_front) /
51  (2.0 * sqrt(param.kAcceleration *
52  param.kComfortableBrakingDeceleration)))) /
53  s_alpha;
54  decimal_t a_out =
55  cur_state.v <= param.kDesiredVelocity
56  ? (z >= 1.0
57  ? param.kAcceleration * (1 - pow(z, 2))
58  : a_free * (1 - pow(z, 2.0 * param.kAcceleration / a_free)))
59  : (z >= 1.0 ? a_free + param.kAcceleration * (1 - pow(z, 2))
60  : a_free);
61  a_out = std::max(std::min(param.kAcceleration, a_out),
62  -param.kHardBrakingDeceleration);
63  *acc = a_out;
64  return kSuccess;
65 }

67 ErrorType IntelligentDriverModel::GetAccDesiredAcceleration(
68  const IntelligentDriverModel::Param &param,
69  const IntelligentDriverModel::State &cur_state, decimal_t *acc) {
70  decimal_t acc_iidm;
71  GetIIdmDesiredAcceleration(param, cur_state, &acc_iidm);
72 
73  // ~ Here we simply use a constant dec, ego comfortable dec, as the acc of
74  // ~ leading vehicle.
75  decimal_t ds = std::max(0.0, cur_state.s_front - cur_state.s);
76  decimal_t acc_cah =
77  (cur_state.v * cur_state.v * -param.kComfortableBrakingDeceleration) /
78  (cur_state.v_front * cur_state.v_front -
79  2 * ds * -param.kComfortableBrakingDeceleration);
80 
81  decimal_t coolness = 0.99;
82 
83  if (acc_iidm >= acc_cah) {
84  *acc = acc_iidm;
85  } else {
86  *acc =
87  (1 - coolness) * acc_iidm +
88  coolness * (acc_cah - param.kComfortableBrakingDeceleration *
89  tanh((acc_iidm - acc_cah) /
90  -param.kComfortableBrakingDeceleration));
91  }
92  return kSuccess;
93 }
