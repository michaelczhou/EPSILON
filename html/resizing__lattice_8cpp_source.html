48  void operator()( const coord_type &q , coord_type &dpdt )
49  {
50  // calculate dpdt = -dH/dq of this hamiltonian system
51  // dp_i/dt = - V_i * q_i^3 - beta*(q_i - q_{i-1})^3 + beta*(q_{i+1} - q_i)^3
52  const int N = q.size();
53  double diff = q[0] - q[N-1];
54  for( int i=0 ; i<N ; ++i )
55  {
56  dpdt[i] = - m_pot[m_pot_start_index+i] * q[i]*q[i]*q[i] -
57  m_beta * diff*diff*diff;
58  diff = q[(i+1) % N] - q[i];
59  dpdt[i] += m_beta * diff*diff*diff;
60  }
61  }
62 
63  void energy_distribution( const coord_type &q , const coord_type &p , coord_type &energies )
64  {
65  // computes the energy per lattice site normalized by total energy
66  const size_t N = q.size();
67  double en = 0.0;
68  for( size_t i=0 ; i<N ; i++ )
69  {
70  const double diff = q[(i+1) % N] - q[i];
71  energies[i] = p[i]*p[i]/2.0
72  + m_pot[m_pot_start_index+i]*q[i]*q[i]*q[i]*q[i]/4.0
73  + m_beta/4.0 * diff*diff*diff*diff;
74  en += energies[i];
75  }
76  en = 1.0/en;
77  for( size_t i=0 ; i<N ; i++ )
78  {
79  energies[i] *= en;
80  }
81  }
82 
83  double energy( const coord_type &q , const coord_type &p )
84  {
85  // calculates the total energy of the excitation
86  const size_t N = q.size();
87  double en = 0.0;
88  for( size_t i=0 ; i<N ; i++ )
89  {
90  const double diff = q[(i+1) % N] - q[i];
91  en += p[i]*p[i]/2.0
92  + m_pot[m_pot_start_index+i]*q[i]*q[i]*q[i]*q[i] / 4.0
93  + m_beta/4.0 * diff*diff*diff*diff;
94  }
95  return en;
96  }

117 int main()
118 {
119  //[ resizing_lattice_initialize
120  //start with 60 sites
121  const int N_start = 60;
122  coord_type q( N_start , 0.0 );
123  q.reserve( max_N );
124  coord_type p( N_start , 0.0 );
125  p.reserve( max_N );
126  // start with uniform momentum distribution over 20 sites
127  fill( p.begin()+20 , p.end()-20 , 1.0/sqrt(20.0) );
128 
129  coord_type distr( N_start , 0.0 );
130  distr.reserve( max_N );
131 
132  // create the system
133  compacton_lattice lattice( max_N , beta , (max_N-N_start)/2 );
134 
135  //create the stepper, note that we use an always_resizer because state size might change during steps
136  typedef symplectic_rkn_sb3a_mclachlan< coord_type , coord_type , double , coord_type , coord_type , double ,
137  range_algebra , default_operations , always_resizer > hamiltonian_stepper;
138  hamiltonian_stepper stepper;
139  hamiltonian_stepper::state_type state = make_pair( q , p );
140  //]
141 
142  //[ resizing_lattice_steps_loop
143  double t = 0.0;
144  const double dt = 0.1;
145  const int steps = 10000;
146  for( int step = 0 ; step < steps ; ++step )
147  {
148  stepper.do_step( boost::ref(lattice) , state , t , dt );
149  lattice.energy_distribution( state.first , state.second , distr );
150  if( distr[10] > 1E-150 )
151  {
152  do_resize( state.first , state.second , distr , state.first.size()+20 );
153  rotate( state.first.begin() , state.first.end()-20 , state.first.end() );
154  rotate( state.second.begin() , state.second.end()-20 , state.second.end() );
155  lattice.change_pot_start( -20 );
156  cout << t << ": resized left to " << distr.size() << ", energy = " << lattice.energy( state.first , state.second ) << endl;
157  }
158  if( distr[distr.size()-10] > 1E-150 )
159  {
160  do_resize( state.first , state.second , distr , state.first.size()+20 );
161  cout << t << ": resized right to " << distr.size() << ", energy = " << lattice.energy( state.first , state.second ) << endl;
162  }
163  t += dt;
164  }
165  //]
166 
167  cout << "final lattice size: " << distr.size() << ", final energy: " << lattice.energy( state.first , state.second ) << endl;
168 }
