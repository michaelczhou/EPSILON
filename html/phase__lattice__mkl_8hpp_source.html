34  phase_lattice_mkl() : m_epsilon( 6.0/(N*N) ) // should be < 8/N^2 to see phase locking
35  {
36  for( size_t i=1 ; i<N-1 ; ++i )
37  m_omega[i] = m_epsilon*(N-i);
38  }

40  void inline operator()( const state_type &x , state_type &dxdt , const double t )
41  {
42  const int n = x.size();
43 
44  dxdt[0] = m_omega[0] + sin( x[1] - x[0] );
45 
46  vdSub( n-1 , &(x[1]) , &(x[0]) , &(m_tmp[0]) );
47  vdSin( n-1 , &(m_tmp[0]) , &(m_tmp[0]) );
48  vdAdd( n-2 , &(m_tmp[0]) , &(m_tmp[1]) , &(dxdt[1]) );
49  vdAdd( n-2 , &(dxdt[1]) , &(m_omega[1]) , &(dxdt[1]) );
50 
51  dxdt[N-1] = m_omega[N-1] + sin( x[N-1] - x[N-2] );
52  }

54 };
