68 /* generic test for all runge kutta steppers */
69 template< class Stepper >
70 struct perform_runge_kutta_test
71 {
72  void operator()( void )
73  {
74 
75  Stepper stepper;
76  const int o = stepper.order()+1; //order of the error is order of approximation + 1
77 
78  const state_type x0 = {{ 0.0 , 1.0 }};
79  state_type x1;
80  const double t = 0.0;
81  /* do a first step with dt=0.1 to get an estimate on the prefactor of the error dx = f * dt^(order+1) */
82  double dt = 0.5;
83  stepper.do_step( osc() , x0 , t , x1 , dt );
84  const double f = 2.0 * std::abs( sin(dt) - x1[0] ) / std::pow( dt , o ); // upper bound
85 
86  std::cout << o << " , " << f << std::endl;
87 
88  /* as long as we have errors above machine precision */
89  while( f*std::pow( dt , o ) > 1E-16 )
90  {
91  // reset stepper which require resetting (fsal steppers)
92  resetter< typename Stepper::stepper_category >::reset( stepper );
93 
94  stepper.do_step( osc() , x0 , t , x1 , dt );
95  std::cout << "Testing dt=" << dt << std::endl;
96  BOOST_CHECK_LT( std::abs( sin(dt) - x1[0] ) , f*std::pow( dt , o ) );
97  dt *= 0.5;
98  }
99  }
100 };

103 /* generic error test for all runge kutta steppers */
104 template< class Stepper >
105 struct perform_runge_kutta_error_test
106 {
107  void operator()( void )
108  {
109  Stepper stepper;
110  const int o = stepper.error_order()+1; //order of the error is order of approximation + 1
111 
112  const state_type x0 = {{ 0.0 , 1.0 }};
113  state_type x1 , x_err;
114  const double t = 0.0;
115  /* do a first step with dt=0.1 to get an estimate on the prefactor of the error dx = f * dt^(order+1) */
116  double dt = 0.5;
117  stepper.do_step( osc() , x0 , t , x1 , dt , x_err );
118  const double f = 2.0 * std::abs( x_err[0] ) / std::pow( dt , o );
119 
120  std::cout << o << " , " << f << " , " << x0[0] << std::endl;
121 
122  /* as long as we have errors above machine precision */
123  while( f*std::pow( dt , o ) > 1E-16 )
124  {
125  // reset stepper which require resetting (fsal steppers)
126  resetter< typename Stepper::stepper_category >::reset( stepper );
127 
128  stepper.do_step( osc() , x0 , t , x1 , dt , x_err );
129  std::cout << "Testing dt=" << dt << ": " << x_err[1] << std::endl;
130  BOOST_CHECK_SMALL( std::abs( x_err[0] ) , f*std::pow( dt , o ) );
131  dt *= 0.5;
132  }
133  }
