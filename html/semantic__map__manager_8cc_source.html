718  queue.pop_front();
719  expanded_nodes++;

735  if (child_ids.empty()) continue;
736  for (auto &id : child_ids) {
737  if (visited_set.count(id) == 0) {
738  visited_set.insert(id);
739  queue.push_back(id);
740  if (it->second.l_change_avbl && id == it->second.l_lane_id) {
741  num_lc_map.insert(std::make_pair(id, num_lc_map.at(cur_id) + 1));

743  num_lc_map.insert(std::make_pair(id, num_lc_map.at(cur_id) + 1));

746  }
747  }
748  }
749  }
750 
751  if (!is_reachable) {
752  *res = false;
753  } else {
754  *res = true;
755  }
756  return kSuccess;
