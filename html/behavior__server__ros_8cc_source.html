40 void BehaviorPlannerServer::JoyCallback(const sensor_msgs::Joy::ConstPtr& msg) {
41  if (bp_.autonomous_level() < 2) return;
42  if (!is_hmi_enabled_) return;
43  int msg_id;
44  if (std::string("").compare(msg->header.frame_id) == 0) {
45  msg_id = 0;
46  } else {
47  msg_id = std::stoi(msg->header.frame_id);
48  }
49  if (msg_id != ego_id_) return;
50  // ~ buttons[2] --> 1 --> lcl
51  // ~ buttons[1] --> 1 --> lcr
52  // ~ buttons[3] --> 1 --> +1m/s
53  // ~ buttons[0] --> 1 --> -1m/s
54  if (msg->buttons[0] == 0 && msg->buttons[1] == 0 && msg->buttons[2] == 0 &&
55  msg->buttons[3] == 0)
56  return;
57 
58  if (msg->buttons[2] == 1) {
59  bp_.set_hmi_behavior(common::LateralBehavior::kLaneChangeLeft);
60  } else if (msg->buttons[1] == 1) {
61  bp_.set_hmi_behavior(common::LateralBehavior::kLaneChangeRight);
62  } else if (msg->buttons[3] == 1) {
63  bp_.set_user_desired_velocity(bp_.user_desired_velocity() + 1.0);
64  } else if (msg->buttons[0] == 1) {
65  bp_.set_user_desired_velocity(bp_.user_desired_velocity() - 1.0);
66  }
67 }

74 void BehaviorPlannerServer::MainThread() {
75  using namespace std::chrono;
76  system_clock::time_point current_start_time{system_clock::now()};
77  system_clock::time_point next_start_time{current_start_time};
78  const milliseconds interval{static_cast<int>(1000.0 / work_rate_)};

82  PlanCycleCallback();
