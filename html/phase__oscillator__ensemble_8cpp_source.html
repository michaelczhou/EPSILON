31 pair< double , double > calc_mean_field( const container_type &x )
32 {
33  size_t n = x.size();
34  double cos_sum = 0.0 , sin_sum = 0.0;
35  for( size_t i=0 ; i<n ; ++i )
36  {
37  cos_sum += cos( x[i] );
38  sin_sum += sin( x[i] );
39  }
40  cos_sum /= double( n );
41  sin_sum /= double( n );
42 
43  double K = sqrt( cos_sum * cos_sum + sin_sum * sin_sum );
44  double Theta = atan2( sin_sum , cos_sum );
45 
46  return make_pair( K , Theta );
47 }

61  void create_frequencies( double g )
62  {
63  boost::mt19937 rng;
64  boost::cauchy_distribution<> cauchy( 0.0 , g );
65  boost::variate_generator< boost::mt19937&, boost::cauchy_distribution<> > gen( rng , cauchy );
66  generate( m_omega.begin() , m_omega.end() , gen );
67  }

69  void set_epsilon( double epsilon ) { m_epsilon = epsilon; }
70 
71  double get_epsilon( void ) const { return m_epsilon; }
72 
73  void operator()( const container_type &x , container_type &dxdt , double /* t */ ) const
74  {
75  pair< double , double > mean = calc_mean_field( x );
76  for( size_t i=0 ; i<x.size() ; ++i )
77  dxdt[i] = m_omega[i] + m_epsilon * mean.first * sin( mean.second - x[i] );
78  }
