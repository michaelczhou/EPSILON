305  template< class System >
306  std::pair< time_type , time_type > do_step( System system )
307  {
308  const size_t max_count = 1000;
309 
310  if( !m_is_deriv_initialized )
311  {
312  typename odeint::unwrap_reference< System >::type &sys = system;
313  sys( get_current_state() , get_current_deriv() , m_t );
314  m_is_deriv_initialized = true;
315  }
316 
317  controlled_step_result res = fail;
318  m_t_old = m_t;
319  size_t count = 0;
320  do
321  {
322  res = m_stepper.try_step( system , get_current_state() , get_current_deriv() , m_t ,
323  get_old_state() , get_old_deriv() , m_dt );
324  if( count++ == max_count )
325  throw std::overflow_error( "dense_output_controlled_explicit : too much iterations!");
326  }
327  while( res == fail );
328  toggle_current_state();
329  return std::make_pair( m_t_old , m_t );
330  }
