98 ErrorType SscMap::ConstructCorridorUsingInitialTrajectory(
99  GridMap3D *p_grid, const vec_E<common::FsVehicle> &trajs) {
100  // ~ Stage I: Get seeds
101  vec_E<Vec3i> traj_seeds;
102  int num_states = static_cast<int>(trajs.size());
103  if (num_states > 1) {
104  bool first_seed_determined = false;
105  for (int k = 0; k < num_states; ++k) {
106  std::array<decimal_t, 3> p_w = {};
107  if (!first_seed_determined) {
108  decimal_t s_0 = initial_fs_.vec_s[0];
109  decimal_t d_0 = initial_fs_.vec_dt[0];
110  decimal_t t_0 = initial_fs_.time_stamp;
111  std::array<decimal_t, 3> p_w_0 = {s_0, d_0, t_0};
112  auto coord_0 = p_grid->GetCoordUsingGlobalPosition(p_w_0);
113 
114  decimal_t s_1 = trajs[k].frenet_state.vec_s[0];
115  decimal_t d_1 = trajs[k].frenet_state.vec_dt[0];
116  decimal_t t_1 = trajs[k].frenet_state.time_stamp;
117  std::array<decimal_t, 3> p_w_1 = {s_1, d_1, t_1};
118  auto coord_1 = p_grid->GetCoordUsingGlobalPosition(p_w_1);
119  // * remove the states out of range
120  if (!p_grid->CheckCoordInRange(coord_1)) {
121  continue;
122  }
123  // earlier than start time
124  if (coord_1[2] <= 0) {
125  continue;
126  }
127 
128  first_seed_determined = true;
129  traj_seeds.push_back(Vec3i(coord_0[0], coord_0[1], coord_0[2]));
130  traj_seeds.push_back(Vec3i(coord_1[0], coord_1[1], coord_1[2]));
131  } else {
132  decimal_t s = trajs[k].frenet_state.vec_s[0];
133  decimal_t d = trajs[k].frenet_state.vec_dt[0];
134  decimal_t t = trajs[k].frenet_state.time_stamp;
135  p_w = {s, d, t};
136  auto coord = p_grid->GetCoordUsingGlobalPosition(p_w);
137  // * remove the states out of range
138  if (!p_grid->CheckCoordInRange(coord)) {
139  continue;
140  }
141  traj_seeds.push_back(Vec3i(coord[0], coord[1], coord[2]));
142  }
143  }
144  }
145 
146  // ~ Stage II: Inflate cubes
147  common::DrivingCorridor driving_corridor;
148  bool is_valid = true;
149  auto seed_num = static_cast<int>(traj_seeds.size());
150  if (seed_num < 2) {
151  driving_corridor.is_valid = false;
152  driving_corridor_vec_.push_back(driving_corridor);
153  is_valid = false;
154  return kWrongStatus;
155  }
156  for (int i = 0; i < seed_num; ++i) {
157  if (i == 0) {
158  common::AxisAlignedCubeNd<int, 3> cube;
159  GetInitialCubeUsingSeed(traj_seeds[i], traj_seeds[i + 1], &cube);
160  if (!CheckIfCubeIsFree(p_grid, cube)) {
161  LOG(ERROR) << "[Ssc] SccMap - Initial cube is not free, seed id: " << i;
162 
163  common::DrivingCube driving_cube;
164  driving_cube.cube = cube;
165  driving_cube.seeds.push_back(traj_seeds[i]);
166  driving_cube.seeds.push_back(traj_seeds[i + 1]);
167  driving_corridor.cubes.push_back(driving_cube);
168 
169  driving_corridor.is_valid = false;
170  driving_corridor_vec_.push_back(driving_corridor);
171  is_valid = false;
172  break;
173  }
174 
175  std::array<bool, 6> dirs_disabled = {false, false, false,
176  false, false, false};
177  InflateCubeIn3dGrid(p_grid, dirs_disabled, config_.inflate_steps, &cube);
178 
179  common::DrivingCube driving_cube;
180  driving_cube.cube = cube;
181  driving_cube.seeds.push_back(traj_seeds[i]);
182  driving_corridor.cubes.push_back(driving_cube);
183  } else {
184  if (CheckIfCubeContainsSeed(driving_corridor.cubes.back().cube,
185  traj_seeds[i])) {
186  driving_corridor.cubes.back().seeds.push_back(traj_seeds[i]);
187  continue;
188  } else {
189  // ~ Get the last seed in cube
190  Vec3i seed_r = driving_corridor.cubes.back().seeds.back();
191  driving_corridor.cubes.back().seeds.pop_back();
192  // ~ Cut cube on time axis
193  driving_corridor.cubes.back().cube.upper_bound[2] = seed_r(2);
194  i = i - 1;
195 
196  common::AxisAlignedCubeNd<int, 3> cube;
197  GetInitialCubeUsingSeed(traj_seeds[i], traj_seeds[i + 1], &cube);
198 
199  if (!CheckIfCubeIsFree(p_grid, cube)) {
200  LOG(ERROR) << "[Ssc] SccMap - Initial cube is not free, seed id: "
201  << i;
202  common::DrivingCube driving_cube;
203  driving_cube.cube = cube;
204  driving_cube.seeds.push_back(traj_seeds[i]);
205  driving_cube.seeds.push_back(traj_seeds[i + 1]);
206  driving_corridor.cubes.push_back(driving_cube);
207 
208  driving_corridor.is_valid = false;
209  driving_corridor_vec_.push_back(driving_corridor);
210  is_valid = false;
211  break;
212  }
213 
214  std::array<bool, 6> dirs_disabled = {false, false, false,
215  false, false, false};
216  InflateCubeIn3dGrid(p_grid, dirs_disabled, config_.inflate_steps,
217  &cube);
218  common::DrivingCube driving_cube;
219  driving_cube.cube = cube;
220  driving_cube.seeds.push_back(traj_seeds[i]);
221  driving_corridor.cubes.push_back(driving_cube);
222  }
223  }
224  }
225  if (is_valid) {
226  // CorridorRelaxation(p_grid, &driving_corridor);
227  // ~ Cut cube on time axis
228  driving_corridor.cubes.back().cube.upper_bound[2] = traj_seeds.back()(2);
229  driving_corridor.is_valid = true;
230  driving_corridor_vec_.push_back(driving_corridor);
231  }
232 
233  return kSuccess;
234 }

258 ErrorType SscMap::CorridorRelaxation(GridMap3D *p_grid,
259  common::DrivingCorridor *p_corridor) {
260  std::array<int, 2> margin = {{50, 10}};
261  int t_trans = 7;
262  int num_cube = p_corridor->cubes.size();
263  for (int i = 0; i < num_cube - 1; ++i) {
264  if (1) // ~ Enable s direction
265  {
266  int cube_0_lb = p_corridor->cubes[i].cube.lower_bound[0];
267  int cube_0_ub = p_corridor->cubes[i].cube.upper_bound[0];
268 
269  int cube_1_lb = p_corridor->cubes[i + 1].cube.lower_bound[0];
270  int cube_1_ub = p_corridor->cubes[i + 1].cube.upper_bound[0];
271 
272  if (abs(cube_0_ub - cube_1_lb) < margin[0]) {
273  int room = margin[0] - abs(cube_0_ub - cube_1_lb);
274  // ~ Upward
275  std::vector<int> up_idx_list;
276  GetTimeCoveredCubeIndices(p_corridor, i + 1, 1, t_trans, &up_idx_list);
277  // ~ Downward
278  std::vector<int> down_idx_list;
279  GetTimeCoveredCubeIndices(p_corridor, i, 0, t_trans, &down_idx_list);
280  for (const auto &idx : up_idx_list) {
281  InflateCubeOnXNegAxis(p_grid, room, &(p_corridor->cubes[idx].cube));
282  }
283  for (const auto &idx : down_idx_list) {
284  InflateCubeOnXPosAxis(p_grid, room, &(p_corridor->cubes[idx].cube));
285  }
286  }
287  if (abs(cube_0_lb - cube_1_ub) < margin[0]) {
288  int room = margin[0] - abs(cube_0_lb - cube_1_ub);
289  // ~ Upward
290  std::vector<int> up_idx_list;
291  GetTimeCoveredCubeIndices(p_corridor, i + 1, 1, t_trans, &up_idx_list);
292  // ~ Downward
293  std::vector<int> down_idx_list;
294  GetTimeCoveredCubeIndices(p_corridor, i, 0, t_trans, &down_idx_list);
295  for (const auto &idx : up_idx_list) {
296  InflateCubeOnXPosAxis(p_grid, room, &(p_corridor->cubes[idx].cube));
297  }
298  for (const auto &idx : down_idx_list) {
299  InflateCubeOnXNegAxis(p_grid, room, &(p_corridor->cubes[idx].cube));
300  }
301  }
302  }
303 
304  if (1) // ~ Enable d direction
305  {
306  int cube_0_lb = p_corridor->cubes[i].cube.lower_bound[1];
307  int cube_0_ub = p_corridor->cubes[i].cube.upper_bound[1];
308 
309  int cube_1_lb = p_corridor->cubes[i + 1].cube.lower_bound[1];
310  int cube_1_ub = p_corridor->cubes[i + 1].cube.upper_bound[1];
311 
312  if (abs(cube_0_ub - cube_1_lb) < margin[1]) {
313  int room = margin[1] - abs(cube_0_ub - cube_1_lb);
314  // ~ Upward
315  std::vector<int> up_idx_list;
316  GetTimeCoveredCubeIndices(p_corridor, i + 1, 1, t_trans, &up_idx_list);
317  // ~ Downward
318  std::vector<int> down_idx_list;
319  GetTimeCoveredCubeIndices(p_corridor, i, 0, t_trans, &down_idx_list);
320  for (const auto &idx : up_idx_list) {
321  InflateCubeOnYNegAxis(p_grid, room, &(p_corridor->cubes[idx].cube));
322  }
323  for (const auto &idx : down_idx_list) {
324  InflateCubeOnYPosAxis(p_grid, room, &(p_corridor->cubes[idx].cube));
325  }
326  }
327  if (abs(cube_0_lb - cube_1_ub) < margin[1]) {
328  int room = margin[1] - abs(cube_0_lb - cube_1_ub);
329  // ~ Upward
330  std::vector<int> up_idx_list;
331  GetTimeCoveredCubeIndices(p_corridor, i + 1, 1, t_trans, &up_idx_list);
332  // ~ Downward
333  std::vector<int> down_idx_list;
334  GetTimeCoveredCubeIndices(p_corridor, i, 0, t_trans, &down_idx_list);
335  for (const auto idx : up_idx_list) {
336  InflateCubeOnYPosAxis(p_grid, room, &(p_corridor->cubes[idx].cube));
337  }
338  for (const auto idx : down_idx_list) {
339  InflateCubeOnYNegAxis(p_grid, room, &(p_corridor->cubes[idx].cube));
340  }
341  }
342  }
343  }
344  return kSuccess;

377 ErrorType SscMap::InflateCubeIn3dGrid(GridMap3D *p_grid,
378  const std::array<bool, 6> &dir_disabled,
379  const std::array<int, 6> &dir_step,
380  common::AxisAlignedCubeNd<int, 3> *cube) {
381  bool x_p_finish = dir_disabled[0];
382  bool x_n_finish = dir_disabled[1];
383  bool y_p_finish = dir_disabled[2];
384  bool y_n_finish = dir_disabled[3];
385  bool z_p_finish = dir_disabled[4];
386 
387  int x_p_step = dir_step[0];
388  int x_n_step = dir_step[1];
389  int y_p_step = dir_step[2];
390  int y_n_step = dir_step[3];
391  int z_p_step = dir_step[4];
392 
393  int t_max_grids = cube->lower_bound[2] + config_.kMaxNumOfGridAlongTime;
394 
395  decimal_t t = t_max_grids * p_grid->dims_resolution(2);
396  decimal_t a_max = config_.kMaxLongitudinalAcc;
397  decimal_t a_min = config_.kMaxLongitudinalDecel;
398  decimal_t d_comp = initial_fs_.vec_s[1] * 1;
399 
400  decimal_t s_u = initial_fs_.vec_s[0] + initial_fs_.vec_s[1] * t +
401  0.5 * a_max * t * t + d_comp;
402  decimal_t s_l = initial_fs_.vec_s[0] + initial_fs_.vec_s[1] * t +
403  0.5 * a_min * t * t - d_comp;
404 
405  int s_idx_u, s_idx_l;
406  p_grid->GetCoordUsingGlobalMetricOnSingleDim(s_u, 0, &s_idx_u);
407  p_grid->GetCoordUsingGlobalMetricOnSingleDim(s_l, 0, &s_idx_l);
408  s_idx_l = std::max(s_idx_l, static_cast<int>((config_.s_back_len / 2.0) /
409  config_.map_resolution[0]));
410 
411  while (!(x_p_finish && x_n_finish && y_p_finish && y_n_finish)) {
412  if (!x_p_finish) x_p_finish = InflateCubeOnXPosAxis(p_grid, x_p_step, cube);
413  if (!x_n_finish) x_n_finish = InflateCubeOnXNegAxis(p_grid, x_n_step, cube);
414 
415  if (!y_p_finish) y_p_finish = InflateCubeOnYPosAxis(p_grid, y_p_step, cube);
416  if (!y_n_finish) y_n_finish = InflateCubeOnYNegAxis(p_grid, y_n_step, cube);
417 
418  if (cube->upper_bound[0] >= s_idx_u) x_p_finish = true;
419  if (cube->lower_bound[0] <= s_idx_l) x_n_finish = true;
420  }
421 
422  // ~ No need to inflate along z-neg
423  while (!z_p_finish) {
424  if (!z_p_finish) z_p_finish = InflateCubeOnZPosAxis(p_grid, z_p_step, cube);
425 
426  if (cube->upper_bound[2] - cube->lower_bound[2] >=
427  config_.kMaxNumOfGridAlongTime) {
428  z_p_finish = true;
429  }
430  }
431 
432  return kSuccess;
433 }
