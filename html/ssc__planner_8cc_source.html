375 ErrorType SscPlanner::UpdateTrajectoryWithCurrentBehavior() {
376  int num_valid_behaviors = static_cast<int>(valid_behaviors_.size());
377  if (num_valid_behaviors < 1) {
378  return kWrongStatus;
379  }
380  bool find_exact_match_behavior = false;
381  int index = 0;
382  for (int i = 0; i < num_valid_behaviors; i++) {
383  if (valid_behaviors_[i] == ego_behavior_) {
384  find_exact_match_behavior = true;
385  index = i;
386  }
387  }
388  bool find_candidate_behavior = false;
389  LateralBehavior candidate_bahavior = common::LateralBehavior::kLaneKeeping;
390  if (!find_exact_match_behavior) {
391  for (int i = 0; i < num_valid_behaviors; i++) {
392  if (valid_behaviors_[i] == candidate_bahavior) {
393  find_candidate_behavior = true;
394  index = i;
395  }
396  }
397  }
398  if (!find_exact_match_behavior && !find_candidate_behavior)
399  return kWrongStatus;
400 
401  trajectory_ = FrenetBezierTrajectory(qp_trajs_[index], stf_);
402  low_spd_alternative_traj_ =
403  FrenetPrimitiveTrajectory(primitive_trajs_[index], stf_);
404  final_corridor_ = corridors_[index];
405  final_ref_states_ = ref_states_list_[index];
406  return kSuccess;
407 }

433 ErrorType SscPlanner::StateTransformForInputData() {
434  vec_E<State> global_state_vec;
435  vec_E<Vec2f> global_point_vec;
436  int num_v;
437 
438  // ~ Stage I. Package states and points
439  // * Ego vehicle state and vertices
440  {
441  global_state_vec.push_back(initial_state_);
442  vec_E<Vec2f> v_vec;
443  common::SemanticsUtils::GetVehicleVertices(ego_vehicle_.param(),
444  initial_state_, &v_vec);
445  num_v = v_vec.size();
446  global_point_vec.insert(global_point_vec.end(), v_vec.begin(), v_vec.end());
447  }
448 
449  // * Ego forward simulation trajs states and vertices
450  {
451  common::VehicleParam ego_param = ego_vehicle_.param();
452  for (int i = 0; i < (int)forward_trajs_.size(); ++i) {
453  if (forward_trajs_[i].size() < 1) continue;
454  for (int k = 0; k < (int)forward_trajs_[i].size(); ++k) {
455  // states
456  State traj_state = forward_trajs_[i][k].state();
457  global_state_vec.push_back(traj_state);
458  // vertices
459  vec_E<Vec2f> v_vec;
460  common::SemanticsUtils::GetVehicleVertices(ego_param, traj_state,
461  &v_vec);
462  global_point_vec.insert(global_point_vec.end(), v_vec.begin(),
463  v_vec.end());
464  }
465  }
466  }
467 
468  // * Surrounding vehicle trajs from MPDM
469  {
470  for (int i = 0; i < surround_forward_trajs_.size(); ++i) {
471  for (auto it = surround_forward_trajs_[i].begin();
472  it != surround_forward_trajs_[i].end(); ++it) {
473  for (int k = 0; k < it->second.size(); ++k) {
474  // states
475  State traj_state = it->second[k].state();
476  global_state_vec.push_back(traj_state);
477  // vertices
478  vec_E<Vec2f> v_vec;
479  common::SemanticsUtils::GetVehicleVertices(it->second[k].param(),
480  traj_state, &v_vec);
481  global_point_vec.insert(global_point_vec.end(), v_vec.begin(),
482  v_vec.end());
483  }
484  }
485  }
486  }
487 
488  // * Obstacle grids
489  {
490  for (auto it = obstacle_grids_.begin(); it != obstacle_grids_.end(); ++it) {
491  Vec2f pt((*it)[0], (*it)[1]);
492  global_point_vec.push_back(pt);
493  }
494  }
495 
496  vec_E<FrenetState> frenet_state_vec(global_state_vec.size());
497  vec_E<Vec2f> fs_point_vec(global_point_vec.size());
498 
499  // ~ Stage II. Do transformation in multi-thread flavor
500 #if USE_OPENMP
501  TicToc timer_stf;
502  StateTransformUsingOpenMp(global_state_vec, global_point_vec,
503  &frenet_state_vec, &fs_point_vec);
504  LOG(WARNING) << "[Ssc]OpenMp transform time cost: " << timer_stf.toc()
505  << " ms.";
506 #else
507  TicToc timer_stf;
508  StateTransformSingleThread(global_state_vec, global_point_vec,
509  &frenet_state_vec, &fs_point_vec);
510  LOG(WARNING) << "[Ssc]Single thread transform time cost: " << timer_stf.toc()
511  << " ms.";
512 #endif
513 
514  // ~ Stage III. Retrieve states and points
515  int offset = 0;
516  // * Ego vehicle state and vertices
517  {
518  fs_ego_vehicle_.frenet_state = frenet_state_vec[offset];
519  fs_ego_vehicle_.vertices.clear();
520  for (int i = 0; i < num_v; ++i) {
521  fs_ego_vehicle_.vertices.push_back(fs_point_vec[offset * num_v + i]);

524  }

526  // * Ego forward simulation trajs states and vertices
527  {
528  forward_trajs_fs_.clear();

531  if (forward_trajs_[j].size() < 1) assert(false);
532  vec_E<common::FsVehicle> traj_fs;

535  fs_v.frenet_state = frenet_state_vec[offset];
536  for (int i = 0; i < num_v; ++i) {
537  fs_v.vertices.push_back(fs_point_vec[offset * num_v + i]);

543  }
544  }

547  {
548  surround_forward_trajs_fs_.clear();
549  for (int j = 0; j < surround_forward_trajs_.size(); ++j) {
550  std::unordered_map<int, vec_E<common::FsVehicle>> sur_trajs;
551  for (auto it = surround_forward_trajs_[j].begin();
552  it != surround_forward_trajs_[j].end(); ++it) {
553  int v_id = it->first;
554  vec_E<common::FsVehicle> traj_fs;
555  for (int k = 0; k < it->second.size(); ++k) {
556  common::FsVehicle fs_v;
557  fs_v.frenet_state = frenet_state_vec[offset];
558  for (int i = 0; i < num_v; ++i) {
559  fs_v.vertices.push_back(fs_point_vec[offset * num_v + i]);
560  }
561  traj_fs.emplace_back(fs_v);
562  offset++;
563  }
564  sur_trajs.insert(
565  std::pair<int, vec_E<common::FsVehicle>>(v_id, traj_fs));
566  }
567  surround_forward_trajs_fs_.emplace_back(sur_trajs);
568  }
569  }
570 
571  // * Obstacle grids
572  {
573  obstacle_grids_fs_.clear();
574  for (int i = 0; i < static_cast<int>(obstacle_grids_.size()); ++i) {
575  obstacle_grids_fs_.push_back(fs_point_vec[offset * num_v + i]);
576  }
577  }
578 
579  ego_frenet_state_ = fs_ego_vehicle_.frenet_state;
580  return kSuccess;
581 }

648  std::vector<decimal_t> t_vec_xy;

651  common::State state;
652  // * check init state
653  if (traj.GetState(traj.begin(), &state) != kSuccess) {
654  LOG(ERROR) << "[Ssc][Validate]State evaluation error";
655  return kWrongStatus;
656  }
657 
658  if ((state.vec_position - initial_state_.vec_position).norm() > 0.1) {
659  LOG(ERROR) << "[Ssc][Validate]Init position miss match";
660  return kWrongStatus;
661  }
662 
663  if (fabs(state.velocity - initial_state_.velocity) > 0.1) {
664  LOG(ERROR) << "[Ssc][Validate]Init vel miss match";
665  return kWrongStatus;
666  }
667  // * check end state
668  if (traj.GetState(traj.end(), &state) != kSuccess) {
669  LOG(ERROR) << "[Ssc][Validate]End state eval error";
670  return kWrongStatus;
671  }
672 
673  for (const auto t : t_vec_xy) {
674  if (traj.GetState(t, &state) != kSuccess) {
675  LOG(ERROR) << "[Ssc][Validate]State eval error";
676  return kWrongStatus;
677  }
678  if (fabs(state.curvature) > 0.33) {
679  LOG(ERROR) << "[Ssc][Validate]initial_state velocity "
680  << initial_state_.velocity << " Curvature " << state.curvature
681  << " invalid.";
682  return kWrongStatus;
683  }
684  }
685  return kSuccess;
