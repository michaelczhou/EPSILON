5 ErrorType StateTransformer::GetStateFromFrenetState(const FrenetState& fs,
6  State* s) const {
7  if (!lane_.IsValid()) {
8  printf("[StateFromFrenetState]Err: lane not valid.\n");
9  return kIllegalInput;
10  }
11  if (!fs.is_ds_usable) {
12  // ~ You may come from high speed traj but vs = 0.
13  return kIllegalInput;
14  }
15 
16  if (LaneDim != 2) {
17  printf("[StateFromFrenetState]Err: cannot support non-plane now.\n");
18  return kIllegalInput;
19  }
20 
21  decimal_t curvature, curvature_derivative;
22  if (lane_.GetCurvatureByArcLength(fs.vec_s[0], &curvature,
23  &curvature_derivative) != kSuccess) {
24  return kWrongStatus;
25  }
26 
27  decimal_t one_minus_curd = 1 - curvature * fs.vec_ds[0];
28  if (one_minus_curd < kEPS) {
29  // ~ the violation is typically caused by lateral dependent trajectories
30  // ~ with very small s (overshotting)
31  return kWrongStatus;
32  }
33 
34  Vecf<LaneDim> lane_pos;
35  if (lane_.GetPositionByArcLength(fs.vec_s[0], &lane_pos) != kSuccess) {
36  return kWrongStatus;
37  }
38 
39  Vecf<LaneDim> vec_tangent;
40  if (lane_.GetTangentVectorByArcLength(fs.vec_s[0], &vec_tangent) !=
41  kSuccess) {
42  return kWrongStatus;
43  }
44 
45  decimal_t lane_orientation = vec2d_to_angle(vec_tangent);
46 
47  Vecf<LaneDim> vec_normal(-vec_tangent[1], vec_tangent[0]);
48  decimal_t tan_delta_theta = fs.vec_ds[1] / one_minus_curd;
49  decimal_t delta_theta = atan2(fs.vec_ds[1], one_minus_curd);
50  decimal_t cn_delta_theta = cos(delta_theta);
51 
52  s->vec_position = vec_normal * fs.vec_ds[0] + lane_pos;
53  s->velocity = fs.vec_s[1] * one_minus_curd / cn_delta_theta;
54  s->angle = normalize_angle(delta_theta + lane_orientation);
55 
56  decimal_t lhs = (fs.vec_ds[2] + (curvature_derivative * fs.vec_ds[0] +
57  curvature * fs.vec_ds[1]) *
58  tan_delta_theta) *
59  cn_delta_theta * cn_delta_theta / one_minus_curd;
60  s->curvature = (lhs + curvature) * cn_delta_theta / one_minus_curd;
61  decimal_t delta_theta_derivative = 1.0 /
62  (1 + tan_delta_theta * tan_delta_theta) *
63  (fs.vec_ds[2] * one_minus_curd +
64  curvature * fs.vec_ds[1] * fs.vec_ds[1]) /
65  pow(one_minus_curd, 2);
66  s->acceleration =
67  fs.vec_s[2] * one_minus_curd / cn_delta_theta +
68  fs.vec_s[1] * fs.vec_s[1] / cn_delta_theta *
69  (one_minus_curd * tan_delta_theta * delta_theta_derivative -
70  (curvature_derivative * fs.vec_ds[0] + curvature * fs.vec_ds[1]));
71  s->time_stamp = fs.time_stamp; // pass the time stamp
72  return kSuccess;
73 }

106 ErrorType StateTransformer::GetFrenetStateFromState(const State& s,
107  FrenetState* fs) const {
108  if (!lane_.IsValid()) {
109  return kIllegalInput;
110  }
111 
112  decimal_t arc_length;
113  if (lane_.GetArcLengthByVecPosition(s.vec_position, &arc_length) !=
114  kSuccess) {
115  return kWrongStatus;
116  }
117 
118  decimal_t curvature, curvature_derivative;
119  if (lane_.GetCurvatureByArcLength(arc_length, &curvature,
120  &curvature_derivative) != kSuccess) {
121  return kWrongStatus;
122  }
123 
124  Vecf<2> lane_position;
125  if (lane_.GetPositionByArcLength(arc_length, &lane_position) != kSuccess) {
126  return kWrongStatus;
127  }
128 
129  Vecf<2> lane_tangent_vec;
130  if (lane_.GetTangentVectorByArcLength(arc_length, &lane_tangent_vec) !=
131  kSuccess) {
132  return kWrongStatus;
133  }
134  decimal_t lane_orientation = vec2d_to_angle(lane_tangent_vec);
135  Vecf<2> lane_normal_vec = Vecf<2>(-lane_tangent_vec[1], lane_tangent_vec[0]);
136 
137  const decimal_t step_tolerance = 0.5;
138  if (fabs((s.vec_position - lane_position).dot(lane_tangent_vec)) >
139  step_tolerance) {
140  // ~ projection deviates
141  // ~ @(denny.ding) check whether this condition is good?
142  return kWrongStatus;
143  }
144 
145  // if (fabs(normalize_angle(s.angle - lane_orientation)) > M_PI / 2.0) {
146  // // printf("[FrenetStateFromState]State angle %lf not on lane (%lf).\n",
147  // // s.angle, lane_orientation);
148  // return kWrongStatus;
149  // }
150 
151  decimal_t d = (s.vec_position - lane_position).dot(lane_normal_vec);
152  decimal_t one_minus_curd = 1 - curvature * d;
153  if (one_minus_curd < kEPS) {
154  // printf("[StateFromFrenetState]d not valid for transform.\n");
155  return kWrongStatus;
156  }
157  decimal_t delta_theta = normalize_angle(s.angle - lane_orientation);
158 
159  decimal_t cn_delta_theta = cos(delta_theta);
160  decimal_t tan_delta_theta = tan(delta_theta);
161  decimal_t ds = one_minus_curd * tan_delta_theta;
162  decimal_t dss =
163  -(curvature_derivative * d + curvature * ds) * tan_delta_theta +
164  one_minus_curd / pow(cn_delta_theta, 2) *
165  (s.curvature * one_minus_curd / cn_delta_theta - curvature);
166  decimal_t sp = s.velocity * cn_delta_theta / one_minus_curd;
167 
168  decimal_t delta_theta_derivative =
169  1 / (1 + pow(tan_delta_theta, 2)) *
170  (dss * one_minus_curd + curvature * pow(ds, 2)) / pow(one_minus_curd, 2);
171 
172  decimal_t spp =
173  (s.acceleration -
174  pow(sp, 2) / cn_delta_theta *
175  (one_minus_curd * tan_delta_theta * delta_theta_derivative -
176  (curvature_derivative * d + curvature * ds))) *
177  cn_delta_theta / one_minus_curd;
178 
179  fs->Load(Vecf<3>(arc_length, sp, spp), Vecf<3>(d, ds, dss),
180  FrenetState::kInitWithDs);
181  fs->time_stamp = s.time_stamp;
182  return kSuccess;
183 }
