27  para.push_back(d);
28  for (int i = 1; i < (int)samples.size(); ++i) {
29  double dx = samples[i](0) - samples[i - 1](0);
30  double dy = samples[i](1) - samples[i - 1](1);
31  d += std::hypot(dx, dy);

33  }
