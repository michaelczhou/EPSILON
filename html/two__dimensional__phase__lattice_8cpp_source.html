29 struct two_dimensional_phase_lattice
30 {
31  two_dimensional_phase_lattice( double gamma = 0.5 )
32  : m_gamma( gamma ) { }
33 
34  void operator()( const state_type &x , state_type &dxdt , double /* t */ ) const
35  {
36  size_t size1 = x.size1() , size2 = x.size2();
37 
38  for( size_t i=1 ; i<size1-1 ; ++i )
39  {
40  for( size_t j=1 ; j<size2-1 ; ++j )
41  {
42  dxdt( i , j ) =
43  coupling_func( x( i + 1 , j ) - x( i , j ) ) +
44  coupling_func( x( i - 1 , j ) - x( i , j ) ) +
45  coupling_func( x( i , j + 1 ) - x( i , j ) ) +
46  coupling_func( x( i , j - 1 ) - x( i , j ) );
47  }
48  }
49 
50  for( size_t i=0 ; i<x.size1() ; ++i ) dxdt( i , 0 ) = dxdt( i , x.size2() -1 ) = 0.0;
51  for( size_t j=0 ; j<x.size2() ; ++j ) dxdt( 0 , j ) = dxdt( x.size1() -1 , j ) = 0.0;
52  }
53 
54  double coupling_func( double x ) const
55  {
56  return sin( x ) - m_gamma * ( 1.0 - cos( x ) );
57  }
58 
59  double m_gamma;
60 };
