116 ErrorType SscPlannerServer::FilterSingularityState(
117  const vec_E<common::State>& hist, common::State* filter_state) {
118  if (hist.empty()) {
119  return kWrongStatus;
120  }
121  decimal_t duration = filter_state->time_stamp - hist.back().time_stamp;
122  decimal_t wheel_base = 2.85;
123  decimal_t max_steer = M_PI / 4.0;
124  decimal_t singular_velocity = kBigEPS;
125  decimal_t max_orientation_rate = tan(max_steer) / 2.85 * singular_velocity;
126  decimal_t max_orientation_change = max_orientation_rate * duration;
127 
128  if (fabs(filter_state->velocity) < singular_velocity &&
129  fabs(normalize_angle(filter_state->angle - hist.back().angle)) >
130  max_orientation_change) {
131  printf(
132  "[SscPlannerServer]Detect singularity velocity %lf angle (%lf, %lf).\n",
133  filter_state->velocity, hist.back().angle, filter_state->angle);
134  filter_state->angle = hist.back().angle;
135  printf("[SscPlannerServer]Filter angle to %lf.\n", hist.back().angle);
136  }
137  return kSuccess;
138 }

166 void SscPlannerServer::MainThread() {
167  using namespace std::chrono;
168  system_clock::time_point current_start_time{system_clock::now()};
169  system_clock::time_point next_start_time{current_start_time};
170  const milliseconds interval{static_cast<int>(1000.0 / work_rate_)};
171  while (true) {
172  current_start_time = system_clock::now();
173  next_start_time = current_start_time + interval;
174  PlanCycleCallback();
175  std::this_thread::sleep_until(next_start_time);
176  }

179 void SscPlannerServer::PlanCycleCallback() {
180  if (!is_replan_on_) {
181  return;
182  }
183  // printf("input buffer size: %d.\n", p_input_smm_buff_->size_approx());
184  // is_map_updated_ = false; // return when no new map
185  while (p_input_smm_buff_->try_dequeue(last_smm_)) {
186  is_map_updated_ = true;
187  }
188 
189  if (!is_map_updated_) return;
190 
191  // Update map
192  auto map_ptr =
193  std::make_shared<semantic_map_manager::SemanticMapManager>(last_smm_);
194  map_adapter_.set_map(map_ptr);
195 
196  PublishData();
197 
198  auto current_time = ros::Time::now().toSec();
199  printf("[SscPlannerServer]>>>>>>>current time %lf.\n", current_time);
200  if (executing_traj_ == nullptr || !executing_traj_->IsValid() ||
201  !use_sim_state_) {
202  // Init planning
203  if (planner_.RunOnce() != kSuccess) {
204  // printf("[SscPlannerServer]Initial planning failed.\n");
205  return;
206  }
207 
208  desired_state_hist_.clear();
209  desired_state_hist_.push_back(last_smm_.ego_vehicle().state());
210  ctrl_state_hist_.clear();
211  ctrl_state_hist_.push_back(last_smm_.ego_vehicle().state());
212  executing_traj_ = std::move(planner_.trajectory());
213  global_init_stamp_ = executing_traj_->begin();
214  printf(
215  "[SscPlannerServer]init plan success with stamp: %lf and angle %lf.\n",
216  global_init_stamp_, last_smm_.ego_vehicle().state().angle);
217  return;
218  }
219 
220  if (current_time > executing_traj_->end()) {
221  printf("[SscPlannerServer]Current time %lf out of [%lf, %lf].\n",
222  current_time, executing_traj_->begin(), executing_traj_->end());
223  printf("[SscPlannerServer]Mission complete.\n");
224  // is_replan_on_ = false;
225  executing_traj_.release();
226  next_traj_.release();
227  return;
228  }
229 
230  // NOTE: comment this block if you want to disable replan
231  {
232  if (next_traj_ == nullptr || !next_traj_->IsValid()) {
233  Replan();
234  return;
235  }
236 
237  if (next_traj_->IsValid()) {
238  executing_traj_ = std::move(next_traj_);
239  // next_traj_ = common::Trajectory();
240  Replan();
241  return;
242  }
243  }

246 void SscPlannerServer::Replan() {
247  if (!is_replan_on_) return;
248  if (executing_traj_ == nullptr || !executing_traj_->IsValid()) return;
249 
250  decimal_t plan_horizon = 1.0 / work_rate_;
251  common::State desired_state;
252  decimal_t cur_time = ros::Time::now().toSec();
253 
254  int num_cycles_exec = std::floor(
255  (executing_traj_->begin() - global_init_stamp_) / plan_horizon);
256  int num_cycles_ahead =
257  cur_time > executing_traj_->begin()
258  ? std::floor((cur_time - global_init_stamp_) / plan_horizon) + 1
259  : num_cycles_exec + 1;
260  decimal_t t = global_init_stamp_ + plan_horizon * num_cycles_ahead;
261 
262  printf(
263  "[SscPlannerServer]init stamp: %lf, plan horizon: %lf, num cycles %d.\n",
264  global_init_stamp_, plan_horizon, num_cycles_ahead);
265  printf(
266  "[SscPlannerServer]Replan at cur time %lf with executing traj begin "
267  "time: %lf to rounded t: %lf.\n",
268  cur_time, executing_traj_->begin(), t);
269 
270  if (executing_traj_->GetState(t, &desired_state) != kSuccess) {
271  printf("[SscPlannerServer]Cannot get desired state at %lf.\n", t);
272  return;
273  }
274  printf(
275  "[SscPlannerServer]t %lf, desired state (x,y,v,a,theta):(%lf, %lf, %lf, "
276  "%lf, %lf).\n",
277  t, desired_state.vec_position[0], desired_state.vec_position[1],
278  desired_state.velocity, desired_state.acceleration, desired_state.angle);
279 
280  FilterSingularityState(desired_state_hist_, &desired_state);
281  desired_state_hist_.push_back(desired_state);
282  if (desired_state_hist_.size() > 100)
283  desired_state_hist_.erase(desired_state_hist_.begin());
284 
285  planner_.set_initial_state(desired_state);
286 
287  time_profile_tool_.tic();
288  if (planner_.RunOnce() != kSuccess) {
289  printf("[SscPlannerServer]Ssc planner core failed in %lf ms.\n",
290  time_profile_tool_.toc());
291  require_intervention_signal_ = true;
292  return;
293  }
294 
295  require_intervention_signal_ = false;
296  printf("[SscPlannerServer]Ssc planner succeed in %lf ms.\n",
297  time_profile_tool_.toc());
298 
299  next_traj_ = std::move(planner_.trajectory());
300 }
