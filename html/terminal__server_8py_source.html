44 def project_world_to_image(point_3dof):
45  x = point_3dof[0] - center_3dof[0]
46  y = point_3dof[1] - center_3dof[1]
47  angle = center_3dof[2]
48  u = width/2 + scale * (x*sin(angle) - y * cos(angle))
49  v = height/2 - scale * (x*cos(angle) + y * sin(angle))
50  return (u, v)

99 def calc_current_steer_acc():
100  if len(state_seq) < 2:
101  return 0.0, 0.0
102  steer_list = []
103  for i in range(len(state_seq)-1):
104  state1 = state_seq[i]
105  steer = atan(state1.curvature * 2.85)
106  steer_list.append(steer)
107  ave_steer = reduce(lambda x, y: x + y, steer_list) / len(steer_list)
108  rotated_angle = ave_steer * 1.25 * (360.0/45.0)
109  return rotated_angle, state_seq[-1].acceleration
