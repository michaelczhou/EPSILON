65 void SscVisualizer::VisualizeQpTrajs(
66  const ros::Time &stamp, const vec_E<common::BezierSpline<5, 2>> &trajs) {
67  if (trajs.empty()) {
68  printf("[SscQP]No valid qp trajs.\n");
69  return;
70  }
71  int id = 0;
72  visualization_msgs::MarkerArray traj_mk_arr;
73  for (int i = 0; i < static_cast<int>(trajs.size()); i++) {
74  visualization_msgs::Marker traj_mk;
75  traj_mk.type = visualization_msgs::Marker::LINE_STRIP;
76  traj_mk.action = visualization_msgs::Marker::MODIFY;
77  traj_mk.id = id++;
78  Vecf<2> pos;
79  for (decimal_t t = trajs[i].begin(); t < trajs[i].end() + kEPS; t += 0.02) {
80  if (trajs[i].evaluate(t, 0, &pos) == kSuccess) {
81  geometry_msgs::Point pt;
82  pt.x = pos[0];
83  pt.y = pos[1];
84  pt.z = t - start_time_;
85  traj_mk.points.push_back(pt);
86  }
87  }
88  common::VisualizationUtil::FillScaleColorInMarker(
89  Vec3f(0.2, 0.2, 0.2), common::cmap.at("magenta"), &traj_mk);
90  traj_mk_arr.markers.push_back(traj_mk);
91  }
92 
93  int num_markers = static_cast<int>(traj_mk_arr.markers.size());
94  common::VisualizationUtil::FillHeaderIdInMarkerArray(
95  stamp, std::string("ssc_map"), last_qp_traj_mk_cnt, &traj_mk_arr);
96  qp_pub_.publish(traj_mk_arr);
97  last_qp_traj_mk_cnt = num_markers;
98 }

256 void SscVisualizer::VisualizeCorridorsInSscSpace(
257  const ros::Time &stamp, const vec_E<common::DrivingCorridor> corridor_vec,
258  const SscMap *p_ssc_map) {
259  if (corridor_vec.empty()) return;
260  visualization_msgs::MarkerArray corridor_vec_marker;
261  int id_cnt = 0;
262  for (const auto &corridor : corridor_vec) {
263  // if (corridor.is_valid) continue;
264  int cube_cnt = 0;
265  for (const auto &driving_cube : corridor.cubes) {
266  // * Show seeds
267  common::ColorARGB color = common::GetJetColorByValue(
268  cube_cnt, corridor.cubes.size() - 1, -kEPS);
269  for (const auto &seed : driving_cube.seeds) {
270  visualization_msgs::Marker seed_marker;
271  decimal_t s_x, s_y, s_z;
272  p_ssc_map->p_3d_grid()->GetGlobalMetricUsingCoordOnSingleDim(seed(0), 0,
273  &s_x);
274  p_ssc_map->p_3d_grid()->GetGlobalMetricUsingCoordOnSingleDim(seed(1), 1,
275  &s_y);
276  p_ssc_map->p_3d_grid()->GetGlobalMetricUsingCoordOnSingleDim(seed(2), 2,
277  &s_z);
278  common::VisualizationUtil::GetRosMarkerSphereUsingPoint(
279  Vec3f(s_x, s_y, s_z - start_time_), color, Vec3f(0.3, 0.3, 0.3),
280  id_cnt++, &seed_marker);
281  corridor_vec_marker.markers.push_back(seed_marker);
282  }
283 
284  // * Show driving cube
285  decimal_t x_max, x_min;
286  decimal_t y_max, y_min;
287  decimal_t z_max, z_min;
288  p_ssc_map->p_3d_grid()->GetGlobalMetricUsingCoordOnSingleDim(
289  driving_cube.cube.upper_bound[0], 0, &x_max);
290  p_ssc_map->p_3d_grid()->GetGlobalMetricUsingCoordOnSingleDim(
291  driving_cube.cube.lower_bound[0], 0, &x_min);
292  p_ssc_map->p_3d_grid()->GetGlobalMetricUsingCoordOnSingleDim(
293  driving_cube.cube.upper_bound[1], 1, &y_max);
294  p_ssc_map->p_3d_grid()->GetGlobalMetricUsingCoordOnSingleDim(
295  driving_cube.cube.lower_bound[1], 1, &y_min);
296  p_ssc_map->p_3d_grid()->GetGlobalMetricUsingCoordOnSingleDim(
297  driving_cube.cube.upper_bound[2], 2, &z_max);
298  p_ssc_map->p_3d_grid()->GetGlobalMetricUsingCoordOnSingleDim(
299  driving_cube.cube.lower_bound[2], 2, &z_min);
300  decimal_t dx = x_max - x_min;
301  decimal_t dy = y_max - y_min;
302  decimal_t dz = z_max - z_min;
303  decimal_t x = x_min + dx / 2.0;
304  decimal_t y = y_min + dy / 2.0;
305  decimal_t z = z_min - start_time_ + dz / 2.0;
306 
307  std::array<decimal_t, 3> aabb_coord = {x, y, z};
308  std::array<decimal_t, 3> aabb_len = {dx, dy, dz};
309  common::AxisAlignedBoundingBoxND<3> map_aabb(aabb_coord, aabb_len);
310  visualization_msgs::Marker map_aabb_marker;
311  map_aabb_marker.id = id_cnt++;
312  common::VisualizationUtil::GetRosMarkerCubeUsingAxisAlignedBoundingBox3D(
313  map_aabb, common::ColorARGB(0.15, 0.3, 1.0, 0.3), &map_aabb_marker);
314  corridor_vec_marker.markers.push_back(map_aabb_marker);
315  ++cube_cnt;
316  }
317  }
318 
319  int num_markers = static_cast<int>(corridor_vec_marker.markers.size());
320  common::VisualizationUtil::FillHeaderIdInMarkerArray(
321  ros::Time::now(), std::string("ssc_map"), last_corridor_mk_cnt,
322  &corridor_vec_marker);
323  corridor_pub_.publish(corridor_vec_marker);
324  last_corridor_mk_cnt = num_markers;
325 }
