144  common::LateralBehavior* lat_behavior, bool* is_cancel_operation) const {

160  }
161  } else {
162  if (action.lat == DcpLatAction::kLaneKeeping) {
163  *is_cancel_operation = true;
164  }
165  }
166  duration += action.t;
167  }
168  if (!find_lat_active_behavior) {
169  *operation_at_seconds = duration + cfg_.sim().duration().layer();
170  *lat_behavior = common::LateralBehavior::kLaneKeeping;
171  *is_cancel_operation = false;
172  }

354 ErrorType EudmPlanner::UpdateSimSetupForScenario(
355  const std::vector<DcpAction>& action_seq,
356  ForwardSimEgoAgent* ego_fsagent) const {
357  // * Get the type of lateral action sequence
358  common::LateralBehavior seq_lat_behavior;
359  decimal_t operation_at_seconds;
360  bool is_cancel_behavior;
361  ClassifyActionSeq(action_seq, &operation_at_seconds, &seq_lat_behavior,
362  &is_cancel_behavior);
363  ego_fsagent->operation_at_seconds = operation_at_seconds;
364  ego_fsagent->is_cancel_behavior = is_cancel_behavior;
365  ego_fsagent->seq_lat_behavior = seq_lat_behavior;
366 
367  // * get action sequence type
368  if (is_cancel_behavior) {
369  ego_fsagent->lat_behavior_longterm = LateralBehavior::kLaneKeeping;
370  ego_fsagent->seq_lat_mode = LatSimMode::kChangeThenCancel;
371  } else {
372  if (seq_lat_behavior == LateralBehavior::kLaneKeeping) {
373  ego_fsagent->seq_lat_mode = LatSimMode::kAlwaysLaneKeep;
374  } else if (action_seq.front().lat == DcpLatAction::kLaneKeeping) {
375  ego_fsagent->seq_lat_mode = LatSimMode::kKeepThenChange;
376  } else {
377  ego_fsagent->seq_lat_mode = LatSimMode::kAlwaysLaneChange;
378  }
379  ego_fsagent->lat_behavior_longterm = seq_lat_behavior;
380  }
381 
382  // * lon
383  decimal_t desired_vel = std::floor(ego_fsagent->vehicle.state().velocity);
384  simulator::IntelligentDriverModel::Param idm_param_tmp;
385  idm_param_tmp = ego_sim_param_.idm_param;
386 
387  switch (action_seq[1].lon) {
388  case DcpLonAction::kAccelerate: {
389  idm_param_tmp.kDesiredVelocity = std::min(
390  desired_vel + cfg_.sim().acc_cmd_vel_gap(), desired_velocity_);
391  idm_param_tmp.kMinimumSpacing *=
392  (1.0 - cfg_.sim().ego().lon_aggressive_ratio());
393  idm_param_tmp.kDesiredHeadwayTime *=
394  (1.0 - cfg_.sim().ego().lon_aggressive_ratio());
395  break;
396  }
397  case DcpLonAction::kDecelerate: {
398  idm_param_tmp.kDesiredVelocity =
399  std::min(std::max(desired_vel - cfg_.sim().dec_cmd_vel_gap(), 0.0),
400  desired_velocity_);
401  break;
402  }
403  case DcpLonAction::kMaintain: {
404  idm_param_tmp.kDesiredVelocity = std::min(desired_vel, desired_velocity_);
405  break;
406  }
407  default: {
408  printf("[Eudm]Error - Lon action not valid\n");
409  assert(false);
410  }
411  }
412  ego_fsagent->sim_param = ego_sim_param_;
413  ego_fsagent->sim_param.idm_param = idm_param_tmp;
414  ego_fsagent->lat_range = cfg_.sim().ego().cooperative_lat_range();
415 
416  return kSuccess;
417 }

436  cfg_.sim().ref_line().forward_len_min()),
437  cfg_.sim().ref_line().forward_len_max());

440  if (map_itf_->GetRefLaneForStateByBehavior(
441  state, std::vector<int>(), LateralBehavior::kLaneKeeping,

445  }

451  state, std::vector<int>(), ego_fsagent->lat_behavior,

457  ego_fsagent->target_stf = common::StateTransformer(lane_target);

461  state, std::vector<int>(), ego_fsagent->lat_behavior_longterm,
462  forward_lane_len, cfg_.sim().ref_line().backward_len_max(), false,
463  &lane_longterm) != kSuccess) {
464  return kWrongStatus;

467  ego_fsagent->longterm_stf = common::StateTransformer(lane_longterm);

471  common::VehicleSet other_vehicles;
472  for (const auto& pv : other_fsagent.forward_sim_agents) {

502

506  front_vehicle.param().d_cr();

509  ego_fsagent->vehicle.state().velocity,

772 ErrorType EudmPlanner::UpdateLateralActionSequence(
773  const int cur_idx, std::vector<DcpAction>* action_seq) const {
774  if (cur_idx == static_cast<int>(action_seq->size()) - 1) {
775  return kSuccess;
776  }
777 
778  switch ((*action_seq)[cur_idx].lat) {
779  case DcpLatAction::kLaneKeeping: {
780  // * no need to update
781  break;
782  }
783  case DcpLatAction::kLaneChangeLeft: {
784  for (int i = cur_idx + 1; i < static_cast<int>(action_seq->size()); ++i) {
785  if ((*action_seq)[i].lat == DcpLatAction::kLaneChangeLeft) {
786  // * LLLLL -> LLKKK
787  (*action_seq)[i].lat = DcpLatAction::kLaneKeeping;
788  } else if ((*action_seq)[i].lat == DcpLatAction::kLaneKeeping) {
789  // * LLKKK -> LLRRR
790  (*action_seq)[i].lat = DcpLatAction::kLaneChangeRight;
791  } else if ((*action_seq)[i].lat == DcpLatAction::kLaneChangeRight) {
792  // * LLRRR -> x
793  return kWrongStatus;
794  }
795  }
796  break;
797  }
798  case DcpLatAction::kLaneChangeRight: {
799  for (int i = cur_idx + 1; i < static_cast<int>(action_seq->size()); ++i) {
800  if ((*action_seq)[i].lat == DcpLatAction::kLaneChangeRight) {
801  // * RRRRR -> RRKKK
802  (*action_seq)[i].lat = DcpLatAction::kLaneKeeping;
803  } else if ((*action_seq)[i].lat == DcpLatAction::kLaneKeeping) {
804  // * RRKKK -> RRLLL
805  (*action_seq)[i].lat = DcpLatAction::kLaneChangeLeft;
806  } else if ((*action_seq)[i].lat == DcpLatAction::kLaneChangeLeft) {
807  // * RRLLL -> x
808  return kWrongStatus;
809  }
810  }
811  break;
812  }
813 
814  default: {
815  std::cout << "[Eudm]Error - Invalid lateral behavior" << std::endl;
816  assert(false);
817  }
818  }
819 
820  return kSuccess;
821 }

1021  traj_a[i].state().velocity *
1022  pow(10, cfg_.cost().safety().rss_lack_speed_power_coeff() *
1023  fabs(traj_a[i].state().velocity - rss_vel_low));
1024  }
1025  }
1026  }
1027  }

1029  *cost = cost_tmp;
1030  return kSuccess;

1033 ErrorType EudmPlanner::StrictSafetyCheck(
1034  const vec_E<common::Vehicle>& ego_traj,
1035  const std::unordered_map<int, vec_E<common::Vehicle>>& surround_trajs,
1036  bool* is_safe, int* collided_id) {
1037  if (!cfg_.safety().strict_check_enable()) {
1038  *is_safe = true;
1039  return kSuccess;
1040  }
1041 
1042  int num_points_ego = ego_traj.size();
1043  if (num_points_ego == 0) {
1044  *is_safe = true;
1045  return kSuccess;
1046  }
1047  // strict collision check
1048  for (auto it = surround_trajs.begin(); it != surround_trajs.end(); it++) {
1049  int num_points_other = it->second.size();
1050  if (num_points_other != num_points_ego) {
1051  *is_safe = false;
1052  LOG(ERROR) << "[Eudm]unsafe due to incomplete sim record for vehicle";
1053  return kSuccess;
1054  }
1055  for (int i = 0; i < num_points_ego; i++) {
1056  common::Vehicle inflated_a, inflated_b;
1057  common::SemanticsUtils::InflateVehicleBySize(
1058  ego_traj[i], cfg_.safety().strict().inflation_w(),
1059  cfg_.safety().strict().inflation_h(), &inflated_a);
1060  common::SemanticsUtils::InflateVehicleBySize(
1061  it->second[i], cfg_.safety().strict().inflation_w(),
1062  cfg_.safety().strict().inflation_h(), &inflated_b);
1063  bool is_collision = false;
1064  map_itf_->CheckCollisionUsingState(inflated_a.param(), inflated_a.state(),
1065  inflated_b.param(), inflated_b.state(),
1066  &is_collision);

1088  auto ego_lat_behavior_this_layer = ego_fsagent.lat_behavior;
1089

1091  auto is_cancel_behavior = ego_fsagent.is_cancel_behavior;

1094  for (const auto& v : other_fsagent.forward_sim_agents) {
1095  vehicle_set.vehicles.insert(std::make_pair(v.first, v.second.vehicle));
1096  }
1097 
1098  decimal_t ego_velocity = ego_fsagent.vehicle.state().velocity;
1099  // f = c1 * fabs(v_ego - v_user), if v_ego < v_user
1100  // f = c2 * fabs(v_ego - v_user - vth), if v_ego > v_user + vth
1101  // unit of this cost is velocity (finally multiplied by duration)
1102  CostStructure cost_tmp;
1103  if (ego_fsagent.vehicle.state().velocity < desired_velocity_) {
1104  cost_tmp.efficiency.ego_to_desired_vel =
1105  cfg_.cost().effciency().ego_lack_speed_to_desired_unit_cost() *
1106  fabs(ego_fsagent.vehicle.state().velocity - desired_velocity_);
1107  } else {
1108  if (ego_fsagent.vehicle.state().velocity >
1109  desired_velocity_ +
1110  cfg_.cost().effciency().ego_desired_speed_tolerate_gap()) {
1111  cost_tmp.efficiency.ego_to_desired_vel =
1112  cfg_.cost().effciency().ego_over_speed_to_desired_unit_cost() *
1113  fabs(ego_fsagent.vehicle.state().velocity - desired_velocity_ -
1114  cfg_.cost().effciency().ego_desired_speed_tolerate_gap());
1115  }
1116  }
1117 
1118  // f = ratio * c1 * fabs(v_ego - v_user) , if v_ego < v_user && v_ego
1119  // > v_leading
1120  // unit of this cost is velocity (finally multiplied by duration)
1121  common::Vehicle leading_vehicle;
1122  decimal_t distance_residual_ratio = 0.0;
1123  if (map_itf_->GetLeadingVehicleOnLane(
1124  ego_fsagent.target_lane, ego_fsagent.vehicle.state(), vehicle_set,
1125  ego_fsagent.lat_range, &leading_vehicle,
1126  &distance_residual_ratio) == kSuccess) {

1128  (leading_vehicle.state().vec_position -
1129  ego_fsagent.vehicle.state().vec_position)
1130  .norm();
1131  if (ego_fsagent.vehicle.state().velocity < desired_velocity_ &&
1132  leading_vehicle.state().velocity < desired_velocity_ &&
1133  distance_to_leading_vehicle <
1134  cfg_.cost().effciency().leading_distance_th()) {
1135  decimal_t ego_blocked_by_leading_velocity =
1136  ego_fsagent.vehicle.state().velocity >
1137  leading_vehicle.state().velocity
1138  ? ego_fsagent.vehicle.state().velocity -
1139  leading_vehicle.state().velocity
1140  : 0.0;
1141  decimal_t leading_to_desired_velocity =
1142  leading_vehicle.state().velocity < desired_velocity_
1143  ? desired_velocity_ - leading_vehicle.state().velocity
1144  : 0.0;
1145  cost_tmp.efficiency.leading_to_desired_vel =
1146  std::max(cfg_.cost().effciency().min_distance_ratio(),
1147  distance_residual_ratio) *
1148  (cfg_.cost().effciency().ego_speed_blocked_by_leading_unit_cost() *
1149  ego_blocked_by_leading_velocity +
1150  cfg_.cost()
1151  .effciency()
1152  .leading_speed_blocked_desired_vel_unit_cost() *
1153  leading_to_desired_velocity);
1154  }
1155  }
1156 
1157  // * safety
1158  for (const auto& surround_traj : surround_trajs) {
1159  decimal_t safety_cost = 0.0;
1160  bool is_safe = true;
1161  int risky_id = 0;
1162  EvaluateSafetyStatus(ego_traj, surround_traj.second, &safety_cost, &is_safe,
1163  &risky_id);
1164  if (!is_safe) {
1165  risky_ids->insert(risky_id);
1166  *is_risky = true;
1167  }
1168  cost_tmp.safety.rss += safety_cost;
1169  }
1170 
1171  if (cfg_.cost().safety().occu_lane_enable()) {
1172  if (lc_info_.forbid_lane_change_left &&
1173  seq_lat_behavior == LateralBehavior::kLaneChangeLeft) {
1174  cost_tmp.safety.occu_lane =
1175  ego_velocity * cfg_.cost().safety().occu_lane_unit_cost();
1176  } else if (lc_info_.forbid_lane_change_right &&
1177  seq_lat_behavior == LateralBehavior::kLaneChangeRight) {
1178  cost_tmp.safety.occu_lane =
1179  ego_velocity * cfg_.cost().safety().occu_lane_unit_cost();
1180  }
1181  }
1182 
1183  // * navigation
1184  if (seq_lat_behavior == LateralBehavior::kLaneChangeLeft ||
1185  seq_lat_behavior == LateralBehavior::kLaneChangeRight) {
1186  if (is_cancel_behavior) {
1187  cost_tmp.navigation.lane_change_preference =
1188  std::max(cfg_.cost().navigation().lane_change_unit_cost_vel_lb(),
1189  ego_velocity) *
1190  cfg_.cost().user().cancel_operation_unit_cost();
1191  } else {
1192  cost_tmp.navigation.lane_change_preference =
1193  std::max(cfg_.cost().navigation().lane_change_unit_cost_vel_lb(),
1194  ego_velocity) *
1195  (seq_lat_behavior == LateralBehavior::kLaneChangeLeft
1196  ? cfg_.cost().navigation().lane_change_left_unit_cost()
1197  : cfg_.cost().navigation().lane_change_right_unit_cost());
1198  if (lc_info_.recommend_lc_left &&
1199  seq_lat_behavior == LateralBehavior::kLaneChangeLeft) {
1200  cost_tmp.navigation.lane_change_preference =
1201  -std::max(cfg_.cost().navigation().lane_change_unit_cost_vel_lb(),
1202  ego_velocity) *
1203  cfg_.cost().navigation().lane_change_left_recommendation_reward();
1204  if (action.lat != DcpLatAction::kLaneChangeLeft) {
1205  cost_tmp.navigation.lane_change_preference +=
1206  std::max(cfg_.cost().navigation().lane_change_unit_cost_vel_lb(),
1207  ego_velocity) *
1208  cfg_.cost().user().late_operate_unit_cost();
1209  }
1210  } else if (lc_info_.recommend_lc_right &&
1211  seq_lat_behavior == LateralBehavior::kLaneChangeRight) {
1212  cost_tmp.navigation.lane_change_preference =
1213  -std::max(cfg_.cost().navigation().lane_change_unit_cost_vel_lb(),
1214  ego_velocity) *
1215  cfg_.cost().navigation().lane_change_right_recommendation_reward();
1216  if (action.lat != DcpLatAction::kLaneChangeRight) {
1217  cost_tmp.navigation.lane_change_preference +=
1218  std::max(cfg_.cost().navigation().lane_change_unit_cost_vel_lb(),
1219  ego_velocity) *
1220  cfg_.cost().user().late_operate_unit_cost();

1223  }
