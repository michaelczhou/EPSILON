49 void EudmPlannerServer::JoyCallback(const sensor_msgs::Joy::ConstPtr &msg) {
50  int msg_id;
51  if (std::string("").compare(msg->header.frame_id) == 0) {
52  msg_id = 0;
53  } else {
54  msg_id = std::stoi(msg->header.frame_id);
55  }
56  if (msg_id != ego_id_) return;
57  // ~ buttons[2] --> 1 --> lcl
58  // ~ buttons[1] --> 1 --> lcr
59  // ~ buttons[3] --> 1 --> +1m/s
60  // ~ buttons[0] --> 1 --> -1m/s
61  if (msg->buttons[0] == 0 && msg->buttons[1] == 0 && msg->buttons[2] == 0 &&
62  msg->buttons[3] == 0 && msg->buttons[4] == 0 && msg->buttons[5] == 0 &&
63  msg->buttons[6] == 0)
64  return;
65 
66  if (msg->buttons[2] == 1) {
67  if (task_.user_perferred_behavior != -1) {
68  task_.user_perferred_behavior = -1;
69  } else {
70  task_.user_perferred_behavior = 0;
71  }
72  } else if (msg->buttons[1] == 1) {
73  if (task_.user_perferred_behavior != 1) {
74  task_.user_perferred_behavior = 1;
75  } else {
76  task_.user_perferred_behavior = 0;
77  }
78  } else if (msg->buttons[3] == 1) {
79  task_.user_desired_vel = task_.user_desired_vel + 1.0;
80  } else if (msg->buttons[0] == 1) {
81  task_.user_desired_vel = std::max(task_.user_desired_vel - 1.0, 0.0);
82  } else if (msg->buttons[4] == 1) {
83  task_.lc_info.forbid_lane_change_left = !task_.lc_info.forbid_lane_change_left;
84  } else if (msg->buttons[5] == 1) {
85  task_.lc_info.forbid_lane_change_right = !task_.lc_info.forbid_lane_change_right;
86  } else if (msg->buttons[6] == 1) {
87  task_.is_under_ctrl = !task_.is_under_ctrl;
88  }
89 }

96 void EudmPlannerServer::MainThread() {
97  using namespace std::chrono;
98  system_clock::time_point current_start_time{system_clock::now()};
99  system_clock::time_point next_start_time{current_start_time};
100  const milliseconds interval{static_cast<int>(1000.0 / work_rate_)};
101  while (true) {
102  current_start_time = system_clock::now();
103  next_start_time = current_start_time + interval;
104  PlanCycleCallback();
105  std::this_thread::sleep_until(next_start_time);
106  }
107 }

109 void EudmPlannerServer::PlanCycleCallback() {
110  if (p_input_smm_buff_ == nullptr) return;
111 
112  bool has_updated_map = false;
113  while (p_input_smm_buff_->try_dequeue(smm_)) {
114  has_updated_map = true;
115  }
116 
117  if (!has_updated_map) return;
118 
119  auto map_ptr =
120  std::make_shared<semantic_map_manager::SemanticMapManager>(smm_);
121 
122  decimal_t replan_duration = 1.0 / work_rate_;
123  double stamp =
124  std::floor(smm_.time_stamp() / replan_duration) * replan_duration;
125 
126  if (bp_manager_.Run(stamp, map_ptr, task_) == kSuccess) {
127  common::SemanticBehavior behavior;
128  bp_manager_.ConstructBehavior(&behavior);
129  smm_.set_ego_behavior(behavior);
130  }
131 
132  if (has_callback_binded_) {
133  private_callback_fn_(smm_);
134  }
135 
136  PublishData();
137 }
