41 ErrorType DcpTree::GenerateActionScript() {
42  action_script_.clear();
43  std::vector<DcpAction> ongoing_action_seq;
44  for (int lon = 0; lon < static_cast<int>(DcpLonAction::MAX_COUNT); lon++) {
45  ongoing_action_seq.clear();
46  ongoing_action_seq.push_back(
47  DcpAction(DcpLonAction(lon), ongoing_action_.lat, ongoing_action_.t));
48 
49  for (int h = 1; h < tree_height_; ++h) {
50  for (int lat = 0; lat < static_cast<int>(DcpLatAction::MAX_COUNT);
51  lat++) {
52  if (lat != static_cast<int>(ongoing_action_.lat)) {
53  auto actions = AppendActionSequence(
54  ongoing_action_seq,
55  DcpAction(DcpLonAction(lon), DcpLatAction(lat), layer_time_),
56  tree_height_ - h);
57  action_script_.push_back(actions);
58  }
59  }
60  ongoing_action_seq.push_back(
61  DcpAction(DcpLonAction(lon), ongoing_action_.lat, layer_time_));
62  }
63  action_script_.push_back(ongoing_action_seq);
64  }
65  // override the last layer time
66  for (auto& action_seq : action_script_) {
67  action_seq.back().t = last_layer_time_;
68  }
69  return kSuccess;
70 }
