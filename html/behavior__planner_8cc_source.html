132  const decimal_t previous_desired_vel, LateralBehavior* mpdm_behavior,
133  decimal_t* mpdm_desired_velocity) {
134  // * get relevant information
135  common::SemanticVehicleSet semantic_vehicle_set;
136  if (map_itf_->GetKeySemanticVehicles(&semantic_vehicle_set) != kSuccess) {
137  printf("[MPDM]fail to get key vehicles.\n");
138  return kWrongStatus;
139  }
140 
141  common::Vehicle ego_vehicle;
142  if (map_itf_->GetEgoVehicle(&ego_vehicle) != kSuccess) {
143  printf("[MPDM]fail to get ego vehicle.\n");
144  return kWrongStatus;
145  }
146 
147  std::cout << "[BehaviorPlanner]" << ego_vehicle.id()
148  << "\tsemantic_vehicle_set num:"
149  << semantic_vehicle_set.semantic_vehicles.size() << std::endl;
150 
151  // * clean the states
152  forward_trajs_.clear();
153  forward_behaviors_.clear();
154  surround_trajs_.clear();
155 
156  // * collect potential behaviors
157  std::vector<LateralBehavior> potential_behaviors{
158  common::LateralBehavior::kLaneKeeping};
159  if (!potential_lcl_lane_ids_.empty())
160  potential_behaviors.push_back(common::LateralBehavior::kLaneChangeLeft);
161  if (!potential_lcr_lane_ids_.empty())
162  potential_behaviors.push_back(common::LateralBehavior::kLaneChangeRight);
163 
164  // * construct <vehicle, ref_lane> pairs
165  const decimal_t max_backward_len = 10.0;
166  for (auto it = semantic_vehicle_set.semantic_vehicles.begin();
167  it != semantic_vehicle_set.semantic_vehicles.end(); ++it) {
168  common::LateralBehavior lat_behavior;
169  if (map_itf_->GetPredictedBehavior(it->second.vehicle.id(),
170  &lat_behavior) != kSuccess) {
171  lat_behavior = common::LateralBehavior::kLaneKeeping;
172  }
173  decimal_t forward_lane_len =
174  std::max(it->second.vehicle.state().velocity * 10.0, 50.0);
175  common::Lane ref_lane;
176  if (map_itf_->GetRefLaneForStateByBehavior(
177  it->second.vehicle.state(), std::vector<int>(), lat_behavior,
178  forward_lane_len, max_backward_len, false, &ref_lane) == kSuccess) {
179  it->second.lane = ref_lane;
180  }
181  }
182 
183  // * forward simulation
184  std::vector<LateralBehavior> valid_behaviors;
185  vec_E<vec_E<common::Vehicle>> valid_forward_trajs;
186  vec_E<std::unordered_map<int, vec_E<common::Vehicle>>> valid_surround_trajs;
187  int num_available_behaviors = static_cast<int>(potential_behaviors.size());
188 
189  // TicToc timer;
190  for (int i = 0; i < num_available_behaviors; i++) {
191  vec_E<common::Vehicle> traj;
192  std::unordered_map<int, vec_E<common::Vehicle>> sur_trajs;
193  if (SimulateEgoBehavior(ego_vehicle, potential_behaviors[i],
194  semantic_vehicle_set, &traj,
195  &sur_trajs) != kSuccess) {
196  printf("[MPDM]fail to sim %d forward.\n",
197  static_cast<int>(potential_behaviors[i]));
198  continue;
199  }
200  valid_behaviors.push_back(potential_behaviors[i]);
201  valid_forward_trajs.push_back(traj);
202  valid_surround_trajs.push_back(sur_trajs);
203  }
204  // printf("[Summary]Time in simulate all the behaviors: %lf ms.\n",
205  // timer.toc());
206 
207  // ! cache
208  forward_behaviors_ = valid_behaviors;
209  forward_trajs_ = valid_forward_trajs;
210  surround_trajs_ = valid_surround_trajs;
211 
212  // * judge forward trajs
213  int num_valid_behaviors = static_cast<int>(valid_behaviors.size());
214  if (num_valid_behaviors < 1) {
215  printf("[MPDM]No valid behaviors.\n");
216  return kWrongStatus;
217  }
218 
219  decimal_t winner_score, winner_desired_vel;
220  LateralBehavior winner_behavior;
221  vec_E<common::Vehicle> winner_forward_traj;
222  if (EvaluateMultiPolicyTrajs(valid_behaviors, valid_forward_trajs,
223  valid_surround_trajs, &winner_behavior,
224  &winner_forward_traj, &winner_score,

226  printf("[MPDM]fail to evaluate multiple policy trajs.\n");
227  return kWrongStatus;
228  }
229  // printf("[Stuck]id: %d choose behavior %d with cost: %lf.\n",
230  // ego_vehicle.id(),
231  // static_cast<int>(winner_behavior), winner_score);
232 
233  // * output

235  if (fabs(winner_desired_vel - ego_vehicle.state().velocity) >

237  if (winner_desired_vel > ego_vehicle.state().velocity) {
238  winner_desired_vel = ego_vehicle.state().velocity + max_vel_cmd_gap;
239  } else {
240  winner_desired_vel = ego_vehicle.state().velocity - max_vel_cmd_gap;
241  }
242  }
243 
244  if (lock_to_hmi_) {
245  auto it = std::find(forward_behaviors_.begin(), forward_behaviors_.end(),
246  hmi_behavior_);
247  if (it != forward_behaviors_.end()) {
248  *mpdm_behavior = hmi_behavior_;
249  } else {
250  *mpdm_behavior = winner_behavior;
251  }
252  } else {
253  *mpdm_behavior = winner_behavior;
254  }
255  *mpdm_desired_velocity = winner_desired_vel;
256  return kSuccess;
257 }

259 ErrorType BehaviorPlanner::OpenloopSimForward(
260  const common::SemanticVehicle& ego_semantic_vehicle,
261  const common::SemanticVehicleSet& agent_vehicles,
262  vec_E<common::Vehicle>* traj,
263  std::unordered_map<int, vec_E<common::Vehicle>>* surround_trajs) {
264  traj->clear();
265  traj->push_back(ego_semantic_vehicle.vehicle);
266  surround_trajs->clear();
267  for (const auto v : agent_vehicles.semantic_vehicles) {
268  surround_trajs->insert(std::pair<int, vec_E<common::Vehicle>>(
269  v.first, vec_E<common::Vehicle>()));
270  surround_trajs->at(v.first).push_back(v.second.vehicle);
271  }
272 
273  int num_steps_forward = static_cast<int>(sim_horizon_ / sim_resolution_);
274  common::Vehicle cur_ego_vehicle = ego_semantic_vehicle.vehicle;
275  common::SemanticVehicleSet semantic_vehicle_set_tmp = agent_vehicles;
276  common::State ego_state;
277  for (int i = 0; i < num_steps_forward; i++) {
278  sim_param_.idm_param.kDesiredVelocity = reference_desired_velocity_;
279  if (planning::OnLaneForwardSimulation::PropagateOnce(
280  common::StateTransformer(ego_semantic_vehicle.lane),
281  cur_ego_vehicle, common::Vehicle(), sim_resolution_, sim_param_,
282  &ego_state) != kSuccess) {
283  return kWrongStatus;
284  }
285  std::unordered_map<int, State> state_cache;
286  for (auto& v : semantic_vehicle_set_tmp.semantic_vehicles) {
287  decimal_t desired_vel =
288  agent_vehicles.semantic_vehicles.at(v.first).vehicle.state().velocity;
289  sim_param_.idm_param.kDesiredVelocity = desired_vel;
290  common::State agent_state;
291  if (planning::OnLaneForwardSimulation::PropagateOnce(
292  common::StateTransformer(v.second.lane), v.second.vehicle,
293  common::Vehicle(), sim_resolution_, sim_param_,
294  &agent_state) != kSuccess) {
295  return kWrongStatus;
296  }
297  state_cache.insert(std::make_pair(v.first, agent_state));
298  }
299 
300  bool is_collision = false;
301  map_itf_->CheckIfCollision(ego_semantic_vehicle.vehicle.param(), ego_state,
302  &is_collision);
303  if (is_collision) return kWrongStatus;
304 
305  // * update and trace
306  cur_ego_vehicle.set_state(ego_state);
307  for (auto& s : state_cache) {
308  semantic_vehicle_set_tmp.semantic_vehicles.at(s.first).vehicle.set_state(
309  s.second);
310  surround_trajs->at(s.first).push_back(
311  semantic_vehicle_set_tmp.semantic_vehicles.at(s.first).vehicle);
312  }
313  traj->push_back(cur_ego_vehicle);
314  }
315  return kSuccess;
316 }

401 ErrorType BehaviorPlanner::EvaluateSafetyCost(
402  const vec_E<common::Vehicle>& traj_a, const vec_E<common::Vehicle>& traj_b,
403  decimal_t* cost) {
404  if (traj_a.size() != traj_b.size()) {
405  return kWrongStatus;
406  }
407  int num_states = static_cast<int>(traj_a.size());
408  decimal_t cost_tmp = 0.0;
409  for (int i = 0; i < num_states; i++) {
410  common::Vehicle inflated_a, inflated_b;
411  common::SemanticsUtils::InflateVehicleBySize(traj_a[i], 1.0, 1.0,
412  &inflated_a);
413  common::SemanticsUtils::InflateVehicleBySize(traj_b[i], 1.0, 1.0,
414  &inflated_b);
415  bool is_collision = false;
416  map_itf_->CheckCollisionUsingState(inflated_a.param(), inflated_a.state(),
417  inflated_b.param(), inflated_b.state(),
418  &is_collision);
419  if (is_collision) {
420  cost_tmp +=
421  0.01 * fabs(traj_a[i].state().velocity - traj_b[i].state().velocity) *
422  0.5;
423  }
424  }
425  *cost = cost_tmp;
426  return kSuccess;

429 ErrorType BehaviorPlanner::EvaluateSinglePolicyTraj(
430  const LateralBehavior& behavior, const vec_E<common::Vehicle>& forward_traj,
431  const std::unordered_map<int, vec_E<common::Vehicle>>& surround_traj,
432  decimal_t* score, decimal_t* desired_vel) {
433  // prepare
434  common::VehicleSet vehicle_set;
435  for (auto it = surround_traj.begin(); it != surround_traj.end(); ++it) {
436  if (!it->second.empty()) {
437  vehicle_set.vehicles.insert(std::make_pair(it->first, it->second.back()));
438  }
439  }
440  // * efficiency
441  common::Vehicle ego_vehicle_terminal = forward_traj.back();
442  decimal_t cost_efficiency_ego_to_desired_vel =
443  fabs(ego_vehicle_terminal.state().velocity -
444  reference_desired_velocity_) /
445  10.0;
446  common::Vehicle leading_vehicle;
447  common::Lane ego_ref_lane;
448  const decimal_t max_backward_len = 10.0;
449  decimal_t forward_lane_len =
450  std::max(ego_vehicle_terminal.state().velocity * 10.0, 50.0);
451  if (map_itf_->GetRefLaneForStateByBehavior(
452  ego_vehicle_terminal.state(), p_route_planner_->navi_path(),
453  common::LateralBehavior::kLaneKeeping, forward_lane_len,
454  max_backward_len, false, &ego_ref_lane) != kSuccess) {
455  printf("fail to get ego ref lane duration evaluation for behavior %d.\n",
456  static_cast<int>(behavior));
457  }
458 
459  decimal_t cost_efficiency_leading_to_desired_vel = 0.0;
460  decimal_t distance_residual_ratio = 0.0;
461  const decimal_t lat_range = 2.2;
462  if (map_itf_->GetLeadingVehicleOnLane(
463  ego_ref_lane, ego_vehicle_terminal.state(), vehicle_set, lat_range,
464  &leading_vehicle, &distance_residual_ratio) == kSuccess) {
465  decimal_t distance_to_leading_vehicle =
466  (leading_vehicle.state().vec_position -
467  ego_vehicle_terminal.state().vec_position)
468  .norm();
469 
470  if (ego_vehicle_terminal.state().velocity < reference_desired_velocity_ &&
471  leading_vehicle.state().velocity < reference_desired_velocity_ &&
472  distance_to_leading_vehicle < 100.0) {
473  cost_efficiency_leading_to_desired_vel =
474  1.5 * distance_residual_ratio *
475  fabs(ego_vehicle_terminal.state().velocity -
476  reference_desired_velocity_) /
477  std::max(2.0, distance_to_leading_vehicle);
478  }
479  }
480  decimal_t cost_efficiency = 0.5 * (cost_efficiency_ego_to_desired_vel +
481  cost_efficiency_leading_to_desired_vel);
482 
483  // * safety
484  decimal_t cost_safety = 0.0;
485  for (auto& traj : surround_traj) {
486  decimal_t safety_tmp = 0.0;
487  EvaluateSafetyCost(forward_traj, traj.second, &safety_tmp);
488  cost_safety += safety_tmp;
489  }
490 
491  // * action
492  decimal_t cost_action = 0.0;
493  if (behavior != common::LateralBehavior::kLaneKeeping) {
494  cost_action += 0.5;
495  }
496  *score = cost_action + cost_safety + cost_efficiency;
497  printf(
498  "[CostDebug]behaivor %d: (action %lf, safety %lf, efficiency ego %lf, "
499  "leading %lf).\n",
500  static_cast<int>(behavior), cost_action, cost_safety,
501  cost_efficiency_ego_to_desired_vel,
502  cost_efficiency_leading_to_desired_vel);
503  GetDesiredVelocityOfTrajectory(forward_traj, desired_vel);
504  return kSuccess;
505 }

507 ErrorType BehaviorPlanner::GetDesiredVelocityOfTrajectory(
508  const vec_E<common::Vehicle> vehicle_vec, decimal_t* vel) {
