35  fpu( const double beta = 1.0 ) : m_beta( beta ) { }

37  // system function defining the ODE
38  void operator()( const container_type &q , container_type &dpdt ) const
39  {
40  size_t n = q.size();
41  double tmp = q[0] - 0.0;
42  double tmp2 = tmp + m_beta * tmp * tmp * tmp;
43  dpdt[0] = -tmp2;
44  for( size_t i=0 ; i<n-1 ; ++i )
45  {
46  tmp = q[i+1] - q[i];
47  tmp2 = tmp + m_beta * tmp * tmp * tmp;
48  dpdt[i] += tmp2;
49  dpdt[i+1] = -tmp2;
50  }
51  tmp = - q[n-1];
52  tmp2 = tmp + m_beta * tmp * tmp * tmp;
53  dpdt[n-1] += tmp2;
54  }
55 
56  // calculates the energy of the system
57  double energy( const container_type &q , const container_type &p ) const
58  {
59  // ...
60  //<-
61  double energy = 0.0;
62  size_t n = q.size();
63 
64  double tmp = q[0];
65  energy += 0.5 * tmp * tmp + 0.25 * m_beta * tmp * tmp * tmp * tmp;
66  for( size_t i=0 ; i<n-1 ; ++i )
67  {
68  tmp = q[i+1] - q[i];
69  energy += 0.5 * ( p[i] * p[i] + tmp * tmp ) + 0.25 * m_beta * tmp * tmp * tmp * tmp;
70  }
71  energy += 0.5 * p[n-1] * p[n-1];
72  tmp = q[n-1];
73  energy += 0.5 * tmp * tmp + 0.25 * m_beta * tmp * tmp * tmp * tmp;
74 
75  return energy;
76  //->
77  }

79  // calculates the local energy of the system
80  void local_energy( const container_type &q , const container_type &p , container_type &e ) const
81  {
82  // ...
83  //<-
84  size_t n = q.size();
85  double tmp = q[0];
86  double tmp2 = 0.5 * tmp * tmp + 0.25 * m_beta * tmp * tmp * tmp * tmp;
87  e[0] = tmp2;
88  for( size_t i=0 ; i<n-1 ; ++i )
89  {
90  tmp = q[i+1] - q[i];
91  tmp2 = 0.25 * tmp * tmp + 0.125 * m_beta * tmp * tmp * tmp * tmp;
92  e[i] += 0.5 * p[i] * p[i] + tmp2 ;
93  e[i+1] = tmp2;
94  }
95  tmp = q[n-1];
96  tmp2 = 0.5 * tmp * tmp + 0.25 * m_beta * tmp * tmp * tmp * tmp;
97  e[n-1] += 0.5 * p[n-1] * p[n-1] + tmp2;
98  //->
99  }
