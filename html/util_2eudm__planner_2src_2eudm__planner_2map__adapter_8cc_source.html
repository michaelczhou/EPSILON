146 bool EudmPlannerMapAdapter::IsLaneConsistent(const int lane_id_old,
147  const int lane_id_new) {
148  if (lane_id_new == lane_id_old) return true;
149  auto p_semantic_lane_set = map_->semantic_lane_set_cptr();
150  const int max_nodes_expanded = 20;
151  int expanded_nodes = 1;
152  std::set<int> visited_set;
153  std::list<int> queue;
154  visited_set.insert(lane_id_old);
155  queue.push_back(lane_id_old);
156  //* BFS
157  int cur_id;
158  while (!queue.empty() && expanded_nodes < max_nodes_expanded) {
159  cur_id = queue.front();
160  queue.pop_front();
161  expanded_nodes++;
162 
163  std::vector<int> child_ids;
164  auto it = p_semantic_lane_set->semantic_lanes.find(cur_id);
165  if (it == p_semantic_lane_set->semantic_lanes.end()) {
166  continue;
167  } else {
168  child_ids = it->second.child_id;
169  }
170  if (child_ids.empty()) continue;
171  for (auto &id : child_ids) {
172  if (visited_set.count(id) == 0) {
173  visited_set.insert(id);
174  queue.push_back(id);
175  }
176  }
177  }
178 
179  if (visited_set.find(lane_id_new) != visited_set.end()) return true;
180  return false;
181 }
