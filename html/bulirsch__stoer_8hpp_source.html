186  template< class System , class StateIn , class DerivIn , class StateOut >
187  controlled_step_result try_step( System system , const StateIn &in , const DerivIn &dxdt , time_type &t , StateOut &out , time_type &dt )
188  {
189  BOOST_USING_STD_MIN();
190  BOOST_USING_STD_MAX();
191 
192  static const value_type val1( 1.0 );
193 
194  typename odeint::unwrap_reference< System >::type &sys = system;
195  if( m_resizer.adjust_size( in , detail::bind( &controlled_error_bs_type::template resize_impl< StateIn > , detail::ref( *this ) , detail::_1 ) ) )
196  {
197  reset(); // system resized -> reset
198  }
199 
200  if( dt != m_dt_last )
201  {
202  reset(); // step size changed from outside -> reset
203  }
204 
205  bool reject( true );
206 
207  time_vector h_opt( m_k_max+1 );
208  inv_time_vector work( m_k_max+1 );
209 
210  time_type new_h = dt;
211 
212  /* m_current_k_opt is the estimated current optimal stage number */
213  for( size_t k = 0 ; k <= m_current_k_opt+1 ; k++ )
214  {
215  /* the stage counts are stored in m_interval_sequence */
216  m_midpoint.set_steps( m_interval_sequence[k] );
217  if( k == 0 )
218  {
219  m_midpoint.do_step( sys , in , dxdt , t , out , dt );
220  /* the first step, nothing more to do */
221  }
222  else
223  {
224  m_midpoint.do_step( sys , in , dxdt , t , m_table[k-1].m_v , dt );
225  extrapolate( k , m_table , m_coeff , out );
226  // get error estimate
227  m_algebra.for_each3( m_err.m_v , out , m_table[0].m_v ,
228  typename operations_type::template scale_sum2< value_type , value_type >( val1 , -val1 ) );
229  const value_type error = m_error_checker.error( m_algebra , in , dxdt , m_err.m_v , dt );
230  h_opt[k] = calc_h_opt( dt , error , k );
231  work[k] = static_cast<value_type>( m_cost[k] ) / h_opt[k];
232 
233  if( (k == m_current_k_opt-1) || m_first )
234  { // convergence before k_opt ?
235  if( error < 1.0 )
236  {
237  //convergence
238  reject = false;
239  if( (work[k] < KFAC2*work[k-1]) || (m_current_k_opt <= 2) )
240  {
241  // leave order as is (except we were in first round)
242  m_current_k_opt = min BOOST_PREVENT_MACRO_SUBSTITUTION( static_cast<int>(m_k_max)-1 , max BOOST_PREVENT_MACRO_SUBSTITUTION( 2 , static_cast<int>(k)+1 ) );
243  new_h = h_opt[k];
244  new_h *= static_cast<value_type>( m_cost[k+1] ) / static_cast<value_type>( m_cost[k] );
245  } else {
246  m_current_k_opt = min BOOST_PREVENT_MACRO_SUBSTITUTION( static_cast<int>(m_k_max)-1 , max BOOST_PREVENT_MACRO_SUBSTITUTION( 2 , static_cast<int>(k) ) );
247  new_h = h_opt[k];
248  }
249  break;
250  }
251  else if( should_reject( error , k ) && !m_first )
252  {
253  reject = true;
254  new_h = h_opt[k];
255  break;
256  }
257  }
258  if( k == m_current_k_opt )
259  { // convergence at k_opt ?
260  if( error < 1.0 )
261  {
262  //convergence
263  reject = false;
264  if( (work[k-1] < KFAC2*work[k]) )
265  {
266  m_current_k_opt = max BOOST_PREVENT_MACRO_SUBSTITUTION( 2 , static_cast<int>(m_current_k_opt)-1 );
267  new_h = h_opt[m_current_k_opt];
268  }
269  else if( (work[k] < KFAC2*work[k-1]) && !m_last_step_rejected )
270  {
271  m_current_k_opt = min BOOST_PREVENT_MACRO_SUBSTITUTION( static_cast<int>(m_k_max-1) , static_cast<int>(m_current_k_opt)+1 );
272  new_h = h_opt[k];
273  new_h *= m_cost[m_current_k_opt]/m_cost[k];
274  } else
275  new_h = h_opt[m_current_k_opt];
276  break;
277  }
278  else if( should_reject( error , k ) )
279  {
280  reject = true;
281  new_h = h_opt[m_current_k_opt];
282  break;
283  }
284  }
285  if( k == m_current_k_opt+1 )
286  { // convergence at k_opt+1 ?
287  if( error < 1.0 )
288  { //convergence
289  reject = false;
290  if( work[k-2] < KFAC2*work[k-1] )
291  m_current_k_opt = max BOOST_PREVENT_MACRO_SUBSTITUTION( 2 , static_cast<int>(m_current_k_opt)-1 );
292  if( (work[k] < KFAC2*work[m_current_k_opt]) && !m_last_step_rejected )
293  m_current_k_opt = min BOOST_PREVENT_MACRO_SUBSTITUTION( static_cast<int>(m_k_max)-1 , static_cast<int>(k) );
294  new_h = h_opt[m_current_k_opt];
295  } else
296  {
297  reject = true;
298  new_h = h_opt[m_current_k_opt];
299  }
300  break;
301  }
302  }
303  }
304 
305  if( !reject )
306  {
307  t += dt;
308  }
309 
310  if( !m_last_step_rejected || boost::numeric::odeint::detail::less_with_sign(new_h, dt, dt) )
311  {
312  m_dt_last = new_h;
313  dt = new_h;
314  }
315 
316  m_last_step_rejected = reject;
317  m_first = false;
318 
319  if( reject )
320  return fail;
321  else
322  return success;
323  }

380  template< class StateInOut >
381  void extrapolate( size_t k , state_table_type &table , const value_matrix &coeff , StateInOut &xest )
382  /* polynomial extrapolation, see http://www.nr.com/webnotes/nr3web21.pdf
383  uses the obtained intermediate results to extrapolate to dt->0
384  */
385  {
386  static const value_type val1 = static_cast< value_type >( 1.0 );
387  for( int j=k-1 ; j>0 ; --j )
388  {
389  m_algebra.for_each3( table[j-1].m_v , table[j].m_v , table[j-1].m_v ,
390  typename operations_type::template scale_sum2< value_type , value_type >( val1 + coeff[k][j] , -coeff[k][j] ) );
391  }
392  m_algebra.for_each3( xest , table[0].m_v , xest ,
393  typename operations_type::template scale_sum2< value_type , value_type >( val1 + coeff[k][0] , -coeff[k][0]) );
394  }

396  time_type calc_h_opt( time_type h , value_type error , size_t k ) const
397  /* calculates the optimal step size for a given error and stage number */
398  {
399  BOOST_USING_STD_MIN();
400  BOOST_USING_STD_MAX();
401  using std::pow;
402  value_type expo( 1.0/(2*k+1) );
403  value_type facmin = pow BOOST_PREVENT_MACRO_SUBSTITUTION( STEPFAC3 , expo );
404  value_type fac;
405  if (error == 0.0)
406  fac=1.0/facmin;
407  else
408  {
409  fac = STEPFAC2 / pow BOOST_PREVENT_MACRO_SUBSTITUTION( error / STEPFAC1 , expo );
410  fac = max BOOST_PREVENT_MACRO_SUBSTITUTION( facmin/STEPFAC4 , min BOOST_PREVENT_MACRO_SUBSTITUTION( 1.0/facmin , fac ) );
411  }
412  return h*fac;
413  }

415  controlled_step_result set_k_opt( size_t k , const inv_time_vector &work , const time_vector &h_opt , time_type &dt )
416  /* calculates the optimal stage number */
417  {
418  if( k == 1 )
419  {
420  m_current_k_opt = 2;
421  return success;
422  }
423  if( (work[k-1] < KFAC1*work[k]) || (k == m_k_max) )
424  { // order decrease
425  m_current_k_opt = k-1;
426  dt = h_opt[ m_current_k_opt ];
427  return success;
428  }
429  else if( (work[k] < KFAC2*work[k-1]) || m_last_step_rejected || (k == m_k_max-1) )
430  { // same order - also do this if last step got rejected
431  m_current_k_opt = k;
432  dt = h_opt[ m_current_k_opt ];
433  return success;
434  }
435  else
436  { // order increase - only if last step was not rejected
437  m_current_k_opt = k+1;
438  dt = h_opt[ m_current_k_opt-1 ] * m_cost[ m_current_k_opt ] / m_cost[ m_current_k_opt-1 ] ;
439  return success;
440  }

450  bool should_reject( value_type error , size_t k ) const
451  {
452  if( k == m_current_k_opt-1 )
453  {
454  const value_type d = m_interval_sequence[m_current_k_opt] * m_interval_sequence[m_current_k_opt+1] /
455  (m_interval_sequence[0]*m_interval_sequence[0]);
456  //step will fail, criterion 17.3.17 in NR
457  return ( error > d*d );
458  }
459  else if( k == m_current_k_opt )
460  {
461  const value_type d = m_interval_sequence[m_current_k_opt] / m_interval_sequence[0];
462  return ( error > d*d );
463  } else
464  return error > 1.0;
465  }
